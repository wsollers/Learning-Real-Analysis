\textbf{Claim.}
Let $A,B \subseteq \mathbb{R}$ with $A\subseteq B$. Assuming the relevant bounds exist,
\[
\sup A \le \sup B
\qquad\text{and}\qquad
\inf B \le \inf A.
\]

\textbf{Proof.}
% Stub:
% Sup: any upper bound of B is an upper bound of A, so the least upper bound of A
% cannot exceed the least upper bound of B.
% Inf: any lower bound of A is a lower bound of B, so greatest lower bound of B
% cannot exceed greatest lower bound of A.

\qed
