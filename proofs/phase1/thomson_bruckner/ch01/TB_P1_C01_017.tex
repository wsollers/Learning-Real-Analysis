\textbf{Claim.}
Let $A \subseteq \mathbb{R}$ and $r>0$. Define $B=\{rx:x\in A\}$. Then
\[
\sup B=r\,\sup A
\qquad\text{and}\qquad
\inf B=r\,\inf A.
\]

\textbf{Proof.}
% Stub:
% For sup:
% (i) show r sup A is an upper bound of B using order scaling by positives;
% (ii) if U bounds B above, then U/r bounds A above; conclude r sup A <= U.
% For inf: analogous with lower bounds.

\qed
