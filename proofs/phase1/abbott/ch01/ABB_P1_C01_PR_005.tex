\noindent\textbf{Problem (Abbott 1.3.6).}
Given sets $A$ and $B$, define
\[
A+B=\{a+b : a\in A \text{ and } b\in B\}.
\]
Follow the steps below to prove that if $A$ and $B$ are nonempty and bounded above then
\[
\sup(A+B)=\sup A+\sup B.
\]
\begin{enumerate}[label=(\alph*)]
\item Let $s=\sup A$ and $t=\sup B$. Show that $s+t$ is an upper bound for $A+B$.
\item Now let $u$ be an arbitrary upper bound for $A+B$, and temporarily fix $a\in A$. Show that $t\le u-a$.
\item Finally, show that $\sup(A+B)=s+t$.
\item Construct another proof of this same fact using Lemma 1.3.8.
\end{enumerate}
