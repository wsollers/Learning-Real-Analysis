\begin{proof}
Suppose $e$ and $e'$ are both identity elements of $G$.

\noindent
\begin{tabular}{T S J}
\toprule
\textbf{Tag} & \multicolumn{1}{p{0.44\textwidth}}{\textbf{Step}} & \textbf{Justification} \\
\midrule
\addlinespace[4pt]

\tagDU
&
e' = e \star e'
&
Since $e$ is an identity, $e \star a = a$ for all $a \in G$.
Apply with $a = e'$.
\\[10pt]

\tagDU
&
e \star e' = e
&
Since $e'$ is an identity, $a \star e' = a$ for all $a \in G$.
Apply with $a = e$.
\\[10pt]

\tagAM
&
\therefore\; e' = e
&
Combining: $e' = e \star e' = e$. \hfill$\blacksquare$
\\[6pt]

\bottomrule
\end{tabular}
\end{proof}
