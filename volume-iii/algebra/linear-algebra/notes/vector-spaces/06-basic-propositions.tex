\section{Basic Propositions}

\begin{proposition}[Uniqueness of the Additive Identity]
\label{prop:unique-additive-identity}
Let $V$ be a vector space over a field $\mathbb{F}$.
The additive identity of $V$ is unique.
\end{proposition}

\begin{remark*}[Fully quantified form]
\[ \forall V\text{ vector space}: \exists!\,\mathbf{0}\in V,\;\forall v,\;v+\mathbf{0}=v. \]
\end{remark*}

\begin{remark*}[Flash]
\FlashQ{State uniqueness of the additive identity.}
\FlashA{A vector space has a unique additive identity $\mathbf{0}$.}
\end{remark*}


% ---------------------------------------------------------

\begin{proposition}[Uniqueness of the Additive Inverse]
\label{prop:unique-additive-inverse}
Let $V$ be a vector space over a field $\mathbb{F}$.
For each $v \in V$, the additive inverse of $v$ is unique.
\end{proposition}

\begin{remark*}[Fully quantified form]
\[ \forall v\in V,\;\exists!\,(-v)\in V:\;v+(-v)=\mathbf{0}. \]
\end{remark*}

\begin{remark*}[Flash]
\FlashQ{State uniqueness of the additive inverse.}
\FlashA{Every vector $v$ has a unique additive inverse $-v$.}
\end{remark*}

\begin{proposition}[Scalar Multiplication by Zero and Negation]
\label{prop:scalar-mult-zero}
Let $V$ be a vector space over a field $\mathbb{F}$,
let $v \in V$, and let $\alpha \in \mathbb{F}$. Then:
\begin{enumerate}
    \item $0_{\mathbb{F}} v = \mathbf{0}$,
    \item $\alpha \mathbf{0} = \mathbf{0}$,
    \item $(-1_{\mathbb{F}}) v = -v$.
\end{enumerate}
\end{proposition}

\begin{remark*}[Fully quantified form]
\[ 0\cdot v=\mathbf{0},\quad (-1)\cdot v=-v,\quad \lambda\cdot\mathbf{0}=\mathbf{0}. \]
\end{remark*}

\begin{remark*}[Flash]
\FlashQ{State scalar multiplication by zero and negation.}
\FlashA{$0v=\mathbf{0}$, $(-1)v=-v$, and $\lambda\mathbf{0}=\mathbf{0}$ for all $v\in V$, $\lambda\in\mathbb{F}$.}
\end{remark*}
