\section{Tuples and Lists}

\begin{definition}[$n$-tuple]
\label{def:n-tuple}
Let $n \in \mathbb{N}$ and let $X$ be a set.
An \emph{$n$-tuple} of elements of $X$ is a function
\[
f : \{1,2,\dots,n\} \to X.
\]
We write such a function as $(x_1,\dots,x_n)$, where $x_i := f(i)$ for each $i$.
\end{definition}

\begin{remark*}[Fully quantified form]
\[ (x_1,\ldots,x_n)\in X_1\times\cdots\times X_n. \]
\end{remark*}

\begin{remark*}[Flash]
\FlashQ{Define an $n$-tuple.}
\FlashA{An ordered list $(x_1,\ldots,x_n)$ with $x_i\in X_i$.}
\end{remark*}

\begin{definition}[Finite list]
\label{def:finite-list}
Let $X$ be a set.
A \emph{finite list} of elements of $X$ is a function
$f : \{1,2,\dots,n\} \to X$ for some $n \in \mathbb{N}$.
\end{definition}

\begin{remark*}[Fully quantified form]
\[ \text{A finite list of length }n\text{ is an element of }X^n. \]
\end{remark*}

\begin{remark*}[Flash]
\FlashQ{Define a finite list.}
\FlashA{A finite ordered sequence of elements from a set $X$.}
\end{remark*}

\begin{definition}[Set]
\label{def:set-la}
A \emph{set} is a collection of distinct elements with no inherent ordering.
\end{definition}

\begin{remark*}[Fully quantified form]
\[ \text{An unordered collection of distinct elements.} \]
\end{remark*}

\begin{remark*}[Flash]
\FlashQ{Define a set (linear algebra context).}
\FlashA{An unordered collection of distinct elements; order and repetition do not matter.}
\end{remark*}

\begin{example}
Consider the elements $1$ and $2$.
The set $\{1,2,2\}$ equals $\{1,2\}$ because sets do not record duplicates.
However, the list $(1,2,2)$ differs from $(1,2)$ because lists record both
order and repetition.
\end{example}

\begin{definition}[Coordinate]
\label{def:coordinate}
Let $\mathbb{F}$ be a field and let $v = (v_1,\dots,v_n) \in \mathbb{F}^n$.
The element $v_i$ is called the \emph{$i$th coordinate} of $v$.
\end{definition}

\begin{remark*}[Fully quantified form]
\[ \text{The }i\text{-th coordinate of }(x_1,\ldots,x_n)\text{ is }x_i. \]
\end{remark*}

\begin{remark*}[Flash]
\FlashQ{Define a coordinate.}
\FlashA{The $i$-th entry $x_i$ of an $n$-tuple $(x_1,\ldots,x_n)$.}
\end{remark*}
