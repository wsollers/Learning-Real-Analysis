\section*{Proof — LA-AXL-C01-1B-E01}

\noindent
\hyperlink{ws-LA-AXL-C01-1B-E01}{\textbf{← Back to worksheet}}

\vspace{0.5em}
\noindent
\textbf{Source.}
\srccite{AxlerLinearAlgebraDoneRight}{Chapter~1, Section~1B, Exercise~1}.

\vspace{0.75em}
\noindent
\textbf{Goal.}
Let $V$ be a vector space over a field $\mathbb{F}$.
Prove that $-(-v) = v$ for every $v \in V$.

\vspace{0.75em}
\noindent

\begin{proof}
Let $V$ be a vector space over a field $\mathbb{F}$ and let $v \in V$.
\begin{align*}
v + (-v) &= 0  &&\text{definition of additive inverse of } v \\
(-v) + (-(-v)) &= 0 &&\text{definition of additive inverse of } {-v} \\
(-v) + v &= 0 &&\text{commutativity of addition}
\end{align*}
Therefore $v$ and $-(-v)$ are both additive inverses of $-v$.
By the uniqueness of additive inverses in $V$, two elements satisfying
the same inverse condition must be equal. We verify this explicitly:
\begin{align*}
v + (-v) &= 0 = (-(-v)) + (-v) &&\text{definition of additive inverse} \\
v + (v + (-v)) &= (-(-v)) + (v + (-v)) &&\text{adding } v \text{ to both sides} \\
v + 0 &= (-(-v)) + 0 &&\text{definition of additive inverse of } v \\
v &= -(-v) &&\text{identity element axiom: } x + 0 = x
\end{align*}
Hence $-(-v) = v$ for every $v \in V$. \qedhere
\end{proof}
