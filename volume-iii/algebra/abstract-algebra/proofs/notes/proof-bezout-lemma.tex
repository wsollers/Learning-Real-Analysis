\subsection*{Bézout's Lemma}
\label{prf:bezout-lemma}

\begin{remark*}[Return]
\hyperref[cor:bezout-lemma]{$\leftarrow$ Back to Corollary (Bézout's Lemma) in Notes}
\end{remark*}

\begin{proof}
\Claim Let $a, b \in \mathbb{Z}$. Then $\gcd(a,b) = 1$ if and only if
$1 = ma + nb$ for some $m, n \in \mathbb{Z}$.

\medskip
\noindent$(\Rightarrow)$ \textit{Assume $\gcd(a,b) = 1$; we show $1 = ma + nb$
for some $m, n \in \mathbb{Z}$.}

\ByThm{Bézout's Identity}, there exist $m, n \in \mathbb{Z}$ such that
$\gcd(a,b) = ma + nb$.
Substituting $\gcd(a,b) = 1$ gives $1 = ma + nb$. \checkbox

\medskip
\noindent$(\Leftarrow)$ \textit{Assume $ma + nb = 1$ for some $m, n \in \mathbb{Z}$;
we show $\gcd(a,b) = 1$.}

\Let $d = \gcd(a,b)$.
\ByThm{Divisibility of Linear Combinations}, since $d \mid a$ and $d \mid b$,
we have $d \mid (ma + nb) = 1$.
The only positive divisor of $1$ is $1$ itself, so $d = 1$,
i.e.\ $\gcd(a,b) = 1$. \checkbox

\Hence $\gcd(a,b) = 1 \iff \exists\, m,n \in \mathbb{Z}: ma + nb = 1$. \AsReq
\end{proof}

\begin{remark*}[Proof shape]
Bidirectional proof. The forward direction is immediate from Bézout's Identity
with $d = 1$. The reverse direction uses the Divisibility of Linear
Combinations lemma: any common divisor of $a$ and $b$ must divide every linear
combination, and in particular must divide $1$, forcing it to equal $1$.
