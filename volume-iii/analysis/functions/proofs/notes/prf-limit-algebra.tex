\begin{theorem*}[Algebra of Function Limits]
Let $f, g : A \to \mathbb{R}$, let $c$ be a cluster point of $A$, and suppose
$\lim_{x\to c}f(x)=L$ and $\lim_{x\to c}g(x)=M$. Then:
\begin{enumerate}[label=(\roman*)]
  \item $\lim_{x\to c}(f+g)(x) = L+M$.
  \item $\lim_{x\to c}(f-g)(x) = L-M$.
  \item $\lim_{x\to c}(kf)(x) = kL$ for any $k\in\mathbb{R}$.
  \item $\lim_{x\to c}(fg)(x) = LM$.
  \item $\lim_{x\to c}(f/g)(x) = L/M$, provided $M\neq 0$.
\end{enumerate}
\end{theorem*}
