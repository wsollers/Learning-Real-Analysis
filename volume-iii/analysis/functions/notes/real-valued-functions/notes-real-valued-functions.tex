\section{Real-Valued Functions}
\label{sec:real-valued-functions}

\begin{tcolorbox}[title={Toolkit --- Real-Valued Functions}]
\begin{itemize}
  \item Pointwise operations: sum, difference, product, quotient,
    scalar multiple, absolute value, max, min. All purely algebraic
    — no limits involved. $fg$ is pointwise product, not composition.
  \item $\operatorname{BoundedOn}(f,A)$: $\exists B>0,\;\forall x\in A:
    |f(x)|\leq B$.
  \item $\operatorname{MonotoneFunction}(f,A)$: increasing or decreasing.
  \item Maximum attained at $x^*$: $f(x^*) = \sup f(A)$, $x^*\in A$.
    Supremum need not be attained.
\end{itemize}
\end{tcolorbox}

% ---------------------------------------------------------
% Definition — Pointwise Operations
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Pointwise Operations}]
\begin{definition}[Pointwise Arithmetic Operations]
\label{def:pointwise-ops-R}
Let $X \subseteq \mathbb{R}$, $f, g : X \to \mathbb{R}$.
\begin{align*}
  (f+g)(x)   &:= f(x)+g(x),  & (f-g)(x) &:= f(x)-g(x), \\
  (fg)(x)    &:= f(x)g(x),   & (cf)(x)  &:= cf(x), \\
  |f|(x)     &:= |f(x)|,     & \max(f,g)(x) &:= \max(f(x),g(x)).
\end{align*}
If $g(x)\neq 0$ for all $x\in X$: $(f/g)(x) := f(x)/g(x)$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  \forall x\in X : (f+g)(x) = f(x)+g(x).
\]
Each operation is pointwise: value at $x$ depends only on $f(x)$,
$g(x)$, not on nearby values.
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{Function}(f+g, X, \mathbb{R})
  \;\colonequiv\;
  \forall x\in X : (f+g)(x) = f(x)+g(x).
\]
\end{remark*}

\begin{remark*}[Product vs.\ composition]
$fg(x) = f(x)g(x)$ (pointwise product). $(f\circ g)(x) = f(g(x))$
(composition, requires $g:X\to X$). For $f(x)=x^2$, $g(x)=2x$:
$fg(x)=2x^3$ but $(f\circ g)(x)=4x^2$.
\end{remark*}

\begin{remark*}[Algebraic structure]
$\mathbb{R}^X$ is a commutative ring under pointwise $+$ and $\cdot$,
zero function $\mathbf{0}(x):=0$, identity $\mathbf{1}(x):=1$.
Scalar multiplication makes it an $\mathbb{R}$-algebra.
\end{remark*}

% ---------------------------------------------------------
% Definition — Bounded Functions
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Boundedness of Functions}]
\begin{definition}[Bounded Above, Bounded Below, Bounded]
\label{def:function-boundedness}
Let $A\subseteq\mathbb{R}$, $f:A\to\mathbb{R}$.
\begin{itemize}
  \item $f$ \textbf{bounded above}: $\exists M,\;\forall x\in A: f(x)\leq M$.
  \item $f$ \textbf{bounded below}: $\exists m,\;\forall x\in A: f(x)\geq m$.
  \item $\operatorname{BoundedOn}(f,A)$: $\exists B>0,\;\forall x\in A: |f(x)|\leq B$.
\end{itemize}
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{BoundedOn}(f,A)
  \;\iff\;
  \exists B>0,\;\forall x\in A: |f(x)|\leq B
  \;\iff\;
  f(A)\subseteq[-B,B].
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{BoundedOn}(f,A)
  \;\colonequiv\;
  \operatorname{BoundedSet}(f(A)).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \neg\,\operatorname{BoundedOn}(f,A)
  \;\iff\;
  \forall B>0,\;\exists x\in A: |f(x)|>B.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Bounded above alone does not imply bounded: $f(x):=-x$ on $\mathbb{R}$
is bounded above by $0$ but unbounded below.
\end{remark*}

% ---------------------------------------------------------
% Definition — Supremum, Infimum, Maximum, Minimum
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Extremal Values}]
\begin{definition}[Supremum, Infimum, Maximum, Minimum of a Function]
\label{def:function-sup-inf-max-min}
Let $A\subseteq\mathbb{R}$, $f:A\to\mathbb{R}$.
\begin{itemize}
  \item $\sup_{x\in A}f(x) := \sup f(A)$;\quad $\inf_{x\in A}f(x) := \inf f(A)$.
  \item $\operatorname{Maximum}(x^*,f(A))$: $x^*\in A$ and $f(x^*)=\sup f(A)$.
  \item $\operatorname{Minimum}(x_*,f(A))$: $x_*\in A$ and $f(x_*)=\inf f(A)$.
\end{itemize}
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{Maximum}(x^*,f(A))
  \;\iff\;
  x^*\in A \;\wedge\; \forall x\in A: f(x)\leq f(x^*).
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{Maximum}(x^*,f(A))
  \;\colonequiv\;
  x^*\in A
  \;\wedge\;
  \operatorname{Supremum}(f(x^*), f(A)).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Supremum exists when $f$ is bounded above; maximum requires the
supremum to be attained at some $x^*\in A$. $f(x)=x$ on $(0,1)$
has supremum $1$ but no maximum.
\end{remark*}

% ---------------------------------------------------------
% Definition — Monotone Functions
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Monotonicity}]
\begin{definition}[Increasing, Decreasing, Monotone, Constant]
\label{def:function-monotonicity}
Let $f:A\to\mathbb{R}$, $A\subseteq\mathbb{R}$ nonempty.
\begin{itemize}
  \item \textbf{Increasing}: $x_1<x_2\Rightarrow f(x_1)\leq f(x_2)$.
  \item \textbf{Strictly increasing}: $x_1<x_2\Rightarrow f(x_1)<f(x_2)$.
  \item \textbf{Decreasing}: $x_1<x_2\Rightarrow f(x_1)\geq f(x_2)$.
  \item \textbf{Strictly decreasing}: $x_1<x_2\Rightarrow f(x_1)>f(x_2)$.
  \item $\operatorname{MonotoneFunction}(f,A)$: increasing or decreasing.
  \item \textbf{Constant}: $\forall x_1,x_2\in A: f(x_1)=f(x_2)$.
\end{itemize}
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  f\text{ strictly increasing}
  \;\iff\;
  \forall x_1,x_2\in A:\;
  x_1<x_2 \Rightarrow f(x_1)<f(x_2).
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{MonotoneFunction}(f,A)
  \;\colonequiv\;
  \bigl(\forall x_1,x_2\in A: x_1<x_2\Rightarrow f(x_1)\leq f(x_2)\bigr)
  \;\vee\;
  \bigl(\forall x_1,x_2\in A: x_1<x_2\Rightarrow f(x_1)\geq f(x_2)\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement — strictly increasing]
\[
  \neg(f\text{ strictly increasing})
  \;\iff\;
  \exists x_1,x_2\in A:\;
  x_1<x_2 \;\wedge\; f(x_1)\geq f(x_2).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Strictly tightens $\leq$ to $<$. Strictly increasing implies
$\operatorname{Injective}(f)$: $f(x_1)=f(x_2)\Rightarrow x_1=x_2$
by contrapositive. A constant function is both increasing and
decreasing but neither strictly.
\end{remark*}
