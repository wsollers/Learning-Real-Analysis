\section{Subsets of $\mathbb{R}$}
\label{sec:subsets-of-R}

\begin{tcolorbox}[title={Toolkit --- Subsets of $\mathbb{R}$}]
\begin{itemize}
  \item Cluster point (= limit point = accumulation point): every
    $\varepsilon$-neighbourhood contains a \emph{distinct} point of $X$.
    Adherent point: every neighbourhood contains \emph{some} point of $X$.
  \item $\operatorname{Within}(x,c,r)$: the unpunctured ball condition
    $|x-c| < r$. Used for adherence and continuity.
  \item $\operatorname{InputTolerance}(x,c,\delta)$: the punctured ball
    $0 < |x-c| < \delta$. Used for cluster points and function limits.
  \item Closed $\iff$ $\overline{X} = X$ $\iff$ sequentially closed.
  \item Heine--Borel: $\operatorname{ClosedSet}(X)$ and
    $\operatorname{BoundedSet}(X)$ $\iff$ every sequence in $X$ has a
    subsequence converging to a limit in $X$.
\end{itemize}
\end{tcolorbox}

\begin{remark*}[Terminological note]
Cluster point = limit point = accumulation point. Bartle and Lebl use
\emph{cluster point}; Rudin uses \emph{limit point}; Tao uses both.
Adherent point (Tao) is the weaker notion that includes isolated points.
\end{remark*}

% ---------------------------------------------------------
% Definition — epsilon-Neighbourhood
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- $\varepsilon$-Neighbourhood}]
\begin{definition}[$\varepsilon$-Neighbourhood]
\label{def:epsilon-neighbourhood}
Let $x \in \mathbb{R}$ and $\varepsilon > 0$.
\[
  V_\varepsilon(x) := (x - \varepsilon, x + \varepsilon)
  = \{\, y \in \mathbb{R} : \operatorname{Within}(y, x, \varepsilon) \,\}.
\]
The \textbf{deleted $\varepsilon$-neighbourhood} of $x$ is
\[
  V_\varepsilon^*(x) := V_\varepsilon(x) \setminus \{x\}
  = \{\, y \in \mathbb{R} : \operatorname{InputTolerance}(y, x, \varepsilon) \,\}.
\]
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  y \in V_\varepsilon(x)
  \;\iff\; \operatorname{Within}(y, x, \varepsilon)
  \;\iff\; |y - x| < \varepsilon.
  \qquad
  y \in V_\varepsilon^*(x)
  \;\iff\; \operatorname{InputTolerance}(y, x, \varepsilon)
  \;\iff\; 0 < |y - x| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The deleted neighbourhood $V_\varepsilon^*(x)$ is the operative set
in the limit definition. The condition $0 < |x - c| < \delta$ means
$\operatorname{InputTolerance}(x, c, \delta)$, i.e., $x \in
V_\delta^*(c)$. The $0 <$ excludes the base point — the defining
structural feature separating limits from continuity.
\end{remark*}

% ---------------------------------------------------------
% Definition — Cluster Point
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Cluster Point}]
\begin{definition}[Cluster Point]
\label{def:cluster-point-R}
Let $X \subseteq \mathbb{R}$ and $x \in \mathbb{R}$. The point $x$
is a \textbf{cluster point} of $X$ if
\[
  \forall\varepsilon > 0,\;
  \exists\, y \in X \setminus \{x\} :\;
  \operatorname{Within}(y, x, \varepsilon).
\]
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ClusterPoint}(x, X)
  \;\iff\;
  \forall\varepsilon > 0,\;
  \exists\, y \in X \setminus \{x\} :\;
  |y - x| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{ClusterPoint}(x, X)
  \;\colonequiv\;
  \forall\varepsilon > 0,\;
  V_\varepsilon^*(x) \cap X \neq \emptyset.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \neg\,\operatorname{ClusterPoint}(x, X)
  \;\iff\;
  \exists\varepsilon > 0,\;
  \forall\, y \in X \setminus \{x\} :\;
  \neg\operatorname{Within}(y, x, \varepsilon).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  \neg\,\operatorname{ClusterPoint}(x, X)
  \;\colonequiv\;
  \exists\varepsilon > 0 :\;
  V_\varepsilon^*(x) \cap X = \emptyset.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A cluster point is approached by $X$ via distinct witnesses
arbitrarily close. A cluster point need not belong to $X$: $0$ is
a cluster point of $(0,1)$ but $0 \notin (0,1)$. The condition
$\operatorname{FunctionLimit}(f,c,L)$ requires
$\operatorname{ClusterPoint}(c, \operatorname{dom}(f))$; without
it the punctured ball condition is vacuous.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Sequential Characterization of Cluster Points
% ---------------------------------------------------------

\begin{theorem}[Sequential Characterization of Cluster Points]
\label{thm:cluster-point-sequential}
$\operatorname{ClusterPoint}(c, A)$ if and only if there exists
$(a_n) \subseteq A \setminus \{c\}$ with $a_n \to c$.

\smallskip\noindent
\hyperref[prf:cluster-point-sequential]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ClusterPoint}(c, A)
  \;\iff\;
  \exists (a_n) \subseteq A \setminus \{c\} :\;
  \operatorname{ConvergentSequence}(a_n, c).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The forward direction: use $\varepsilon = 1/n$ to build the witness
sequence. The reverse direction is immediate since the sequence
witnesses the ball condition for every $\varepsilon$. This bridges
the $\varepsilon$-$\delta$ definition with the sequential language.
\end{remark*}

% ---------------------------------------------------------
% Definition — Adherent Point and Isolated Point
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Adherent Point; Isolated Point}]
\begin{definition}[Adherent Point; Isolated Point]
\label{def:adherent-point-R}
Let $X \subseteq \mathbb{R}$ and $x \in \mathbb{R}$.

$x$ is an \textbf{adherent point} of $X$ if
\[
  \forall\varepsilon > 0,\;
  \exists\, y \in X :\;
  \operatorname{Within}(y, x, \varepsilon).
\]

A point $x \in X$ is an \textbf{isolated point} of $X$ if
$\neg\,\operatorname{ClusterPoint}(x, X)$; equivalently,
$\exists\varepsilon > 0 : V_\varepsilon(x) \cap X = \{x\}$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{AdherentPoint}(x, X)
  \;\iff\;
  \forall\varepsilon > 0 :\;
  V_\varepsilon(x) \cap X \neq \emptyset.
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{AdherentPoint}(x, X)
  \;\colonequiv\;
  \forall\varepsilon > 0,\;
  \exists\, y \in X :\;
  \operatorname{Within}(y, x, \varepsilon).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  \neg\,\operatorname{AdherentPoint}(x, X)
  \;\colonequiv\;
  \exists\varepsilon > 0 :\;
  V_\varepsilon(x) \cap X = \emptyset.
\]
\end{remark*}

\begin{remark*}[Cluster vs.\ adherent]
The only difference: $\operatorname{Within}$ vs
$\operatorname{InputTolerance}$ — adherent permits the witness $y =
x$; cluster requires $y \neq x$. Partition:
$\text{Adherent}(X) = \operatorname{ClusterPoint}(X) \cup
\operatorname{Isolated}(X)$.
\end{remark*}

% ---------------------------------------------------------
% Definition — Interior, Exterior, Boundary
% ---------------------------------------------------------

\begin{definition}[Interior, Exterior, Boundary]
\label{def:interior-exterior-boundary-R}
Let $X \subseteq \mathbb{R}$, $x \in \mathbb{R}$.

$\operatorname{InteriorPoint}(x, X)$: $\exists\varepsilon > 0 :
V_\varepsilon(x) \subseteq X$.

$\operatorname{BoundaryPoint}(x, X)$: $\forall\varepsilon > 0 :
V_\varepsilon(x) \cap X \neq \emptyset \;\wedge\;
V_\varepsilon(x) \cap X^c \neq \emptyset$.

Interior $X^\circ := \{x : \operatorname{InteriorPoint}(x,X)\}$;\quad
boundary $\partial X := \{x : \operatorname{BoundaryPoint}(x,X)\}$.
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
  \operatorname{InteriorPoint}(x,X)
  &\;\iff\; \exists\varepsilon>0,\;
    \forall y : \operatorname{Within}(y,x,\varepsilon) \Rightarrow y \in X. \\
  \operatorname{BoundaryPoint}(x,X)
  &\;\iff\; \forall\varepsilon>0 :
    V_\varepsilon(x)\cap X\neq\emptyset
    \wedge V_\varepsilon(x)\cap X^c\neq\emptyset.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
$X^\circ$, $\partial X$, $\mathrm{Ext}(X)$ partition $\mathbb{R}$.
$\operatorname{OpenSet}(X) \iff X = X^\circ$.
$\operatorname{ClosedSet}(X) \iff X \supseteq \partial X$.
$\overline{X} = X^\circ \cup \partial X$.
\end{remark*}

% ---------------------------------------------------------
% Definition — Closure
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Closure}]
\begin{definition}[Closure]
\label{def:closure-R}
$\overline{X} := \{x \in \mathbb{R} : \operatorname{AdherentPoint}(x, X)\}$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  x \in \overline{X}
  \;\iff\;
  \operatorname{AdherentPoint}(x, X)
  \;\iff\;
  \forall\varepsilon>0 : V_\varepsilon(x)\cap X\neq\emptyset.
\]
\end{remark*}

\begin{remark*}[Canonical examples]
$\overline{(a,b)} = [a,b]$;\quad $\overline{\mathbb{Q}} = \mathbb{R}$;\quad
$\overline{\mathbb{Z}} = \mathbb{Z}$;\quad
$\overline{\{1/n\}} = \{1/n\}\cup\{0\}$.
\end{remark*}

% ---------------------------------------------------------
% Lemma — Elementary Properties of Closures
% ---------------------------------------------------------

\begin{lemma}[Elementary Properties of Closures]
\label{lem:closure-elementary}
Let $X, Y \subseteq \mathbb{R}$. Then: (i) $X \subseteq \overline{X}$;
(ii) $\overline{X \cup Y} = \overline{X} \cup \overline{Y}$;
(iii) $\overline{X \cap Y} \subseteq \overline{X} \cap \overline{Y}$;
(iv) $X \subseteq Y \Rightarrow \overline{X} \subseteq \overline{Y}$.

\smallskip\noindent
\hyperref[prf:closure-elementary]{\textit{Go to proof.}}
\end{lemma}

\begin{remark*}[Standard quantified statement]
\[
  \overline{X \cup Y} = \overline{X} \cup \overline{Y};\quad
  \overline{X \cap Y} \subseteq \overline{X} \cap \overline{Y}.
\]
\end{remark*}

\begin{remark*}[Failure of equality in (iii)]
Take $X = (0,1)$, $Y = (1,2)$: $\overline{X\cap Y} = \emptyset$ but
$\overline{X}\cap\overline{Y} = \{1\}$.
\end{remark*}

% ---------------------------------------------------------
% Definition — Closed Set
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Closed Subset of $\mathbb{R}$}]
\begin{definition}[Closed Subset of $\mathbb{R}$]
\label{def:closed-set-R}
$\operatorname{ClosedSet}(E) \iff \overline{E} = E$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ClosedSet}(E)
  \;\iff\;
  \forall x \in \mathbb{R} :\;
  \operatorname{AdherentPoint}(x,E) \Rightarrow x \in E.
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{ClosedSet}(E)
  \;\colonequiv\;
  \forall x :\;
  \operatorname{AdherentPoint}(x, E) \Rightarrow x \in E.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \neg\,\operatorname{ClosedSet}(E)
  \;\iff\;
  \exists x \notin E :\; \operatorname{AdherentPoint}(x, E).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  \neg\,\operatorname{ClosedSet}(E)
  \;\colonequiv\;
  \exists x \notin E :\;
  \forall\varepsilon>0 : V_\varepsilon(x)\cap E\neq\emptyset.
\]
\end{remark*}

\begin{remark*}[Interpretation]
$E$ is closed iff it contains its own boundary. Examples:
$[a,b]$ closed; $(a,b)$ not closed; $\mathbb{R}$ and $\emptyset$
both closed; $\mathbb{Q}$ not closed.
\end{remark*}

% ---------------------------------------------------------
% Corollary — Sequentially Closed
% ---------------------------------------------------------

\begin{corollary}[Closed iff Sequentially Closed]
\label{cor:closed-iff-seq-limits}
$\operatorname{ClosedSet}(X) \iff \forall (a_n)\subseteq X,\;
a_n \to x \Rightarrow x \in X$.

\smallskip\noindent
\hyperref[prf:closed-iff-seq-limits]{\textit{Go to proof.}}
\end{corollary}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ClosedSet}(X)
  \;\iff\;
  \forall (a_n)\subseteq X,\;
  \operatorname{ConvergentSequence}(a_n, x)
  \Rightarrow x \in X.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A closed set cannot be escaped by convergent sequences. Every adherent
point has a witness sequence; closedness forces the limit into the set.
\end{remark*}

% ---------------------------------------------------------
% Corollary — Every Point of an Interval Is a Cluster Point
% ---------------------------------------------------------

\begin{corollary}[Every Point of an Interval Is a Cluster Point]
\label{cor:interval-all-limit-points}
If $I$ is an interval, then $\operatorname{ClusterPoint}(x, I)$ for
every $x \in I$.

\smallskip\noindent
\hyperref[prf:interval-all-limit-points]{\textit{Go to proof.}}
\end{corollary}

\begin{remark*}[Interpretation]
This justifies intervals as natural domains for function limits.
$\operatorname{FunctionLimit}(f,c,L)$ requires
$\operatorname{ClusterPoint}(c, \operatorname{dom}(f))$; on any
interval this is free.
\end{remark*}

% ---------------------------------------------------------
% Definition — Bounded Subset
% ---------------------------------------------------------

\begin{definition}[Bounded Subset of $\mathbb{R}$]
\label{def:bounded-set-R}
$\operatorname{BoundedSet}(X) \iff \exists M>0,\;\forall x\in X:
|x|\leq M$.
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{BoundedSet}(X)
  \;\iff\;
  \exists M>0,\;\forall x\in X : |x|\leq M.
  \qquad
  \neg\,\operatorname{BoundedSet}(X)
  \;\iff\;
  \forall M>0,\;\exists x\in X : |x|>M.
\]
\end{remark*}

% ---------------------------------------------------------
% Theorem — Heine-Borel
% ---------------------------------------------------------

\begin{theorem}[Heine--Borel Theorem for $\mathbb{R}$]
\label{thm:heine-borel}
$\operatorname{ClosedSet}(X) \wedge \operatorname{BoundedSet}(X)$
$\iff$ every $(a_n)\subseteq X$ has a subsequence converging to
a limit in $X$.

\smallskip\noindent
\hyperref[prf:heine-borel]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ClosedSet}(X) \wedge \operatorname{BoundedSet}(X)
  \;\iff\;
  \forall (a_n)\subseteq X,\;
  \exists (a_{n_k}),\;\exists L\in X : a_{n_k}\to L.
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{SequentiallyCompact}(X)
  \;\colonequiv\;
  \operatorname{ClosedSet}(X) \wedge \operatorname{BoundedSet}(X).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \neg\bigl(\operatorname{ClosedSet}(X)\wedge\operatorname{BoundedSet}(X)\bigr)
  \;\iff\;
  \exists (a_n)\subseteq X:\;
  \text{no subsequence converges to a limit in } X.
\]
\end{remark*}

\begin{remark*}[Failure mode decomposition]
Either $\operatorname{BoundedSet}(X)$ fails — a sequence escapes to
$\pm\infty$ — or $\operatorname{ClosedSet}(X)$ fails — a convergent
subsequence exists but its limit lies outside $X$.
\end{remark*}

\begin{remark*}[Interpretation]
Closed prevents limits from escaping through boundary gaps; bounded
prevents escape to infinity. Both are required. Forward: bounded
gives Bolzano--Weierstrass; closed keeps the limit in $X$. Reverse:
negate either hypothesis to construct a sequence with no convergent
subsequence in $X$.
\end{remark*}

% ---------------------------------------------------------
% Definition — True Near a Point
% ---------------------------------------------------------

\begin{definition}[True Near a Point]
\label{def:true-near}
Property $P(f(x))$ is \textbf{true near $x_0$} if
$\exists\delta>0,\;\forall x : \operatorname{InputTolerance}(x,x_0,
\delta) \Rightarrow P(f(x))$.
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
  P(f(x))\text{ true near }x_0
  \;\iff\;
  \exists\delta>0,\;\forall x :
  0 < |x-x_0| < \delta \Rightarrow P(f(x)).
\]
\end{remark*}

\begin{remark*}[Interpretation]
$\operatorname{FunctionLimit}(f,x_0,L)$ states: for every
$\varepsilon>0$, the property $\operatorname{OutputTolerance}(f(x),
L,\varepsilon)$ is true near $x_0$.
\end{remark*}
