\section{One-Sided Limits}
\label{sec:one-sided-limits}

% ---------------------------------------------------------
% Toolkit
% ---------------------------------------------------------

\begin{tcolorbox}[title={Toolkit --- One-Sided Limits}]
\begin{itemize}
  \item Right-hand limit $f(c^+)$: approach $c$ from above only
    ($x > c$).
  \item Left-hand limit $f(c^-)$: approach $c$ from below only
    ($x < c$).
  \item Sequential criteria hold for each direction separately.
  \item Two-sided limit exists iff both one-sided limits exist and
    are equal: $\lim_{x \to c} f(x) = L \iff f(c^+) = L = f(c^-)$.
  \item One-sided limits are the natural tool for jump
    discontinuities, monotone functions, and piecewise-defined
    functions.
\end{itemize}
\end{tcolorbox}

% ---------------------------------------------------------
% Definition — One-Sided Limits
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Right-Hand and Left-Hand Limits}]
\begin{definition}[Right-Hand and Left-Hand Limits]
\label{def:one-sided-limits}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$c \in \mathbb{R}$.

\medskip
(i) Suppose $c$ is a cluster point of $A \cap (c, \infty)$.
The \textbf{right-hand limit} of $f$ at $c$ is $L$, written
$\lim_{x \to c^+} f(x) = L$ or $f(c^+) = L$, if
\[
  \forall\varepsilon > 0,\;
  \exists\delta > 0,\;
  \forall x \in A :\;
  0 < x - c < \delta \;\Rightarrow\; |f(x) - L| < \varepsilon.
\]

(ii) Suppose $c$ is a cluster point of $A \cap (-\infty, c)$.
The \textbf{left-hand limit} of $f$ at $c$ is $L$, written
$\lim_{x \to c^-} f(x) = L$ or $f(c^-) = L$, if
\[
  \forall\varepsilon > 0,\;
  \exists\delta > 0,\;
  \forall x \in A :\;
  0 < c - x < \delta \;\Rightarrow\; |f(x) - L| < \varepsilon.
\]
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  f(c^+) = L
  \;\iff\;
  \forall\varepsilon > 0,\;
  \exists\delta > 0,\;
  \forall x \in A :\;
  c < x < c + \delta \;\Rightarrow\; |f(x) - L| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The right-hand limit restricts attention to inputs strictly to the
right of $c$. It is the limit of $f$ restricted to $A \cap (c,
\infty)$. The left-hand limit is the symmetric restriction. These
are the natural tools for studying behavior at points where the
function has different left and right behavior — jump discontinuities,
boundary points of intervals, and piecewise-defined functions.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Sequential Criterion for Right-Hand Limits
% ---------------------------------------------------------

\begin{theorem}[Sequential Criterion for Right-Hand Limits]
\label{thm:sequential-criterion-right-hand-limit}
Let $f : A \to \mathbb{R}$ and let $c$ be a cluster point of
$A \cap (c, \infty)$. The following are equivalent:
\begin{enumerate}[label=(\roman*)]
  \item $\lim_{x \to c^+} f(x) = L$.
  \item For every sequence $(x_n) \subseteq A$ with $x_n > c$ for
    all $n$ and $x_n \to c$, the sequence $(f(x_n)) \to L$.
\end{enumerate}

\smallskip
\noindent
\hyperref[prf:sequential-criterion-right-hand-limit]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  f(c^+) = L
  \;\iff\;
  \forall (x_n) \subseteq A \cap (c, \infty) :\;
  x_n \to c \;\Rightarrow\; f(x_n) \to L.
\]
\end{remark*}

% ---------------------------------------------------------
% Corollary — Sequential Criterion for Left-Hand Limits
% ---------------------------------------------------------

\begin{corollary}[Sequential Criterion for Left-Hand Limits]
\label{thm:sequential-criterion-left-hand-limit}
Let $f : A \to \mathbb{R}$ and let $c$ be a cluster point of
$A \cap (-\infty, c)$. The following are equivalent:
\begin{enumerate}[label=(\roman*)]
  \item $\lim_{x \to c^-} f(x) = L$.
  \item For every sequence $(x_n) \subseteq A$ with $x_n < c$ for
    all $n$ and $x_n \to c$, the sequence $(f(x_n)) \to L$.
\end{enumerate}

\smallskip
\noindent
\hyperref[prf:sequential-criterion-left-hand-limit]{\textit{Go to proof.}}
\end{corollary}

\begin{remark*}[Symmetry]
The proof is the reflection of the right-hand criterion: replace
$0 < x - c < \delta$ by $0 < c - x < \delta$ and construct $x_n$
with $c - 1/n < x_n < c$ in place of $c < x_n < c + 1/n$. All
other steps are word-for-word identical.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Two-Sided Limit via One-Sided Limits
% ---------------------------------------------------------

\begin{theorem}[Two-Sided Limit via One-Sided Limits]
\label{thm:two-sided-limit-iff-matching-one-sided-limits}
Let $f : A \to \mathbb{R}$ and let $c$ be a cluster point of both
$A \cap (c, \infty)$ and $A \cap (-\infty, c)$. Then
\[
  \lim_{x \to c} f(x) = L
  \;\iff\;
  \lim_{x \to c^+} f(x) = L = \lim_{x \to c^-} f(x).
\]

\smallskip
\noindent
\hyperref[prf:two-sided-limit-iff-matching-one-sided-limits]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \lim_{x \to c} f(x) = L
  \;\iff\;
  f(c^+) = L \;\wedge\; f(c^-) = L.
\]
\end{remark*}

\begin{remark*}[Failure mode]
If $f(c^+) \neq f(c^-)$, the two-sided limit does not exist.
The Heaviside step function $H(x) = 0$ for $x < 0$, $H(x) = 1$
for $x \geq 0$ has $H(0^-) = 0 \neq 1 = H(0^+)$, so
$\lim_{x \to 0} H(x)$ does not exist.
\end{remark*}

\begin{remark*}[Interpretation]
The two-sided limit exists and equals $L$ precisely when the
function approaches $L$ from both sides simultaneously. The
one-sided limits are the finer instruments that detect directional
behavior; the two-sided limit is a coarser summary that exists
only when both directions agree.
\end{remark*}
