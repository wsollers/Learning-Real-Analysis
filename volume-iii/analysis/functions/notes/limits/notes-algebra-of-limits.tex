\section{Algebra of Limits}
\label{sec:algebra-of-limits-functions}

\begin{tcolorbox}[title={Toolkit --- Algebra of Limits}]
\begin{itemize}
  \item All five rules follow from the Sequential Criterion plus
    algebra of sequence limits. No new $\varepsilon$-$\delta$ work.
  \item Bridge: $(x_n)\subseteq A\setminus\{c\}$, $x_n\to c$
    $\Rightarrow$ $f(x_n)\to L$, $g(x_n)\to M$
    $\Rightarrow$ apply sequence algebra
    $\Rightarrow$ lift back by Sequential Criterion.
  \item Quotient: $M\neq 0$ forces $\operatorname{BoundedAwayFromZeroNear}(g,c)$,
    making $f/g$ well-defined on a punctured neighbourhood.
\end{itemize}
\end{tcolorbox}

% ---------------------------------------------------------
% Theorem — Algebra of Limits
% ---------------------------------------------------------

\begin{theorem}[Algebra of Limits of Functions]
\label{thm:limit-algebra}
Let $f,g:A\to\mathbb{R}$, $c$ a cluster point of $A$,
$\operatorname{FunctionLimit}(f,c,L)$, $\operatorname{FunctionLimit}(g,c,M)$.
Then:
\begin{enumerate}[label=(\roman*)]
  \item $\operatorname{FunctionLimit}(f+g,\,c,\,L+M)$.
  \item $\operatorname{FunctionLimit}(f-g,\,c,\,L-M)$.
  \item $\operatorname{FunctionLimit}(\alpha f,\,c,\,\alpha L)$
    for all $\alpha\in\mathbb{R}$.
  \item $\operatorname{FunctionLimit}(fg,\,c,\,LM)$.
  \item If $M\neq 0$: $g(x)\neq 0$ for all $x$ in some punctured
    neighbourhood of $c$, and
    $\operatorname{FunctionLimit}(f/g,\,c,\,L/M)$.
\end{enumerate}

\smallskip\noindent
\hyperref[prf:limit-algebra]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement — sum rule]
\[
  \operatorname{FunctionLimit}(f,c,L)
  \;\wedge\;
  \operatorname{FunctionLimit}(g,c,M)
  \;\Rightarrow\;
  \ForallOT{\varepsilon},\;
  \ExistsIT{\delta},\;
  \forall x\in A:\;
  \operatorname{InputTolerance}(x,c,\delta)
  \Rightarrow
  \operatorname{OutputTolerance}((f+g)(x),L+M,\varepsilon).
\]
\end{remark*}

\begin{remark*}[Definition predicate reading — sum rule]
\[
  \operatorname{FunctionLimit}(f+g,c,L+M)
  \;\colonequiv\;
  \ForallOT{\varepsilon},\;
  \ExistsIT{\delta},\;
  \forall x\in A:\;
  \operatorname{InputTolerance}(x,c,\delta)
  \Rightarrow
  \operatorname{OutputTolerance}((f+g)(x),L+M,\varepsilon).
\]
\end{remark*}

\begin{remark*}[Proof strategy]
Fix $(x_n)\subseteq A\setminus\{c\}$ with $x_n\to c$. By the
Sequential Criterion: $f(x_n)\to L$, $g(x_n)\to M$. Algebra of
sequence limits gives $(f+g)(x_n)\to L+M$, $(fg)(x_n)\to LM$, etc.
Sequential Criterion lifts back to function limits. Five rules, one
line each.
\end{remark*}

\begin{remark*}[Consequences]
$\lim_{x\to c}(\alpha f+\beta g)(x)=\alpha L+\beta M$. For any
polynomial $p$: $\lim_{x\to c}p(x)=p(c)$. For rational $P/Q$ with
$Q(c)\neq 0$: $\lim_{x\to c}P(x)/Q(x)=P(c)/Q(c)$.
\end{remark*}
