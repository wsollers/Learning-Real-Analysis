\subsection{Characterization of Discontinuity}
\label{sec:discontinuity}

\begin{tcolorbox}[title={Toolkit --- Discontinuity}]
\begin{itemize}
  \item $\operatorname{DiscontinuousAt}(f,a)$ is the negation of $\operatorname{ContinuousAt}(f,a)$.
  \item Quantifier reversal governs failure of continuity.
  \item Three equivalent formulations: $\varepsilon$-$\delta$, sequential, neighbourhood.
  \item Taxonomy: removable, jump, essential.
\end{itemize}
\end{tcolorbox}

% ---------------------------------------------------------
% Definition — Point of Discontinuity
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Point of Discontinuity}]
\begin{definition}[Point of Discontinuity]
\label{def:point-of-discontinuity}
A function $f$ is discontinuous at $a$ if it is not continuous at $a$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\operatorname{DiscontinuousAt}(f,a) \\
&\quad\Longleftrightarrow\quad
\exists \varepsilon_0 > 0 \; \forall \delta > 0 \; \exists x \in A \\
&\qquad \operatorname{InputTolerance}(x,a,\delta)
\;\wedge\;
\neg \operatorname{OutputTolerance}(f(x),f(a),\varepsilon_0)
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
\operatorname{DiscontinuousAt}(f,a)
\;\colonequiv\;
\neg \operatorname{ContinuousAt}(f,a)
\]
\end{remark*}

\begin{remark*}[Interpretation]
A fixed output gap persists regardless of how small the input neighborhood is.
\end{remark*}

% ---------------------------------------------------------
% Definition — Sequential Characterization
% ---------------------------------------------------------

\begin{definition}[Sequential Characterization of Discontinuity]
\label{def:sequential-discontinuity-at-a-point}
A function $f$ is discontinuous at $x_0$ if there exists a sequence
$(x_n)$ with $x_n \to x_0$ such that $f(x_n)$ does not converge to $f(x_0)$.
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\operatorname{DiscontinuousAt}(f,x_0) \\
&\quad\Longleftrightarrow\quad
\exists (x_n)\subseteq A \\
&\qquad \operatorname{ConvergentSequence}(x_n,x_0)
\;\wedge\;
\neg \operatorname{ConvergentSequence}(f(x_n),f(x_0))
\end{aligned}
\]
\end{remark*}

% ---------------------------------------------------------
% Definition — Neighbourhood Characterization
% ---------------------------------------------------------

\begin{definition}[Neighbourhood Characterization of Discontinuity]
\label{def:neighborhood-discontinuity-at-a-point}
A function $f$ is discontinuous at $x_0$ if there exists a neighborhood
$V$ of $f(x_0)$ such that for every neighborhood $U$ of $x_0$, there exists
$x \in U \cap A$ with $f(x) \notin V$.
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\operatorname{DiscontinuousAt}(f,x_0) \\
&\quad\Longleftrightarrow\quad
\exists \varepsilon_0 > 0 \; \forall \delta > 0 \; \exists x \in A \\
&\qquad \operatorname{InputTolerance}(x,x_0,\delta)
\;\wedge\;
\neg \operatorname{OutputTolerance}(f(x),f(x_0),\varepsilon_0)
\end{aligned}
\]
\end{remark*}

% ---------------------------------------------------------
% Lemma — Equivalence
% ---------------------------------------------------------

\begin{lemma}[Equivalent Formulations of Discontinuity at a Point]
\label{lem:equivalent-discontinuity-at-a-point}
The $\varepsilon$-$\delta$, sequential, and neighbourhood formulations are equivalent.

% ---------------------------------------------------------
% Definition — Types of Discontinuity
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Types of Discontinuity}]
\begin{definition}[Types of Discontinuity]
\label{def:types-of-discontinuity-at-a-point}
Let $x_0 \in A$.
\begin{enumerate}[label=(\roman*)]
  \item Removable: $\operatorname{FunctionLimit}(f,x_0,L)$ and $L \neq f(x_0)$.
  \item Jump: $\operatorname{LeftLimit}(f,x_0,L_-)$ and $\operatorname{RightLimit}(f,x_0,L_+)$ with $L_- \neq L_+$.
  \item Essential: not removable.
\end{enumerate}
\end{definition}
\end{tcolorbox}
