\section{Continuity}
\label{sec:continuity}

% ---------------------------------------------------------
% Toolkit
% ---------------------------------------------------------

\begin{tcolorbox}[title={Toolkit — Continuity}]
\begin{itemize}
  \item Three equivalent forms: $\varepsilon$-$\delta$, neighbourhood,
    sequential. All three are used in proofs.
  \item Key structural rule: $\delta$ may depend on both $\varepsilon$
    and the base point $c$. When $\delta$ depends only on $\varepsilon$,
    the condition becomes uniform continuity.
  \item Isolated points are automatically continuous. Non-trivial
    verification is required only at cluster points.
  \item Algebra: sums, differences, products, scalar multiples, and
    (when denominator is nonzero) quotients of continuous functions
    are continuous.
  \item Composition preserves continuity: if $f$ is continuous at $c$
    and $g$ is continuous at $f(c)$, then $g \circ f$ is continuous
    at $c$.
\end{itemize}
\end{tcolorbox}

% ---------------------------------------------------------
% Definition — Relative Neighborhood
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Relative Neighborhood}]
\begin{definition}[Relative Neighborhood]
\label{def:relative-neighborhood}
Let $E \subseteq \mathbb{R}$ and let $a \in E$. A
\textbf{neighborhood of $a$ in $E$}, denoted $U_E(a)$, is a set of
the form
\[
  U_E(a) \;:=\; E \cap U(a),
\]
where $U(a)$ is a neighborhood of $a$ in $\mathbb{R}$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  U_E(a) \;=\; E \cap (a - \delta,\, a + \delta)
  \quad \text{for some } \delta > 0.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The relative neighborhood is the portion of the domain $E$ that
lies within a full $\delta$-ball around $a$. It relativizes the
standard neighborhood concept to an arbitrary subset: properties
stated in terms of $U_E(a)$ are local properties of $f$ on $E$
near $a$, unaffected by behavior of $\mathbb{R}$ outside $E$. This
is the correct setting for stating local properties of functions
defined on proper subsets of $\mathbb{R}$.
\end{remark*}

% ---------------------------------------------------------
% Definition — Continuity at a Point (epsilon-delta)
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Continuity at a Point}]
\begin{definition}[Continuity at a Point]
\label{def:continuous-at-point}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$c \in A$. The function $f$ is \textbf{continuous at $c$} if for
every $\varepsilon > 0$ there exists $\delta > 0$ such that for
every $x \in A$,
\[
  |x - c| < \delta \;\Rightarrow\; |f(x) - f(c)| < \varepsilon.
\]
If $f$ is not continuous at $c$, then $f$ is \textbf{discontinuous
at $c$}.

\smallskip
\noindent
\hyperref[prf:equivalent-forms-continuity]{\textit{Go to proof.}}
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  f \text{ continuous at } c
  \;\iff\;
  \forall\varepsilon > 0,\;
  \exists\delta > 0,\;
  \forall x \in A:\;
  |x - c| < \delta \;\Rightarrow\; |f(x) - f(c)| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  f \text{ discontinuous at } c
  \;\iff\;
  \exists\varepsilon_0 > 0,\;
  \forall\delta > 0,\;
  \exists x \in A:\;
  |x - c| < \delta \;\wedge\; |f(x) - f(c)| \geq \varepsilon_0.
\]
\end{remark*}

\begin{remark*}[Failure mode decomposition]
Discontinuity at $c$ decomposes into three distinct failure modes:
\[
  \underbrace{%
    \lim_{x \to c} f(x) \text{ exists but } \neq f(c)
  }_{\text{removable discontinuity}}
  \;\vee\;
  \underbrace{%
    \text{one-sided limits exist and are unequal}
  }_{\text{jump discontinuity}}
  \;\vee\;
  \underbrace{%
    \text{no one-sided limit exists}
  }_{\text{essential discontinuity}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Three features distinguish continuity from the limit definition.
First, $x = c$ is permitted — the implication holds for all $x \in A$,
not just $x \neq c$. Second, the target value is $f(c)$ itself, not an
independently specified limit $L$. Third, $c$ must belong to $A$.
At a cluster point $c$, continuity is equivalent to
$\lim_{x \to c} f(x) = f(c)$. At an isolated point, the condition
holds vacuously for every $\varepsilon > 0$ since no $x \in A$
satisfies $0 < |x - c| < \delta$ for small $\delta$, so every
isolated point of $A$ is automatically a continuity point.
\end{remark*}

% ---------------------------------------------------------
% Definition — Continuity at a Point (neighbourhood form)
% ---------------------------------------------------------

\begin{definition}[Continuity at a Point — Neighbourhood Form]
\label{def:continuous-at-point-nbhd}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$c \in A$. The function $f$ is \textbf{continuous at $c$} in the
neighbourhood sense if for every $\varepsilon$-neighbourhood
$V_\varepsilon(f(c))$ of $f(c)$, there exists a $\delta$-neighbourhood
$V_\delta(c)$ of $c$ such that
\[
  f\!\bigl(A \cap V_\delta(c)\bigr) \;\subseteq\; V_\varepsilon(f(c)).
\]
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
  \forall\varepsilon > 0,\;
  \exists\delta > 0:\;
  \forall x \in A \cap V_\delta(c),\;
  f(x) \in V_\varepsilon(f(c)).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The neighbourhood form packages the $\varepsilon$-$\delta$ condition
as an image-inclusion statement: the image of the $\delta$-slice of
$A$ must land inside the $\varepsilon$-ball around $f(c)$. It uses
the full neighbourhood $V_\delta(c)$, not the punctured version
$V_\delta^*(c)$ required by the limit definition. This is the form
of continuity most naturally generalized to metric spaces, where
$|x - c| < \delta$ is replaced by $d(x,c) < \delta$.
\end{remark*}

% ---------------------------------------------------------
% Definition — Continuity at a Point (sequential form)
% ---------------------------------------------------------

\begin{definition}[Continuity at a Point — Sequential Form]
\label{def:continuous-at-point-seq}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$c \in A$. The function $f$ is \textbf{continuous at $c$} in the
sequential sense if for every sequence $(x_n) \subseteq A$ with
$x_n \to c$, the sequence $(f(x_n))$ converges to $f(c)$.
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
  \forall (x_n) \subseteq A:\;
  x_n \to c \;\Rightarrow\; f(x_n) \to f(c).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  f \text{ discontinuous at } c
  \;\iff\;
  \exists (x_n) \subseteq A:\;
  x_n \to c \;\wedge\; f(x_n) \not\to f(c).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The sequential form converts the point-proximity condition into a
sequence-convergence condition. It differs from the sequential
criterion for function limits in one essential respect: sequences
$(x_n)$ may include $x_n = c$, since continuity at $c$ requires
$f(c)$ itself to be consistent with the limiting behavior, not
merely the behavior off $c$. The negation — a single sequence
converging to $c$ whose image does not converge to $f(c)$ — is the
standard tool for exhibiting discontinuities.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Three Equivalent Forms of Continuity
% ---------------------------------------------------------

\begin{theorem}[Three Equivalent Forms of Continuity]
\label{thm:continuity-three-equiv}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$c \in A$. The following are equivalent:
\begin{enumerate}[label=(\roman*)]
  \item $\forall\varepsilon>0,\;\exists\delta>0,\;\forall x\in A:\;
    |x-c|<\delta \Rightarrow |f(x)-f(c)|<\varepsilon$.
    \hfill ($\varepsilon$-$\delta$)
  \item $\forall\,V_\varepsilon(f(c)),\;\exists\,V_\delta(c):\;
    f(A \cap V_\delta(c)) \subseteq V_\varepsilon(f(c))$.
    \hfill (neighbourhood)
  \item $\forall (x_n)\subseteq A:\;
    x_n \to c \Rightarrow f(x_n) \to f(c)$.
    \hfill (sequential)
\end{enumerate}

\smallskip
\noindent
\hyperref[prf:equivalent-forms-continuity]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
\text{(i)}\quad
  &\forall\varepsilon>0,\;
   \exists\delta>0,\;
   \forall x\in A:\;
   |x-c|<\delta \Rightarrow |f(x)-f(c)|<\varepsilon \\
\text{(ii)}\quad
  &\forall\varepsilon>0,\;
   \exists\delta>0:\;
   f(A\cap V_\delta(c)) \subseteq V_\varepsilon(f(c)) \\
\text{(iii)}\quad
  &\forall (x_n)\subseteq A:\;
   x_n\to c \Rightarrow f(x_n)\to f(c)
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The equivalence (i)$\iff$(ii) is notational: $x \in V_\delta(c)$
means $|x-c|<\delta$, and $f(x) \in V_\varepsilon(f(c))$ means
$|f(x)-f(c)|<\varepsilon$. The equivalence (i)$\iff$(iii) mirrors
the sequential criterion for function limits, but with the critical
difference that sequences may include $x_n = c$. The proof of
(iii)$\Rightarrow$(i) is by contrapositive: negating (i) constructs
a sequence via the $1/n$ trick that converges to $c$ but whose image
does not converge to $f(c)$, contradicting (iii).
\end{remark*}

% ---------------------------------------------------------
% Definition — Discontinuity Criterion
% ---------------------------------------------------------

\begin{definition}[Discontinuity Criterion]
\label{def:discontinuity-criterion}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$c \in A$. The function $f$ is \textbf{discontinuous at $c$} if and
only if there exists a sequence $(x_n) \subseteq A$ such that
$x_n \to c$ but $f(x_n) \not\to f(c)$.
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
  f \text{ discontinuous at } c
  \;\iff\;
  \exists (x_n) \subseteq A:\;
  x_n \to c \;\wedge\; f(x_n) \not\to f(c).
\]
\end{remark*}

\begin{remark*}[Interpretation]
This is the negation of the sequential form of continuity. A single
witness sequence suffices: it converges in the domain but its image
fails to converge to the function value. The criterion is the
standard tool for proving discontinuity — exhibit one bad sequence.
\end{remark*}

% ---------------------------------------------------------
% Definition — Continuity on a Set
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Continuity on a Set}]
\begin{definition}[Continuity on a Set]
\label{def:continuity-on-set}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$B \subseteq A$. The function $f$ is \textbf{continuous on $B$} if
$f$ is continuous at every point of $B$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  f \text{ continuous on } B
  \;\iff\;
  \forall c \in B,\;
  \forall\varepsilon > 0,\;
  \exists\delta > 0,\;
  \forall x \in A:\;
  |x - c| < \delta \Rightarrow |f(x) - f(c)| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Continuity on a set is a pointwise condition: $\delta$ may depend on
both $\varepsilon$ and the base point $c$. This is the crucial
structural difference from uniform continuity, where a single $\delta$
must serve all $c \in B$ simultaneously. The pointwise condition is
logically weaker and admits functions whose local control deteriorates
as $c$ varies — for example, $f(x) = 1/x$ on $(0,1)$ is continuous
on $(0,1)$ but not uniformly so.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Algebra of Continuity at a Point
% ---------------------------------------------------------

\begin{theorem}[Algebra of Continuity at a Point]
\label{thm:continuity-algebra-at-a-point}
Let $A \subseteq \mathbb{R}$, let $f, g : A \to \mathbb{R}$, let
$b \in \mathbb{R}$, and let $c \in A$. Assume $f$ and $g$ are
continuous at $c$.
\begin{enumerate}[label=(\alph*)]
  \item The functions $f+g$, $f-g$, $fg$, and $bf$ are continuous
    at $c$.
  \item If $h : A \to \mathbb{R}$ is continuous at $c$ and
    $h(x) \neq 0$ for all $x \in A$, then $f/h$ is continuous
    at $c$.
\end{enumerate}

\smallskip
\noindent
\hyperref[prf:algebra-of-continuity-at-a-point]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{(a)}\quad
  \bigl(f \text{ cts at } c\bigr) \wedge
  \bigl(g \text{ cts at } c\bigr)
  \;\Rightarrow\;
  f+g,\; f-g,\; fg,\; bf \text{ cts at } c. \\[4pt]
&\text{(b)}\quad
  \bigl(f \text{ cts at } c\bigr) \wedge
  \bigl(h \text{ cts at } c\bigr) \wedge
  \bigl(\forall x \in A,\; h(x) \neq 0\bigr)
  \;\Rightarrow\;
  f/h \text{ cts at } c.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The proof reduces continuity at $c$ to the limit of the function
at $c$, then applies the algebra of limits of functions. Continuity
is preserved under all four arithmetic operations, with the natural
restriction that division requires the denominator to be nonzero
throughout $A$, not merely at $c$. This is stronger than what the
limit theorem requires (denominator nonzero only at $c$) because
$h$ must be well-defined as a function on all of $A$.
\end{remark*}

% Theorem — Algebra of Continuity on a Set
% ---------------------------------------------------------

\begin{theorem}[Algebra of Continuity on a Set]
\label{thm:continuity-algebra-on-a-set}
Let $A \subseteq \mathbb{R}$, let $f, g : A \to \mathbb{R}$, and
let $b \in \mathbb{R}$. Assume $f$ and $g$ are continuous on $A$.
\begin{enumerate}[label=(\alph*)]
  \item The functions $f+g$, $f-g$, $fg$, and $bf$ are continuous
    on $A$.
  \item If $h : A \to \mathbb{R}$ is continuous on $A$ and
    $h(x) \neq 0$ for all $x \in A$, then $f/h$ is continuous
    on $A$.
\end{enumerate}

\smallskip
\noindent
\hyperref[prf:algebra-of-continuity-on-a-set]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{(a)}\quad
  \bigl(f \text{ cts on } A\bigr) \wedge
  \bigl(g \text{ cts on } A\bigr)
  \;\Rightarrow\;
  f+g,\; f-g,\; fg,\; bf \text{ cts on } A. \\[4pt]
&\text{(b)}\quad
  \bigl(h \text{ cts on } A\bigr) \wedge
  \bigl(\forall x \in A,\; h(x) \neq 0\bigr)
  \;\Rightarrow\;
  f/h \text{ cts on } A.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
This is the pointwise theorem applied at an arbitrary $c \in A$. The
set-level statement follows immediately by universally quantifying
over $c$. It packages the algebraic stability of continuity into a
single statement about function spaces: the set $C(A)$ of continuous
functions on $A$ is closed under the standard algebraic operations.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Composition of Continuous Functions
% ---------------------------------------------------------

\begin{theorem}[Composition of Continuous Functions]
\label{thm:composition-of-continuous-functions}
Let $X, Y \subseteq \mathbb{R}$, let $f : X \to \mathbb{R}$ and
$g : Y \to \mathbb{R}$ with $f(X) \subseteq Y$. If $f$ is continuous
at $x_0 \in X$ and $g$ is continuous at $f(x_0) \in Y$, then the
composition $h = g \circ f : X \to \mathbb{R}$ is continuous at
$x_0$.

\smallskip
\noindent
\hyperref[prf:composition-is-continuous]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \bigl(f \text{ cts at } x_0\bigr)
  \;\wedge\;
  \bigl(g \text{ cts at } f(x_0)\bigr)
  \;\Rightarrow\;
  \bigl(g \circ f \text{ cts at } x_0\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Continuity is preserved by composition: closeness of $x$ to $x_0$
forces closeness of $f(x)$ to $f(x_0)$, which in turn forces
closeness of $g(f(x))$ to $g(f(x_0))$. The $\varepsilon$-$\delta$
proof makes this chain explicit: given $\varepsilon > 0$, extract
$\delta_1$ from continuity of $g$ at $f(x_0)$, then extract $\delta$
from continuity of $f$ at $x_0$ using $\delta_1$ as the new target.
The condition $f(X) \subseteq Y$ ensures $g$ is defined on the range
of $f$.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Local Properties of Continuous Functions
% ---------------------------------------------------------

\begin{theorem}[Local Properties of Continuous Functions]
\label{thm:local-properties-of-continuous-functions}
Let $E \subseteq \mathbb{R}$ and let $f : E \to \mathbb{R}$ be
continuous at $a \in E$. Then:
\begin{enumerate}[label=\arabic*$^\circ$]
  \item \textbf{Local boundedness.} $f$ is bounded on some
    neighborhood $U_E(a)$.
  \item \textbf{Local sign stability.} If $f(a) \neq 0$, then in
    some neighborhood $U_E(a)$, all values of $f$ have the same
    sign as $f(a)$.
  \item \textbf{Algebra at a point.} If $g : U_E(a) \to \mathbb{R}$
    is continuous at $a$, then $f+g$ and $f-g$ are continuous at
    $a$, and $f/g$ is continuous at $a$ provided $g(a) \neq 0$.
  \item \textbf{Composition.} If $g : Y \to \mathbb{R}$ is
    continuous at $b \in Y$ and $f : E \to Y$ satisfies
    $f(a) = b$, then $g \circ f$ is continuous at $a$.
\end{enumerate}

\smallskip
\noindent
\hyperref[prf:local-properties-of-continuous-functions]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
1^\circ\quad
  &\exists\,\delta_1 > 0,\; \exists M > 0:\;
   \forall x \in E,\; |x-a| < \delta_1
   \;\Rightarrow\; |f(x)| \leq M. \\[4pt]
2^\circ\quad
  &f(a) \neq 0
   \;\Rightarrow\;
   \exists\,\delta_2 > 0:\;
   \forall x \in E,\; |x-a| < \delta_2
   \;\Rightarrow\; \operatorname{sgn}(f(x)) = \operatorname{sgn}(f(a)).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Continuity at a point forces local regularity in four respects.
Part $1^\circ$ says the function cannot blow up near $a$. Part
$2^\circ$ is sign preservation: if $f(a) > 0$ (respectively $< 0$),
then $f$ remains positive (respectively negative) in a full
neighborhood. Parts $3^\circ$ and $4^\circ$ package the algebra and
composition theorems into the local setting. These four properties
are the standard toolkit for proving local consequences of continuity
in arguments where only behavior near $a$ is relevant.
\end{remark*}

% ---------------------------------------------------------
% Lemma — Power Function Is Continuous
% ---------------------------------------------------------

\begin{lemma}[Power Function Is Continuous]
\label{lem:power-function-continuous}
Let $n \in \mathbb{N}$. The function $f : \mathbb{R} \to \mathbb{R}$
defined by $f(x) = x^n$ is continuous.

\smallskip
\noindent
\hyperref[prf:power-function-is-continuous]{\textit{Go to proof.}}
\end{lemma}

\begin{remark*}[Standard quantified statement]
\[
  \forall n \in \mathbb{N},\;
  \forall x_0 \in \mathbb{R},\;
  \forall\varepsilon > 0,\;
  \exists\delta > 0,\;
  \forall x \in \mathbb{R}:\;
  |x - x_0| < \delta \Rightarrow |x^n - x_0^n| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Every monomial is continuous on $\mathbb{R}$. The proof factors
$x^n - x_0^n = (x - x_0)(x^{n-1} + x^{n-2}x_0 + \cdots +
x_0^{n-1})$ and bounds the second factor uniformly near $x_0$. This
lemma is the base case for the continuity of polynomials.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Polynomials Are Continuous
% ---------------------------------------------------------

\begin{theorem}[Any Real Polynomial Is Continuous]
\label{thm:polynomials-are-continuous}
Let $P(x) = a_0 + a_1 x + \cdots + a_n x^n$ be a real polynomial.
Then $P$ is continuous on $\mathbb{R}$.

\smallskip
\noindent
\hyperref[prf:polynomials-are-continuous]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \forall\, P \in \mathbb{R}[x],\;
  \forall x_0 \in \mathbb{R},\;
  \forall\varepsilon > 0,\;
  \exists\delta > 0,\;
  \forall x \in \mathbb{R}:\;
  |x - x_0| < \delta \Rightarrow |P(x) - P(x_0)| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Polynomials are continuous because the class of continuous functions
is closed under addition and scalar multiplication, and each monomial
$x^k$ is continuous by the Power Function lemma. The argument is
purely structural: apply the algebra of continuous functions finitely
many times. This result, combined with the quotient rule, yields the
continuity of rational functions on their domains.
\end{remark*}

% ---------------------------------------------------------
% Proposition — Rational Functions Are Continuous on Their Domain
% ---------------------------------------------------------

\begin{proposition}[Rational Functions Are Continuous on Their Domain]
\label{prop:rational-functions-continuous}
Let $P$ and $Q$ be real polynomials and let
$D = \{\, x \in \mathbb{R} \mid Q(x) \neq 0 \,\}$. The rational
function $R(x) = P(x)/Q(x)$ is continuous on $D$.

\smallskip
\noindent
\hyperref[prf:rational-function-continuous-on-domain]{\textit{Go to proof.}}
\end{proposition}

\begin{remark*}[Standard quantified statement]
\[
  \forall x_0 \in D,\;
  \forall\varepsilon > 0,\;
  \exists\delta > 0,\;
  \forall x \in D:\;
  |x - x_0| < \delta \Rightarrow |R(x) - R(x_0)| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Both $P$ and $Q$ are continuous everywhere; at every point of $D$
the denominator is nonzero, so the quotient rule for continuous
functions applies. The domain $D$ is open in $\mathbb{R}$ since $Q$
is continuous and $\{0\}$ is closed, so $D = Q^{-1}(\mathbb{R}
\setminus \{0\})$ is the preimage of an open set under a continuous
function.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Square Root Preserves Continuity
% ---------------------------------------------------------

\begin{theorem}[Square Root Preserves Continuity]
\label{thm:square-root-preserves-continuity}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$ with
$f(x) \geq 0$ for all $x \in A$, and define
$(\sqrt{f})(x) := \sqrt{f(x)}$ for $x \in A$.
\begin{enumerate}[label=(\alph*)]
  \item If $f$ is continuous at $c \in A$, then $\sqrt{f}$ is
    continuous at $c$.
  \item If $f$ is continuous on $A$, then $\sqrt{f}$ is continuous
    on $A$.
\end{enumerate}

\smallskip
\noindent
\hyperref[prf:square-root-preserves-continuity]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \bigl(f \text{ cts at } c,\; f \geq 0\bigr)
  \;\Rightarrow\;
  \bigl(\sqrt{f} \text{ cts at } c\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The square root operation preserves continuity when the function
is nonneg\-a\-tive-valued. Part (b) follows from part (a) by
pointwise application. The proof of part (a) treats the cases
$f(c) > 0$ and $f(c) = 0$ separately: when $f(c) > 0$, rationalize
$|\sqrt{f(x)} - \sqrt{f(c)}| = |f(x) - f(c)| / (\sqrt{f(x)} +
\sqrt{f(c)})$ and bound the denominator away from zero; when
$f(c) = 0$, use $|\sqrt{f(x)}| = \sqrt{f(x)} \leq \sqrt{\varepsilon}$
once $|f(x)| < \varepsilon$.
\end{remark*}

% ---------------------------------------------------------
% Proposition — Exponentiation Is Continuous (base variable)
% ---------------------------------------------------------

\begin{proposition}[Exponentiation Is Continuous, I]
\label{prop:exponentiation-continuous-base}
Let $a > 0$. The function $f : \mathbb{R} \to \mathbb{R}$ defined
by $f(x) := a^x$ is continuous.

\smallskip
\noindent
\hyperref[prf:exponentiation-is-continuous-1]{\textit{Go to proof.}}
\end{proposition}

\begin{remark*}[Standard quantified statement]
\[
  \forall a > 0,\;
  \forall x_0 \in \mathbb{R},\;
  \forall\varepsilon > 0,\;
  \exists\delta > 0,\;
  \forall x \in \mathbb{R}:\;
  |x - x_0| < \delta \Rightarrow |a^x - a^{x_0}| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Interpretation]
For a fixed positive base $a$, real exponentiation in the exponent
variable is continuous on all of $\mathbb{R}$. The proof uses the
construction of real exponentiation and the continuity properties
established in that context. This is part of the foundational
justification that the exponential function behaves as expected.
\end{remark*}

% Proposition — Exponentiation Is Continuous (exponent variable)
% ---------------------------------------------------------

\begin{proposition}[Exponentiation Is Continuous, II]
\label{prop:exponentiation-continuous-exponent}
Let $p \in \mathbb{R}$. The function $f : (0,\infty) \to \mathbb{R}$
defined by $f(x) := x^p$ is continuous.

\smallskip
\noindent
\hyperref[prf:exponentiation-is-continuous-1]{\textit{Go to proof.}}
\end{proposition}

\begin{remark*}[Standard quantified statement]
\[
  \forall p \in \mathbb{R},\;
  \forall x_0 > 0,\;
  \forall\varepsilon > 0,\;
  \exists\delta > 0,\;
  \forall x > 0:\;
  |x - x_0| < \delta \Rightarrow |x^p - x_0^p| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Interpretation]
For a fixed real exponent $p$, exponentiation in the base variable
is continuous on the positive reals. Together with Proposition
\hyperref[prop:exponentiation-continuous-base]{I}, this establishes
that the two-variable function $(a,x) \mapsto a^x$ is separately
continuous in each variable on its natural domain.
\end{remark*}

% ---------------------------------------------------------
% Proposition — Continuous Extension to a Closed Interval
% ---------------------------------------------------------

\begin{proposition}[Continuous Extension to a Closed Interval]
\label{prop:continuous-extension-closed-interval}
Let $f : (a,b] \to \mathbb{R}$ be continuous. Suppose
$L := \lim_{x \to a^+} f(x)$ exists. Define
$\widetilde{f} : [a,b] \to \mathbb{R}$ by
\[
  \widetilde{f}(x) :=
  \begin{cases}
    L, & x = a, \\
    f(x), & x \in (a,b].
  \end{cases}
\]
Then $\widetilde{f}$ is continuous on $[a,b]$.

\smallskip
\noindent
\hyperref[prf:continuous-extension-to-closed-interval]{\textit{Go to proof.}}
\end{proposition}

\begin{remark*}[Standard quantified statement]
\[
  \Bigl(\lim_{x \to a^+} f(x) = L \in \mathbb{R}\Bigr)
  \;\Rightarrow\;
  \widetilde{f} \text{ is continuous on } [a,b].
\]
\end{remark*}

\begin{remark*}[Interpretation]
If the right-hand limit at the missing endpoint exists, the function
extends continuously to that endpoint by assignment. Continuity of
$\widetilde{f}$ away from $a$ is inherited from $f$. Continuity at
$a$ follows from the definition of $\widetilde{f}(a) = L$ and the
one-sided limit. This proposition is used in the proof that uniformly
continuous functions on open intervals extend to their closure.
\end{remark*}

% ---------------------------------------------------------
% Definition — Space of Continuous Functions
% ---------------------------------------------------------

\begin{definition}[Space of Continuous Functions]
\label{def:space-of-continuous-functions}
Let $X \subseteq \mathbb{R}$. The \textbf{space of continuous
functions on $X$} is
\[
  C^0(X;\mathbb{R}) \;:=\;
  \{\, f : X \to \mathbb{R} : f \text{ is continuous on } X \,\}.
\]
When the codomain is clear from context, this space is denoted
$C^0(X)$.
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
  f \in C^0(X)
  \;\iff\;
  \forall c \in X,\;
  \forall\varepsilon > 0,\;
  \exists\delta > 0,\;
  \forall x \in X:\;
  |x - c| < \delta \Rightarrow |f(x) - f(c)| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The set $C^0(X;\mathbb{R})$ collects all continuous real-valued
functions on the fixed domain $X$ into a single mathematical object.
It is closed under pointwise addition and scalar multiplication, so
it forms a real vector space. On compact $X$, it carries the
uniform norm $\|f\|_\infty = \sup_{x \in X} |f(x)|$, making it a
Banach space. This is the setting for the Weierstrass approximation
theorem and for the study of function sequences.
\end{remark*}

% ---------------------------------------------------------
% Proposition — Sign Preservation
% ---------------------------------------------------------

\begin{proposition}[Sign Preservation]
\label{prop:sign-preservation}
Let $f : E \to \mathbb{R}$ be continuous at $x_0 \in E$. If
$f(x_0) > 0$, then there exists $\delta > 0$ such that
\[
  f(x) > 0 \quad \text{for all } x \in E \text{ with }
  |x - x_0| < \delta.
\]
Analogously, if $f(x_0) < 0$, then there exists $\delta > 0$
such that $f(x) < 0$ for all $x \in E$ with $|x - x_0| < \delta$.

\smallskip
\noindent
\hyperref[prf:sign-preservation]{\textit{Go to proof.}}
\end{proposition}

\begin{remark*}[Standard quantified statement]
\[
  \bigl(f \text{ cts at } x_0\bigr) \wedge
  \bigl(f(x_0) > 0\bigr)
  \;\Rightarrow\;
  \exists\,\delta > 0,\;
  \forall x \in E:\;
  |x - x_0| < \delta \Rightarrow f(x) > 0.
\]
\end{remark*}

\begin{remark*}[Contrapositive quantified statement]
\[
  \bigl(\forall\delta > 0,\;\exists x \in E:\;
    |x - x_0| < \delta \wedge f(x) \leq 0\bigr)
  \;\Rightarrow\;
  \bigl(f(x_0) \leq 0 \;\vee\; f \text{ not cts at } x_0\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A continuous function that is positive (respectively negative) at a
point remains positive (respectively negative) throughout some
neighborhood of that point. The proof applies the $\varepsilon$-$\delta$
definition with $\varepsilon = f(x_0)/2 > 0$: the resulting $\delta$
forces $|f(x) - f(x_0)| < f(x_0)/2$, which gives $f(x) > f(x_0)/2
> 0$. Sign preservation is the key ingredient in the proof of the
Intermediate Value Theorem and in the argument that continuous
functions on compact sets attain their extreme values.
\end{remark*}
