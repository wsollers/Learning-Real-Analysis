\section{Continuity}
\label{sec:continuity}

% ---------------------------------------------------------
% Toolkit
% ---------------------------------------------------------

\begin{tcolorbox}[title={Toolkit --- Continuity}]
\begin{itemize}
  \item Three equivalent forms: $\varepsilon$-$\delta$, neighbourhood,
    sequential. The $\varepsilon$-$\delta$ form uses
    $\operatorname{Within}(x,c,\delta)$ — \emph{unpunctured}.
    This is the critical structural difference from function limits,
    which use $\operatorname{InputTolerance}(x,c,\delta)$ (punctured).
  \item $\delta$ may depend on both $\varepsilon$ and the base point
    $c$. When $\delta$ depends only on $\varepsilon$, the condition
    is uniform continuity.
  \item Isolated points are automatically continuous.
  \item Algebra: sums, differences, products, scalar multiples, and
    quotients (denominator nonzero) of continuous functions are
    continuous.
  \item Composition preserves continuity: $f$ cts at $c$, $g$ cts at
    $f(c)$ $\Rightarrow$ $g \circ f$ cts at $c$.
\end{itemize}
\end{tcolorbox}

% ---------------------------------------------------------
% Definition — Relative Neighborhood
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Relative Neighborhood}]
\begin{definition}[Relative Neighborhood]
\label{def:relative-neighborhood}
Let $E \subseteq \mathbb{R}$ and let $a \in E$. A
\textbf{neighborhood of $a$ in $E$}, denoted $U_E(a)$, is a set of
the form
\[
  U_E(a) \;:=\; E \cap U(a),
\]
where $U(a)$ is a neighborhood of $a$ in $\mathbb{R}$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  U_E(a) = E \cap (a - \delta,\, a + \delta)
  \;=\;
  \{\, x \in E : \operatorname{Within}(x, a, \delta) \,\}
  \quad\text{for some } \delta > 0.
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{AdherentPoint}(a, E)
  \;\wedge\;
  U_E(a) = E \cap (a-\delta, a+\delta)
  \;\colonequiv\;
  \{\, x \in E : \operatorname{Within}(x, a, \delta) \,\}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The relative neighborhood restricts a full $\delta$-ball to the
domain $E$. Properties stated in terms of $U_E(a)$ are local
properties of $f$ on $E$ near $a$, unaffected by what $\mathbb{R}$
looks like outside $E$. This is the correct setting for continuity
on proper subsets.
\end{remark*}

% ---------------------------------------------------------
% Definition — Continuity at a Point (epsilon-delta)
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Continuity at a Point}]
\begin{definition}[Continuity at a Point]
\label{def:continuous-at-point}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$c \in A$. The function $f$ is \textbf{continuous at $c$} if
\[
  \ForallOT{\varepsilon},\;
  \ExistsIT{\delta},\;
  \forall x \in A :\;
  \operatorname{Within}(x, c, \delta)
  \;\Rightarrow\;
  \operatorname{OutputTolerance}(f(x), f(c), \varepsilon).
\]
If $f$ is not continuous at $c$, then $f$ is \textbf{discontinuous
at $c$}.

\smallskip\noindent
\hyperref[prf:equivalent-forms-continuity]{\textit{Go to proof.}}
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ContinuousAt}(f, c)
  \;\iff\;
  \forall\varepsilon > 0,\;
  \exists\delta > 0,\;
  \forall x \in A :\;
  |x - c| < \delta \;\Rightarrow\; |f(x) - f(c)| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{ContinuousAt}(f, c)
  \;\colonequiv\;
  \ForallOT{\varepsilon},\;
  \ExistsIT{\delta},\;
  \forall x \in A :\;
  \operatorname{Within}(x, c, \delta)
  \;\Rightarrow\;
  \operatorname{OutputTolerance}(f(x), f(c), \varepsilon).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \operatorname{DiscontinuousAt}(f, c)
  \;\iff\;
  \exists\varepsilon_0 > 0,\;
  \forall\delta > 0,\;
  \exists x \in A :\;
  \operatorname{Within}(x, c, \delta)
  \;\wedge\;
  \neg\,\operatorname{OutputTolerance}(f(x), f(c), \varepsilon_0).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  \operatorname{DiscontinuousAt}(f, c)
  \;\colonequiv\;
  \exists\varepsilon_0 > 0,\;
  \forall\delta > 0,\;
  \exists x \in A :\;
  |x - c| < \delta \;\wedge\; |f(x) - f(c)| \geq \varepsilon_0.
\]
\end{remark*}

\begin{remark*}[Failure mode decomposition]
Discontinuity at $c$ decomposes into three structurally distinct
cases:
\[
  \underbrace{%
    \lim_{x\to c} f(x) \text{ exists but } \neq f(c)
  }_{\text{removable}}
  \;\vee\;
  \underbrace{%
    f(c^-), f(c^+) \text{ exist but are unequal}
  }_{\text{jump}}
  \;\vee\;
  \underbrace{%
    \text{no one-sided limit exists}
  }_{\text{essential}}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Three features distinguish $\operatorname{ContinuousAt}(f,c)$ from
$\operatorname{FunctionLimit}(f,c,L)$. First, the input condition
is $\operatorname{Within}$ (unpunctured): $x = c$ is permitted, so
the behavior at the point itself is included. Second, the target is
$f(c)$ — not an independently specified $L$. Third, $c$ must belong
to $A$. At a cluster point, continuity is equivalent to
$\operatorname{FunctionLimit}(f,c,f(c))$. At an isolated point,
$\operatorname{Within}(x,c,\delta) \cap A = \{c\}$ for small
$\delta$, so the condition holds vacuously — every isolated point is
a continuity point.
\end{remark*}

% ---------------------------------------------------------
% Definition — Continuity at a Point (neighbourhood form)
% ---------------------------------------------------------

\begin{definition}[Continuity at a Point --- Neighbourhood Form]
\label{def:continuous-at-point-nbhd}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$c \in A$. The function $f$ is \textbf{continuous at $c$} in the
neighbourhood sense if for every $\varepsilon$-neighbourhood
$V_\varepsilon(f(c))$ of $f(c)$, there exists a $\delta$-neighbourhood
$V_\delta(c)$ of $c$ such that
\[
  f\!\bigl(A \cap V_\delta(c)\bigr) \;\subseteq\; V_\varepsilon(f(c)).
\]
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ContinuousAt}(f,c)\text{ (nbhd)}
  \;\iff\;
  \ForallOT{\varepsilon},\;
  \ExistsIT{\delta},\;
  \forall x \in A \cap V_\delta(c) :\;
  f(x) \in V_\varepsilon(f(c)).
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{ContinuousAt}(f,c)\text{ (nbhd)}
  \;\colonequiv\;
  \ForallOT{\varepsilon},\;
  \ExistsIT{\delta},\;
  f\bigl(A \cap V_\delta(c)\bigr) \subseteq V_\varepsilon(f(c)).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The neighbourhood form uses the full ball $V_\delta(c)$ — not the
punctured $V_\delta^*(c)$ required by the limit definition. This is
the algebraic trace of the $\operatorname{Within}$ vs
$\operatorname{InputTolerance}$ distinction: continuity includes the
base point $c$ in the input set; limits do not. The neighbourhood
form is the version most naturally extended to metric and topological
spaces.
\end{remark*}

% ---------------------------------------------------------
% Definition — Continuity at a Point (sequential form)
% ---------------------------------------------------------

\begin{definition}[Continuity at a Point --- Sequential Form]
\label{def:continuous-at-point-seq}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$c \in A$. The function $f$ is \textbf{continuous at $c$} in the
sequential sense if for every sequence $(x_n) \subseteq A$ with
$x_n \to c$, the sequence $(f(x_n))$ converges to $f(c)$.
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ContinuousAt}(f,c)\text{ (seq)}
  \;\iff\;
  \forall (x_n) \subseteq A :\;
  x_n \to c \;\Rightarrow\; f(x_n) \to f(c).
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{ContinuousAt}(f,c)\text{ (seq)}
  \;\colonequiv\;
  \forall (x_n) \subseteq A :\;
  \operatorname{ConvergentSequence}(x_n, c)
  \;\Rightarrow\;
  \operatorname{ConvergentSequence}(f(x_n), f(c)).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \operatorname{DiscontinuousAt}(f,c)\text{ (seq)}
  \;\iff\;
  \exists (x_n) \subseteq A :\;
  x_n \to c \;\wedge\; f(x_n) \not\to f(c).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The sequential form of continuity differs from the sequential form
of function limits in one precise way: sequences $(x_n)$ may include
$x_n = c$. The limit sequential form requires $(x_n) \subseteq A
\setminus \{c\}$. This mirrors the $\operatorname{Within}$ vs
$\operatorname{InputTolerance}$ distinction exactly. The negation —
one witness sequence that converges but whose image does not — is the
standard tool for disproving continuity at a point.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Three Equivalent Forms of Continuity
% ---------------------------------------------------------

\begin{theorem}[Three Equivalent Forms of Continuity]
\label{thm:continuity-three-equiv}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$c \in A$. The following are equivalent:
\begin{enumerate}[label=(\roman*)]
  \item $\ForallOT{\varepsilon},\;\ExistsIT{\delta},\;\forall x\in A:\;
    \operatorname{Within}(x,c,\delta) \Rightarrow
    \operatorname{OutputTolerance}(f(x),f(c),\varepsilon)$.
    \hfill ($\varepsilon$-$\delta$)
  \item $\forall\,V_\varepsilon(f(c)),\;\exists\,V_\delta(c):\;
    f(A \cap V_\delta(c)) \subseteq V_\varepsilon(f(c))$.
    \hfill (neighbourhood)
  \item $\forall (x_n)\subseteq A:\;
    x_n \to c \Rightarrow f(x_n) \to f(c)$.
    \hfill (sequential)
\end{enumerate}

\smallskip\noindent
\hyperref[prf:equivalent-forms-continuity]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
(i)\quad
  &\forall\varepsilon>0,\;
   \exists\delta>0,\;
   \forall x\in A:\;
   |x-c|<\delta \Rightarrow |f(x)-f(c)|<\varepsilon. \\[3pt]
(ii)\quad
  &\forall\varepsilon>0,\;\exists\delta>0:\;
   f(A\cap V_\delta(c)) \subseteq V_\varepsilon(f(c)). \\[3pt]
(iii)\quad
  &\forall (x_n)\subseteq A:\;
   x_n\to c \Rightarrow f(x_n)\to f(c).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \operatorname{DiscontinuousAt}(f,c)
  \;\iff\;
  \exists (x_n)\subseteq A :\;
  x_n \to c \;\wedge\; f(x_n) \not\to f(c).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The equivalence (i)$\iff$(ii) is notational: $V_\delta(c)$ is the
ball $\{|x-c|<\delta\}$ and $V_\varepsilon(f(c))$ is the ball
$\{|y-f(c)|<\varepsilon\}$, so (i) and (ii) state the same condition
in two notations. The equivalence (i)$\iff$(iii) mirrors the
Sequential Criterion for limits but with $(x_n) \subseteq A$ (not
$A \setminus \{c\}$): the base point is included. The proof of
(iii)$\Rightarrow$(i) is by contrapositive: negate (i) to get
$\varepsilon_0$ and build $(x_n)$ via $\delta = 1/n$ — a sequence
converging to $c$ whose image does not converge to $f(c)$.
\end{remark*}

% ---------------------------------------------------------
% Definition — Discontinuity Criterion
% ---------------------------------------------------------

\begin{definition}[Discontinuity Criterion]
\label{def:discontinuity-criterion}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$c \in A$. The function $f$ is \textbf{discontinuous at $c$} if and
only if
\[
  \exists (x_n) \subseteq A :\;
  x_n \to c \;\wedge\; f(x_n) \not\to f(c).
\]
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{DiscontinuousAt}(f,c)
  \;\iff\;
  \exists (x_n) \subseteq A :\;
  \operatorname{ConvergentSequence}(x_n, c)
  \;\wedge\;
  \neg\,\operatorname{ConvergentSequence}(f(x_n), f(c)).
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  \operatorname{DiscontinuousAt}(f,c)
  \;\colonequiv\;
  \neg\,\operatorname{ContinuousAt}(f,c).
\]
\end{remark*}

\begin{remark*}[Interpretation]
One witness sequence is sufficient: it converges in the domain but
its image fails to converge to the function value. The criterion is
the operative tool for proving discontinuity — exhibit one bad
sequence and the work is done.
\end{remark*}

% ---------------------------------------------------------
% Definition — Continuity on a Set
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Continuity on a Set}]
\begin{definition}[Continuity on a Set]
\label{def:continuity-on-set}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$, and let
$B \subseteq A$. The function $f$ is \textbf{continuous on $B$} if
$\operatorname{ContinuousAt}(f, c)$ for every $c \in B$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ContinuousOn}(f, B)
  \;\iff\;
  \forall c \in B,\;
  \ForallOT{\varepsilon},\;
  \ExistsIT{\delta},\;
  \forall x \in A :\;
  \operatorname{Within}(x,c,\delta)
  \Rightarrow
  \operatorname{OutputTolerance}(f(x),f(c),\varepsilon).
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  \operatorname{ContinuousOn}(f, B)
  \;\colonequiv\;
  \forall c \in B :\;
  \operatorname{ContinuousAt}(f, c).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \neg\,\operatorname{ContinuousOn}(f, B)
  \;\iff\;
  \exists c \in B :\;
  \operatorname{DiscontinuousAt}(f, c).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Continuity on a set is a pointwise condition: $\delta$ may depend on
both $\varepsilon$ and the base point $c \in B$. This is the
essential structural difference from $\operatorname{UniformlyContinuous}
(f,B)$, where a single $\delta$ must serve all $c \in B$
simultaneously. The pointwise condition admits functions whose local
control deteriorates as $c$ varies — $f(x) = 1/x$ on $(0,1)$ is
$\operatorname{ContinuousOn}$ but not $\operatorname{UniformlyContinuous}$.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Algebra of Continuity at a Point
% ---------------------------------------------------------

\begin{theorem}[Algebra of Continuity at a Point]
\label{thm:continuity-algebra-at-a-point}
Let $A \subseteq \mathbb{R}$, let $f, g : A \to \mathbb{R}$, let
$b \in \mathbb{R}$, and let $c \in A$. If
$\operatorname{ContinuousAt}(f,c)$ and $\operatorname{ContinuousAt}(g,c)$,
then:
\begin{enumerate}[label=(\alph*)]
  \item $\operatorname{ContinuousAt}(f+g, c)$,
    $\operatorname{ContinuousAt}(f-g, c)$,
    $\operatorname{ContinuousAt}(fg, c)$,
    $\operatorname{ContinuousAt}(bf, c)$.
  \item If $\operatorname{ContinuousAt}(h,c)$ and $h(x) \neq 0$ for
    all $x \in A$, then $\operatorname{ContinuousAt}(f/h, c)$.
\end{enumerate}

\smallskip\noindent
\hyperref[prf:algebra-of-continuity-at-a-point]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{(a)}\quad
  \operatorname{ContinuousAt}(f,c)
  \wedge \operatorname{ContinuousAt}(g,c)
  \;\Rightarrow\;
  \operatorname{ContinuousAt}(f+g,c),\;
  \operatorname{ContinuousAt}(fg,c),\;
  \operatorname{ContinuousAt}(bf,c). \\[4pt]
&\text{(b)}\quad
  \operatorname{ContinuousAt}(h,c)
  \wedge (\forall x \in A,\; h(x)\neq 0)
  \;\Rightarrow\;
  \operatorname{ContinuousAt}(f/h,c).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The proof reduces $\operatorname{ContinuousAt}(f,c)$ to
$\operatorname{FunctionLimit}(f,c,f(c))$, then applies the algebra
of function limits. Part (b) requires $h(x) \neq 0$ throughout $A$
— not merely at $c$ — because $h$ must be well-defined as a function
on $A$. This is a stronger hypothesis than the limit quotient rule,
which only needs $h(c) \neq 0$.
\end{remark*}

% Theorem — Algebra of Continuity on a Set
% ---------------------------------------------------------

\begin{theorem}[Algebra of Continuity on a Set]
\label{thm:continuity-algebra-on-a-set}
Let $A \subseteq \mathbb{R}$, let $f, g : A \to \mathbb{R}$, and
let $b \in \mathbb{R}$. If $\operatorname{ContinuousOn}(f,A)$ and
$\operatorname{ContinuousOn}(g,A)$, then:
\begin{enumerate}[label=(\alph*)]
  \item $\operatorname{ContinuousOn}(f+g,A)$,
    $\operatorname{ContinuousOn}(f-g,A)$,
    $\operatorname{ContinuousOn}(fg,A)$,
    $\operatorname{ContinuousOn}(bf,A)$.
  \item If $\operatorname{ContinuousOn}(h,A)$ and $h(x) \neq 0$ for
    all $x \in A$, then $\operatorname{ContinuousOn}(f/h,A)$.
\end{enumerate}

\smallskip\noindent
\hyperref[prf:algebra-of-continuity-on-a-set]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ContinuousOn}(f,A)
  \wedge \operatorname{ContinuousOn}(g,A)
  \;\Rightarrow\;
  \operatorname{ContinuousOn}(f+g,A),\;
  \operatorname{ContinuousOn}(fg,A).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The pointwise theorem applied at an arbitrary $c \in A$. The set
$C^0(A)$ of continuous functions on $A$ is closed under the
standard ring operations: it is a subalgebra of $\mathbb{R}^A$.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Composition of Continuous Functions
% ---------------------------------------------------------

\begin{theorem}[Composition of Continuous Functions]
\label{thm:composition-of-continuous-functions}
Let $X, Y \subseteq \mathbb{R}$, let $f : X \to \mathbb{R}$ and
$g : Y \to \mathbb{R}$ with $f(X) \subseteq Y$. If
$\operatorname{ContinuousAt}(f, x_0)$ and
$\operatorname{ContinuousAt}(g, f(x_0))$, then
$\operatorname{ContinuousAt}(g \circ f, x_0)$.

\smallskip\noindent
\hyperref[prf:composition-is-continuous]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ContinuousAt}(f,x_0)
  \;\wedge\;
  \operatorname{ContinuousAt}(g,f(x_0))
  \;\Rightarrow\;
  \operatorname{ContinuousAt}(g \circ f,\, x_0).
\]
\end{remark*}

\begin{remark*}[Contrapositive quantified statement]
\[
  \operatorname{DiscontinuousAt}(g \circ f,\, x_0)
  \;\Rightarrow\;
  \operatorname{DiscontinuousAt}(f, x_0)
  \;\vee\;
  \operatorname{DiscontinuousAt}(g, f(x_0)).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Continuity is preserved by composition. The proof chains two
$\operatorname{Within}$ implications: $\operatorname{Within}(x,x_0,\delta)$
forces $\operatorname{Within}(f(x),f(x_0),\delta_1)$ by continuity
of $f$, which forces $\operatorname{OutputTolerance}(g(f(x)),
g(f(x_0)),\varepsilon)$ by continuity of $g$. The condition
$f(X) \subseteq Y$ ensures $g$ is defined on the range of $f$.
Note the contrast with $\operatorname{FunctionLimit}$ composition,
which required the extra hypothesis $g(c_2) = L$: here no such
hypothesis is needed because continuity already includes the base
point.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Local Properties of Continuous Functions
% ---------------------------------------------------------

\begin{theorem}[Local Properties of Continuous Functions]
\label{thm:local-properties-of-continuous-functions}
Let $E \subseteq \mathbb{R}$ and let $f : E \to \mathbb{R}$ with
$\operatorname{ContinuousAt}(f, a)$. Then:
\begin{enumerate}[label=\arabic*$^\circ$]
  \item \textbf{Local boundedness.}
    $\operatorname{BoundedNear}(f, a)$: $f$ is bounded on some
    neighborhood $U_E(a)$.
  \item \textbf{Sign stability.} If $f(a) \neq 0$, then
    $\operatorname{sgn}(f(x)) = \operatorname{sgn}(f(a))$ for all
    $x$ in some $U_E(a)$.
  \item \textbf{Algebra.} If $\operatorname{ContinuousAt}(g, a)$
    for $g : U_E(a) \to \mathbb{R}$, then $f+g$, $f-g$ are cts at
    $a$, and $f/g$ is cts at $a$ when $g(a) \neq 0$.
  \item \textbf{Composition.} If $\operatorname{ContinuousAt}(g, f(a))$
    for $g : Y \to \mathbb{R}$ with $f(E) \subseteq Y$, then
    $\operatorname{ContinuousAt}(g \circ f, a)$.
\end{enumerate}

\smallskip\noindent
\hyperref[prf:local-properties-of-continuous-functions]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
1^\circ\quad
  &\operatorname{ContinuousAt}(f,a)
  \;\Rightarrow\;
  \operatorname{BoundedNear}(f,a). \\[3pt]
2^\circ\quad
  &\operatorname{ContinuousAt}(f,a) \wedge f(a)\neq 0
  \;\Rightarrow\;
  \exists\,\delta>0,\;\forall x\in E:\;
  \operatorname{Within}(x,a,\delta)
  \Rightarrow \operatorname{sgn}(f(x))=\operatorname{sgn}(f(a)).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
These four local consequences of $\operatorname{ContinuousAt}(f,a)$
package the most-used tools in a single theorem. Part $1^\circ$
prevents blow-up near $a$. Part $2^\circ$ is sign preservation — the
key ingredient in the IVT and EVT proofs. Parts $3^\circ$ and
$4^\circ$ are the pointwise algebra and composition theorems
re-stated as local consequences. The theorem is load-bearing in all
arguments that reason about behavior near a single point.
\end{remark*}

% ---------------------------------------------------------
% Lemma — Power Function Is Continuous
% ---------------------------------------------------------

\begin{lemma}[Power Function Is Continuous]
\label{lem:power-function-continuous}
Let $n \in \mathbb{N}$. The function $f(x) = x^n$ satisfies
$\operatorname{ContinuousOn}(f, \mathbb{R})$.

\smallskip\noindent
\hyperref[prf:power-function-is-continuous]{\textit{Go to proof.}}
\end{lemma}

\begin{remark*}[Standard quantified statement]
\[
  \forall n \in \mathbb{N},\;
  \forall x_0 \in \mathbb{R},\;
  \ForallOT{\varepsilon},\;
  \ExistsIT{\delta},\;
  \forall x \in \mathbb{R} :\;
  \operatorname{Within}(x,x_0,\delta)
  \Rightarrow
  \operatorname{OutputTolerance}(x^n, x_0^n, \varepsilon).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Every monomial is continuous on $\mathbb{R}$. The proof factors
$x^n - x_0^n = (x-x_0)(x^{n-1}+x^{n-2}x_0+\cdots+x_0^{n-1})$ and
bounds the second factor uniformly near $x_0$. This lemma is the
base case for the continuity of polynomials via the algebra of
continuous functions.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Polynomials Are Continuous
% ---------------------------------------------------------

\begin{theorem}[Any Real Polynomial Is Continuous]
\label{thm:polynomials-are-continuous}
Let $P(x) = a_0 + a_1 x + \cdots + a_n x^n$ be a real polynomial.
Then $\operatorname{ContinuousOn}(P, \mathbb{R})$.

\smallskip\noindent
\hyperref[prf:polynomials-are-continuous]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \forall P \in \mathbb{R}[x],\;
  \forall x_0 \in \mathbb{R},\;
  \ForallOT{\varepsilon},\;
  \ExistsIT{\delta},\;
  \forall x \in \mathbb{R} :\;
  \operatorname{Within}(x,x_0,\delta)
  \Rightarrow
  \operatorname{OutputTolerance}(P(x),P(x_0),\varepsilon).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Polynomials are continuous because $C^0(\mathbb{R})$ is closed under
addition and scalar multiplication, and each monomial $x^k$ is in
$C^0(\mathbb{R})$ by the power function lemma. The argument is
structural: apply the algebra of continuous functions finitely many
times.
\end{remark*}

% ---------------------------------------------------------
% Proposition — Rational Functions Are Continuous on Their Domain
% ---------------------------------------------------------

\begin{proposition}[Rational Functions Are Continuous on Their Domain]
\label{prop:rational-functions-continuous}
Let $P, Q$ be real polynomials and let
$D = \{x \in \mathbb{R} : Q(x) \neq 0\}$.
Then $\operatorname{ContinuousOn}(P/Q, D)$.

\smallskip\noindent
\hyperref[prf:rational-function-continuous-on-domain]{\textit{Go to proof.}}
\end{proposition}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ContinuousOn}(P/Q,\, D)
  \;\colonequiv\;
  \forall x_0 \in D,\;
  \ForallOT{\varepsilon},\;
  \ExistsIT{\delta},\;
  \forall x \in D :\;
  \operatorname{Within}(x,x_0,\delta)
  \Rightarrow
  \operatorname{OutputTolerance}(R(x),R(x_0),\varepsilon).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Both $P$ and $Q$ are in $C^0(\mathbb{R})$; at every $x_0 \in D$ the
denominator is nonzero, so the quotient rule for continuous functions
applies. The domain $D$ is open: $Q$ is continuous and $D =
Q^{-1}(\mathbb{R} \setminus \{0\})$ is the preimage of an open set.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Square Root Preserves Continuity
% ---------------------------------------------------------

\begin{theorem}[Square Root Preserves Continuity]
\label{thm:square-root-preserves-continuity}
Let $A \subseteq \mathbb{R}$, let $f : A \to \mathbb{R}$ with
$f \geq 0$ on $A$, and define $\sqrt{f}(x) := \sqrt{f(x)}$.
\begin{enumerate}[label=(\alph*)]
  \item $\operatorname{ContinuousAt}(f,c) \Rightarrow
    \operatorname{ContinuousAt}(\sqrt{f},c)$.
  \item $\operatorname{ContinuousOn}(f,A) \Rightarrow
    \operatorname{ContinuousOn}(\sqrt{f},A)$.
\end{enumerate}

\smallskip\noindent
\hyperref[prf:square-root-preserves-continuity]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ContinuousAt}(f,c) \wedge f \geq 0
  \;\Rightarrow\;
  \operatorname{ContinuousAt}(\sqrt{f},c).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The square root operation preserves continuity on nonnegative-valued
functions. The proof splits on whether $f(c) > 0$ or $f(c) = 0$.
For $f(c) > 0$: rationalize $|\sqrt{f(x)} - \sqrt{f(c)}| =
|f(x)-f(c)| / (\sqrt{f(x)}+\sqrt{f(c)})$ and bound the denominator
away from zero. For $f(c) = 0$: use $\sqrt{f(x)} \leq
\sqrt{\varepsilon}$ once $|f(x)| < \varepsilon$.
\end{remark*}

and the continuity properties established in the elementary functions
chapter.
\end{remark*}

% Proposition — Continuous Extension to a Closed Interval
% ---------------------------------------------------------

\begin{proposition}[Continuous Extension to a Closed Interval]
\label{prop:continuous-extension-closed-interval}
Let $\operatorname{ContinuousOn}(f, (a,b])$ and suppose
$L := \lim_{x \to a^+} f(x)$ exists. Define $\widetilde{f} :
[a,b] \to \mathbb{R}$ by $\widetilde{f}(a) := L$ and
$\widetilde{f}(x) := f(x)$ for $x \in (a,b]$. Then
$\operatorname{ContinuousOn}(\widetilde{f}, [a,b])$.

\smallskip\noindent
\hyperref[prf:continuous-extension-to-closed-interval]{\textit{Go to proof.}}
\end{proposition}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{RightLimit}(f, a, L)
  \;\Rightarrow\;
  \operatorname{ContinuousOn}(\widetilde{f},\, [a,b]).
\]
\end{remark*}

\begin{remark*}[Interpretation]
If the right-hand limit at the missing endpoint exists, the function
extends continuously to the closed interval. Continuity at $a$ uses
$\operatorname{Within}$ (not $\operatorname{InputTolerance}$) so the
definition of $\widetilde{f}(a) = L$ is used directly — the limit
condition provides the output tolerance for sequences approaching $a$
from the right, and $\widetilde{f}(a) = L$ provides it for $x = a$.
This proposition underlies the proof that uniformly continuous
functions on open intervals extend to their closure.
\end{remark*}

% ---------------------------------------------------------
% Definition — Space of Continuous Functions
% ---------------------------------------------------------

\begin{definition}[Space of Continuous Functions]
\label{def:space-of-continuous-functions}
Let $X \subseteq \mathbb{R}$. The \textbf{space of continuous
functions on $X$} is
\[
  C^0(X;\mathbb{R}) \;:=\;
  \{\, f : X \to \mathbb{R} : \operatorname{ContinuousOn}(f,X) \,\}.
\]
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
  f \in C^0(X)
  \;\iff\;
  \operatorname{ContinuousOn}(f,X)
  \;\iff\;
  \forall c \in X,\;
  \ForallOT{\varepsilon},\;
  \ExistsIT{\delta},\;
  \forall x \in X :\;
  \operatorname{Within}(x,c,\delta)
  \Rightarrow
  \operatorname{OutputTolerance}(f(x),f(c),\varepsilon).
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
\[
  f \in C^0(X)
  \;\colonequiv\;
  \operatorname{ContinuousOn}(f,X).
\]
\end{remark*}

\begin{remark*}[Interpretation]
$C^0(X)$ is closed under pointwise addition and scalar
multiplication — it is a real vector space and in fact a
commutative ring. On compact $X$ it carries the uniform norm
$\|f\|_\infty = \sup_{x \in X}|f(x)|$, making it a Banach space.
This is the natural setting for the Weierstrass approximation
theorem and for uniform convergence theory.
\end{remark*}

% Proposition — Sign Preservation
% ---------------------------------------------------------

\begin{proposition}[Sign Preservation]
\label{prop:sign-preservation}
Let $\operatorname{ContinuousAt}(f, x_0)$ where $f : E \to \mathbb{R}$.
If $f(x_0) > 0$, then
\[
  \exists\,\delta > 0,\;
  \forall x \in E :\;
  \operatorname{Within}(x,x_0,\delta) \Rightarrow f(x) > 0.
\]
Analogously, $f(x_0) < 0$ implies $f(x) < 0$ on some
$U_E(x_0)$.

\smallskip\noindent
\hyperref[prf:sign-preservation]{\textit{Go to proof.}}
\end{proposition}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ContinuousAt}(f,x_0) \wedge f(x_0)>0
  \;\Rightarrow\;
  \exists\,\delta>0,\;\forall x\in E:\;
  \operatorname{Within}(x,x_0,\delta) \Rightarrow f(x)>0.
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
  \operatorname{ContinuousAt}(f,x_0) \wedge f(x_0)>0
  \;\Rightarrow\;
  \operatorname{BoundedAwayFromZeroNear}(f, x_0).
\]
\end{remark*}

\begin{remark*}[Contrapositive quantified statement]
\[
  \bigl(\forall\delta>0,\;\exists x\in E:\;
    \operatorname{Within}(x,x_0,\delta) \wedge f(x)\leq 0\bigr)
  \;\Rightarrow\;
  f(x_0) \leq 0 \;\vee\; \operatorname{DiscontinuousAt}(f,x_0).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A continuous function that is positive at $x_0$ remains positive
throughout some neighborhood. The proof applies
$\operatorname{ContinuousAt}(f,x_0)$ with $\varepsilon = f(x_0)/2$:
the resulting $\delta$ forces $|f(x)-f(x_0)| < f(x_0)/2$, so
$f(x) > f(x_0)/2 > 0$. Sign preservation is the key ingredient in
the proofs of the Intermediate Value Theorem (a function cannot
cross zero without hitting it) and the Extreme Value Theorem.
\end{remark*}
