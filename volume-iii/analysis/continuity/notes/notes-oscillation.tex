\section{Oscillation}
\label{sec:oscillation}

% ---------------------------------------------------------
% Toolkit
% ---------------------------------------------------------

\begin{tcolorbox}[title={Toolkit — Oscillation}]
\begin{itemize}
  \item $\omega(f;E) = \sup_{x_1,x_2 \in E} |f(x_1) - f(x_2)|$ is
    the diameter of the image $f(E)$.
  \item $\omega(f;x_0) = \lim_{\delta \to 0^+} \omega(f; U_\delta(x_0)
    \cap E)$ measures residual local spread.
  \item Continuity at $x_0$ if and only if $\omega(f;x_0) = 0$.
  \item Discontinuity set: $D(f) = \{x \in E : \omega(f;x) > 0\} =
    \bigcup_{n=1}^{\infty} \{x \in E : \omega(f;x) \geq 1/n\}$.
  \item Each threshold set $\{x : \omega(f;x) \geq 1/n\}$ is closed,
    so $D(f)$ is an $F_\sigma$ set.
\end{itemize}
\end{tcolorbox}

% ---------------------------------------------------------
% Definition — Oscillation on a Set
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Oscillation on a Set}]
\begin{definition}[Oscillation on a Set]
\label{def:oscillation-on-a-set}
Let $E \subseteq \mathbb{R}$ and let $f : E \to \mathbb{R}$. The
\textbf{oscillation of $f$ on $E$} is
\[
  \omega(f;\, E)
  \;:=\;
  \sup_{x_1, x_2 \in E} |f(x_1) - f(x_2)|
  \;=\;
  \operatorname{diam}(f(E)).
\]
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  \omega(f;\,E)
  \;=\;
  \sup\bigl\{\, |f(x_1) - f(x_2)| : x_1, x_2 \in E \,\bigr\}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The oscillation on a set records only the spread of the image, not
the location of its values. A small oscillation means all values of
$f$ on $E$ lie close together. Zero oscillation means $f$ is
constant on $E$. The equivalence $\omega(f;E) = \operatorname{diam}(f(E))$
makes the geometric content transparent: oscillation on $E$ is the
diameter of the image of $E$.
\end{remark*}

% ---------------------------------------------------------
% Definition — Oscillation at a Point
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Oscillation at a Point}]
\begin{definition}[Oscillation at a Point]
\label{def:oscillation-at-a-point}
Let $E \subseteq \mathbb{R}$, let $f : E \to \mathbb{R}$, and let
$x_0 \in E$. The \textbf{oscillation of $f$ at $x_0$} is
\[
  \omega(f;\, x_0)
  \;:=\;
  \lim_{\delta \to 0^+}
  \omega\bigl(f;\, U_\delta(x_0) \cap E\bigr),
\]
where $U_\delta(x_0) = \{\, x \in \mathbb{R} : |x - x_0| < \delta \,\}$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  \omega(f;\, x_0)
  \;=\;
  \lim_{\delta \to 0^+}
  \sup\bigl\{\, |f(x_1) - f(x_2)|
    : x_1, x_2 \in U_\delta(x_0) \cap E \,\bigr\}.
\]
\end{remark*}

\begin{remark*}[The limit exists]
For $0 < \delta_1 < \delta_2$, the inclusion
$U_{\delta_1}(x_0) \cap E \subseteq U_{\delta_2}(x_0) \cap E$
forces
\[
  \omega\bigl(f;\, U_{\delta_1}(x_0) \cap E\bigr)
  \;\leq\;
  \omega\bigl(f;\, U_{\delta_2}(x_0) \cap E\bigr).
\]
The function $\delta \mapsto \omega(f;\, U_\delta(x_0) \cap E)$ is
therefore monotone nonincreasing as $\delta \to 0^+$ and bounded
below by $0$, so the limit defining $\omega(f;\, x_0)$ exists.
\end{remark*}

\begin{remark*}[Interpretation]
The oscillation at $x_0$ measures the residual spread of nearby
function values as the neighborhood shrinks to the point. If $f$
collapses to a single cluster near $x_0$, the oscillation is $0$.
If a positive spread persists no matter how small the neighborhood,
the oscillation is positive — and the function is discontinuous at
$x_0$.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Continuity Characterised by Oscillation
% ---------------------------------------------------------

\begin{theorem}[Continuity Characterised by Oscillation]
\label{thm:continuity-iff-zero-oscillation}
Let $E \subseteq \mathbb{R}$, let $f : E \to \mathbb{R}$, and let
$x_0 \in E$. Then
\[
  f \text{ is continuous at } x_0
  \;\iff\;
  \omega(f;\, x_0) = 0.
\]

\smallskip
\noindent
\hyperref[prf:continuity-iff-zero-oscillation]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
  &\Bigl(\forall \varepsilon > 0,\;
    \exists \delta > 0,\;
    \forall x \in E:\;
    |x - x_0| < \delta \Rightarrow |f(x) - f(x_0)| < \varepsilon\Bigr) \\
  &\quad\iff\quad
  \lim_{\delta \to 0^+}
  \omega\bigl(f;\, U_\delta(x_0) \cap E\bigr) = 0.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The $\varepsilon$-$\delta$ definition requires controlling all nearby
values relative to $f(x_0)$. The oscillation formulation is
symmetric: it controls the spread between any two nearby values,
without singling out $f(x_0)$. The equivalence holds because once
all nearby values are close to each other, and $x_0$ is one of those
nearby points, all nearby values must be close to $f(x_0)$.
Continuity collapses to the vanishing of a single local invariant.
\end{remark*}

% ---------------------------------------------------------
% Proposition — Discontinuity Set via Oscillation
% ---------------------------------------------------------

\begin{proposition}[Discontinuity Set via Oscillation]
\label{prop:discontinuity-set-via-oscillation}
Let $E \subseteq \mathbb{R}$ and let $f : E \to \mathbb{R}$. The
discontinuity set of $f$ satisfies
\[
  D(f)
  \;=\;
  \{\, x \in E : \omega(f;\, x) > 0 \,\}
  \;=\;
  \bigcup_{n=1}^{\infty}
  \Bigl\{\, x \in E : \omega(f;\, x) \geq \tfrac{1}{n} \Bigr\}.
\]

\smallskip
\noindent
\hyperref[prf:discontinuity-set-via-oscillation]{\textit{Go to proof.}}
\end{proposition}

\begin{remark*}[Standard quantified statement]
\[
  \forall x \in E:\;
  x \in D(f)
  \;\iff\;
  \omega(f;\, x) > 0
  \;\iff\;
  \exists n \in \mathbb{N}:\;
  \omega(f;\, x) \geq \tfrac{1}{n}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The decomposition of $D(f)$ into threshold layers
$\{x : \omega(f;x) \geq 1/n\}$ is fundamental. Each layer
corresponds to a definite minimum amount of surviving oscillation.
These threshold sets are closed, so $D(f)$ is an $F_\sigma$ set —
a countable union of closed sets. This topological classification of
the discontinuity set is the starting point for both Baire category
arguments and the Riemann integrability criterion: a bounded function
is Riemann integrable if and only if its discontinuity set has
Lebesgue measure zero.
\end{remark*}
