\section{Limits at Infinity}
\label{sec:limits-at-infinity}

% ---------------------------------------------------------
% Toolkit
% ---------------------------------------------------------

\begin{tcolorbox}[title={Toolkit — Limits at Infinity}]
\begin{itemize}
  \item $+\infty$ adherent to $X$ $\iff$ $\sup X = +\infty$ $\iff$ $X$ is unbounded above.
  \item $\lim_{x \to +\infty} f(x) = L$: replace $|x-c| < \delta$ with $x > M$.
  \item All standard limit laws carry over with the same proofs.
  \item Analogue of sequence limits: $x > M$ plays the role of $n \geq N$.
\end{itemize}
\end{tcolorbox>

% ---------------------------------------------------------
% Definition — Infinite Adherent Points
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition — Infinite Adherent Points}]
\begin{definition}[Infinite Adherent Points]
\label{def:infinite-adherent-points}
Let $X \subseteq \mathbb{R}$.
\begin{itemize}
  \item $+\infty$ is \textbf{adherent} to $X$ if $\sup X = +\infty$,
    i.e., for every $M \in \mathbb{R}$ there exists $x \in X$ with
    $x > M$.
  \item $-\infty$ is \textbf{adherent} to $X$ if $\inf X = -\infty$,
    i.e., for every $M \in \mathbb{R}$ there exists $x \in X$ with
    $x < M$.
\end{itemize}
A set $X$ is bounded if and only if neither $+\infty$ nor $-\infty$
is adherent to $X$.
\end{definition}
\end{tcolorbox>

\begin{remark*}[Standard quantified statement]
\[
  +\infty \text{ adherent to } X
  \;\colonequiv\;
  \forall M \in \mathbb{R},\;
  \exists x \in X :\; x > M
  \;\colonequiv\;
  \sup X = +\infty.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The finite adherent point definition requires elements of $X$ to be
arbitrarily close to a point $x_0 \in \mathbb{R}$. The infinite
version requires elements of $X$ to be arbitrarily large. Both are
instances of the general notion of adherence in the extended real
line $\overline{\mathbb{R}} = \mathbb{R} \cup \{-\infty, +\infty\}$,
where neighborhoods of $+\infty$ are the rays $(M, +\infty)$.
\end{remark*}

% ---------------------------------------------------------
% Definition — Limits at Infinity
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition — Limits at Infinity}]
\begin{definition}[Limits at Infinity]
\label{def:limit-at-infinity}
Let $X \subseteq \mathbb{R}$ with $+\infty$ adherent to $X$, and let
$f : X \to \mathbb{R}$. The function $f(x)$ \textbf{converges to $L$
as $x \to +\infty$}, written $\lim_{x \to +\infty} f(x) = L$, if for
every $\varepsilon > 0$ there exists $M \in \mathbb{R}$ such that
\[
  x \in X,\; x > M
  \;\Rightarrow\;
  |f(x) - L| < \varepsilon.
\]
Similarly, $\lim_{x \to -\infty} f(x) = L$ if for every $\varepsilon
> 0$ there exists $M \in \mathbb{R}$ such that
\[
  x \in X,\; x < M
  \;\Rightarrow\;
  |f(x) - L| < \varepsilon.
\]
\end{definition}
\end{tcolorbox>

\begin{remark*}[Standard quantified statement]
\[
  \lim_{x \to +\infty} f(x) = L
  \;\colonequiv\;
  \forall \varepsilon > 0,\;
  \exists M \in \mathbb{R},\;
  \forall x \in X :\;
  x > M \Rightarrow |f(x) - L| < \varepsilon.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
  \lim_{x \to +\infty} f(x) \neq L
  \;\colonequiv\;
  \exists \varepsilon_0 > 0,\;
  \forall M \in \mathbb{R},\;
  \exists x \in X :\;
  x > M \;\wedge\; |f(x) - L| \geq \varepsilon_0.
\]
\end{remark*}

\begin{remark*}[Comparison with finite limits]
The finite limit $\lim_{x \to c} f(x) = L$ uses $|x - c| < \delta$
as the proximity condition. The limit at infinity replaces this with
$x > M$: instead of $x$ being within distance $\delta$ of $c$, $x$
must lie beyond the threshold $M$. The output condition
$|f(x) - L| < \varepsilon$ is identical. This is exactly the
translation of sequence convergence $a_n \to L$ (i.e., $n \geq N
\Rightarrow |a_n - L| < \varepsilon$) to the function setting.
\end{remark*}

\begin{remark*}[Limit laws]
All standard limit laws — sum, product, quotient, composition, and
the squeeze theorem — carry over to limits at infinity. The proofs
are identical to those for finite limits, with the $\delta$-neighborhood
condition replaced by the $M$-threshold condition. The canonical
non-example is $f(x) = \sin x$: no $L$ satisfies the definition
because $\sin x$ oscillates indefinitely for all large $x$.
\end{remark*}
