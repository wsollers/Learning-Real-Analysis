\subsection{Monotone Functions and Continuity}
\label{sec:monotone-functions}

\begin{tcolorbox}[title={Toolkit --- Monotone Functions}]
\begin{itemize}
  \item Every monotone function has $\operatorname{LeftLimit}$ and
    $\operatorname{RightLimit}$ at every interior point.
  \item $\operatorname{ContinuousAt}(f,c)\iff f(c^-)=f(c)=f(c^+)
    \iff j_f(c)=0$.
  \item All discontinuities of a monotone function are of first kind
    — jump only, no oscillation.
  \item At most countably many discontinuities: disjoint jump
    intervals inject into $\mathbb{Q}$.
  \item Strictly monotone and continuous on $I$ $\Rightarrow$
    $f^{-1}$ exists, is strictly monotone, and is continuous.
  \item $\operatorname{ContinuousOn}(f,[a,b])\wedge\operatorname{Injective}(f)
    \iff f$ strictly monotone.
\end{itemize}
\end{tcolorbox}

% ---------------------------------------------------------
% Theorem — One-Sided Limits of Monotone Functions
% ---------------------------------------------------------

\begin{theorem}[One-Sided Limits of Monotone Functions]
\label{thm:monotone-one-sided-limits}
Let $f$ be increasing on interval $I$, $c$ interior to $I$. Then:
\begin{enumerate}[label=(\roman*)]
  \item $\operatorname{LeftLimit}(f,c,\sup\{f(x):x<c\})$.
  \item $\operatorname{RightLimit}(f,c,\inf\{f(x):x>c\})$.
\end{enumerate}

\smallskip\noindent
\hyperref[prf:monotone-one-sided-limits]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{MonotoneFunction}(f,I)
  \;\Rightarrow\;
  \forall c\text{ interior to }I:\;
  \operatorname{LeftLimit}(f,c,f(c^-))
  \;\wedge\;
  \operatorname{RightLimit}(f,c,f(c^+)),\;
  \text{both finite.}
\]
\end{remark*>

\begin{remark*}[Definition predicate reading]
\[
  f(c^-) \;:=\; \sup\{f(x):x\in I,\,x<c\},
  \qquad
  f(c^+) \;:=\; \inf\{f(x):x\in I,\,x>c\}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Monotonicity constrains values on either side of $c$ to form bounded
monotone sets, forcing both one-sided limits to exist. The proof is
a direct $\varepsilon$-$\delta$ argument using the definition of
supremum to produce the required $x_0<c$.
\end{remark*}

% ---------------------------------------------------------
% Corollary — Continuity Criterion for Monotone Functions
% ---------------------------------------------------------

\begin{corollary}[Continuity Criterion for Monotone Functions]
\label{cor:monotone-continuity-criterion}
For $f$ increasing on $I$, $c$ interior:
$\operatorname{ContinuousAt}(f,c)\iff f(c^-)=f(c)=f(c^+)
\iff\sup\{f(x):x<c\}=f(c)=\inf\{f(x):x>c\}$.

\smallskip\noindent
\hyperref[prf:monotone-continuity-criterion]{\textit{Go to proof.}}
\end{corollary>

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ContinuousAt}(f,c)
  \;\iff\;
  f(c^-)=f(c)=f(c^+).
\]
\end{remark*}

\begin{remark*}[Interpretation]
For a monotone function, continuity at $c$ means no gap in the graph
at $c$. The jump $j_f(c)=f(c^+)-f(c^-)$ measures that gap; continuity
holds exactly when $j_f(c)=0$.
\end{remark*}

% ---------------------------------------------------------
% Definition — Jump of a Function
% ---------------------------------------------------------

\begin{tcolorbox}[title={Definition --- Jump of a Function}]
\begin{definition}[Jump of a Function]
\label{def:jump-of-function}
For $f$ increasing on $I$, $c\in I$:
$j_f(c):=\operatorname{RightLimit}(f,c,\cdot)-\operatorname{LeftLimit}(f,c,\cdot)
=f(c^+)-f(c^-)$.
$\operatorname{ContinuousAt}(f,c)\iff j_f(c)=0$.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
  j_f(c)=f(c^+)-f(c^-)\geq 0.
  \qquad
  \operatorname{ContinuousAt}(f,c)\iff j_f(c)=0.
\]
\end{remark*}

\begin{remark*}[Interpretation]
A positive jump means $f$ leaps upward at $c$, skipping all values
in $(f(c^-),f(c^+))$.
\end{remark*}

% ---------------------------------------------------------
% Proposition — Discontinuities Are of First Kind
% ---------------------------------------------------------

\begin{proposition}[Discontinuities of a Monotone Function Are of First Kind]
\label{prop:monotone-discontinuities-first-kind}
$\operatorname{MonotoneFunction}(f,E)$, $a\in E$ a discontinuity
$\Rightarrow\operatorname{FirstKind}(f,a)$: both $f(a^-)$, $f(a^+)$
exist finitely.

\smallskip\noindent
\hyperref[prf:monotone-discontinuities-first-kind]{\textit{Go to proof.}}
\end{proposition>

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{MonotoneFunction}(f,E)
  \;\wedge\;
  \operatorname{DiscontinuousAt}(f,a)
  \;\Rightarrow\;
  \operatorname{FirstKind}(f,a).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A monotone function may jump but cannot oscillate. Oscillation at
$a$ would require $f$ to move both upward and downward arbitrarily
close to $a$, contradicting monotonicity.
\end{remark*}

% ---------------------------------------------------------
% Corollary — Jump Intervals Are Disjoint
% ---------------------------------------------------------

\begin{corollary}[Jump Intervals Are Disjoint]
\label{cor:jump-intervals-for-monotone-discontinuities}
For $f$ nondecreasing, distinct discontinuities $a\neq b$:
$(f(a^-),f(a^+))\cap(f(b^-),f(b^+))=\emptyset$.

\smallskip\noindent
\hyperref[prf:jump-intervals-for-monotone-discontinuities]{\textit{Go to proof.}}
\end{corollary>

\begin{remark*}[Standard quantified statement]
\[
  a<b\text{ discontinuities, }f\text{ nondecreasing}
  \;\Rightarrow\;
  f(a^+)\leq f(b^-)
  \;\Rightarrow\;
  (f(a^-),f(a^+))\cap(f(b^-),f(b^+))=\emptyset.
\]
\end{remark*}

\begin{remark*}[Interpretation]
For $a<b$, any $x\in(a,b)$ satisfies $f(a^+)\leq f(x)\leq f(b^-)$,
so jump intervals at $a$ and $b$ are disjoint. This is the geometric
heart of countability of discontinuities.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Discontinuities Are Countable
% ---------------------------------------------------------

\begin{theorem}[Discontinuities of a Monotone Function Are Countable]
\label{thm:monotone-discontinuities-countable}
$\operatorname{MonotoneFunction}(f,I)$
$\Rightarrow|\{c\in I:\operatorname{DiscontinuousAt}(f,c)\}|\leq\aleph_0$.

\smallskip\noindent
\hyperref[prf:monotone-discontinuities-countable]{\textit{Go to proof.}}
\end{theorem>

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{MonotoneFunction}(f,I)
  \;\Rightarrow\;
  D_f:=\{c\in I:\operatorname{DiscontinuousAt}(f,c)\}\text{ is countable.}
\]
\end{remark*}

\begin{remark*}[Interpretation]
At each discontinuity $c$, the jump $j_f(c)>0$ carves out a nonempty
open interval. These jump intervals are pairwise disjoint. Each
contains a rational, giving an injection $D_f\hookrightarrow\mathbb{Q}$.
Hence $D_f$ is countable. The result is sharp: countably infinitely
many discontinuities are possible.
\end{remark*}

% ---------------------------------------------------------
% Corollary — Monotone Function Is Continuous on a Dense Set
% ---------------------------------------------------------

\begin{corollary}[Monotone Function Is Continuous on a Dense Set]
\label{cor:monotone-continuous-on-dense-set}
$\operatorname{MonotoneFunction}(f,[a,b])$
$\Rightarrow C_f:=\{x:\operatorname{ContinuousAt}(f,x)\}$ is dense in $[a,b]$.

\smallskip\noindent
\hyperref[prf:monotone-continuous-on-dense-set]{\textit{Go to proof.}}
\end{corollary>

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{MonotoneFunction}(f,[a,b])
  \;\Rightarrow\;
  \forall(c,d)\subseteq[a,b],\;\exists x\in(c,d):
  \operatorname{ContinuousAt}(f,x).
\]
\end{remark*}

\begin{remark*}[Interpretation]
$D_f$ is countable hence nowhere dense; $C_f=[a,b]\setminus D_f$ is
dense. Continuity is generic; discontinuity is exceptional.
\end{remark*}

% ---------------------------------------------------------
% Proposition — Continuous Injective Is Strictly Monotone
% ---------------------------------------------------------

\begin{proposition}[Continuous Injective Is Strictly Monotone]
\label{prop:continuous-injective-iff-strictly-monotone}
$\operatorname{ContinuousOn}(f,[a,b])$:
$\operatorname{Injective}(f)\iff f$ strictly monotone.

\smallskip\noindent
\hyperref[prf:continuous-injective-is-strictly-monotone]{\textit{Go to proof.}}
\end{proposition>

\begin{remark*}[Standard quantified statement]
\[
  \operatorname{ContinuousOn}(f,[a,b]):\;
  \operatorname{Injective}(f)
  \;\iff\;
  \operatorname{StrictlyIncreasing}(f)
  \;\vee\;
  \operatorname{StrictlyDecreasing}(f).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Non-monotonicity forces a local extremum; IVT then produces two
distinct preimages of some value, contradicting injectivity. The
converse is immediate: strict monotonicity implies injectivity.
\end{remark*}

% ---------------------------------------------------------
% Theorem — Continuous Inverse Theorem
% ---------------------------------------------------------

\begin{theorem}[Continuous Inverse Theorem]
\label{thm:continuous-inverse-theorem}
$I$ an interval, $f$ strictly monotone and
$\operatorname{ContinuousOn}(f,I)$
$\Rightarrow\operatorname{InverseFunction}(f,f^{-1})$ and
$f^{-1}$ strictly monotone and $\operatorname{ContinuousOn}(f^{-1},f(I))$.

\smallskip\noindent
\hyperref[prf:inverse-strictly-monotone]{\textit{Go to proof.}}
\end{theorem>

\begin{remark*}[Standard quantified statement]
\[
  f\text{ strictly monotone and cts on }I
  \;\Rightarrow\;
  \operatorname{ContinuousOn}(f^{-1},f(I)).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Continuity on $I$ forces $f(I)$ to be an interval (Preservation of
Intervals). A jump discontinuity of $f^{-1}$ would create a gap in
$f(I)$, contradicting that $f(I)$ is an interval. Hence $f^{-1}$
is continuous. Note: no extra hypothesis is needed because
$\operatorname{ContinuousAt}$ uses $\operatorname{Within}$ (not
$\operatorname{InputTolerance}$), so the base point is included.
\end{remark*}
