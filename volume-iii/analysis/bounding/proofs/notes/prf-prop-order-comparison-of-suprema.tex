\begin{proposition*}[Order Comparison of Suprema]
Let $A, B \subseteq \mathbb{R}$ be non-empty and bounded above. If for every $x \in A$ there exists $y \in B$ such that $x \le y$, then $\sup A \le \sup B$.
\end{proposition*}
