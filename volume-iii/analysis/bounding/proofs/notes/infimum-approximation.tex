\subsection*{$\varepsilon$-Characterization of Infimum}
\label{prf:infimum-approximation}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[cor:eps-approx]{Back to Corollary in Notes}
\end{remark*}

\noindent\textit{Proof A — by contradiction.}

\begin{proof}
\Claim Let $A \subseteq \mathbb{R}$ be nonempty and bounded below, and let
$i = \inf A$.
Then for every $\varepsilon > 0$ there exists $a \in A$ with $a < i + \varepsilon$.

\Given $i = \inf A$; let $\varepsilon > 0$.
\Goal $\exists\, a \in A$ such that $a < i + \varepsilon$.

\Strategy Suppose for contradiction that no such $a$ exists.
Then $a \ge i + \varepsilon$ for all $a \in A$, so $i + \varepsilon$ is a lower
bound of $A$.
But $i + \varepsilon > i$, contradicting $i$ being the \emph{greatest} lower bound.
\Therefore the assumption is false; such an $a$ exists. \AsReq
\end{proof}

\medskip
\noindent\textit{Proof B — direct.}

\begin{proof}
\Claim Same statement.

\Given $i = \inf A$; let $\varepsilon > 0$.
\Goal $\exists\, a \in A$ such that $a < i + \varepsilon$.

Since $i + \varepsilon > i$ and $i$ is the greatest lower bound, $i + \varepsilon$
is \emph{not} a lower bound of $A$.
\ByDef lower bound, there must exist $a \in A$ with $a < i + \varepsilon$. \AsReq
\end{proof}

\begin{remark*}[Proof shape]
Mirror image of the supremum approximation proofs.
A value strictly above $\inf A$ cannot be a lower bound; both proofs exploit
this, one by contradiction, one directly.
\end{remark*}

\begin{remark*}[Dependencies]
Definition of infimum (greatest lower bound).
\end{remark*}
