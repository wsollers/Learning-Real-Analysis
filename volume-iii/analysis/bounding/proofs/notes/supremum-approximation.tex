\begin{lemma*}[Supremum Approximation Lemma]
Let $s = \sup A$. Then for every $\varepsilon > 0$ there exists $a \in A$
such that
\[
s - \varepsilon < a .
\]
\end{lemma*}

\begin{proof}
Assume for contradiction that there exists $\varepsilon_0 > 0$ such that

\[
\neg(\exists a \in A \text{ such that } s - \varepsilon_0 < a).
\]

\begin{remark}
Negating the existential statement gives the universal statement
\[
\forall a \in A,\quad a \le s - \varepsilon_0.
\]
\end{remark}

This means that $s - \varepsilon_0$ is an upper bound for $A$.

But $s = \sup A$ is the \textbf{least upper bound}. However,

\[
s - \varepsilon_0 < s.
\]

Thus $s - \varepsilon_0$ would be an upper bound smaller than $s$,
contradicting the fact that $s$ is the supremum.

Therefore our assumption is false, and there exists $a \in A$ such that

\[
s - \varepsilon < a
\]

for every $\varepsilon > 0$.
\end{proof}

\begin{lemma*}[Supremum Approximation]
Let \(A \subset \mathbb{R}\) be nonempty and bounded above, and let
\[
s = \sup A .
\]
Then for every \(\varepsilon > 0\) there exists \(a \in A\) such that
\[
s - \varepsilon < a .
\]
\end{lemma*}

\begin{proof}
Since \(s = \sup A\), we know that
\[
a \le s \quad \text{for all } a \in A.
\]

Now let \(\varepsilon > 0\). Note that
\[
s - \varepsilon < s.
\]

If \(s - \varepsilon\) were an upper bound of \(A\), then it would be an
upper bound smaller than \(s\). But \(s\) is the \emph{least upper bound}
of \(A\).

Therefore \(s - \varepsilon\) cannot be an upper bound of \(A\).
Hence there must exist some \(a \in A\) such that
\[
s - \varepsilon < a.
\]
\end{proof}
