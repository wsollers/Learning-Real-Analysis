\subsubsection[Diameter Monotonicity]{Proof --- Diameter Monotonicity}

\bigskip

\noindent
\textbf{Source.}
See notes \texttt{notes-bounding-monotonicty.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Let $(X, d)$ be a metric space and $A \subseteq B \subseteq X$.
Show $\operatorname{diam}(A) \le \operatorname{diam}(B)$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  A \subseteq B \subseteq X
  \;\Rightarrow\;
  \sup\{d(x,y) : x,y \in A\}
  \;\le\;
  \sup\{d(x,y) : x,y \in B\}.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
The set of pairwise distances $\{d(x,y) : x,y \in A\}$ is a subset of
$\{d(x,y) : x,y \in B\}$ when $A \subseteq B$.
Monotonicity of supremum then gives the result.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $A \subseteq B \subseteq X \Rightarrow \operatorname{diam}(A) \le \operatorname{diam}(B)$.

\Given Metric space $(X,d)$, $A \subseteq B \subseteq X$.

\Goal $\operatorname{diam}(A) \le \operatorname{diam}(B)$.

\Strategy Show the distance set of $A$ is contained in the distance set of $B$;
apply Monotonicity of Supremum.

\medskip

% --- show {d(x,y): x,y ∈ A} ⊆ {d(x,y): x,y ∈ B} ---

% --- apply Monotonicity of Supremum ---

\end{proof}

\begin{remark*}[Proof shape]
Direct subset comparison of distance-sets plus the monotonicity lemma.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of diameter: $\operatorname{diam}(A) = \sup\{d(x,y) : x,y \in A\}$,
        $\operatorname{diam}(\varnothing) = 0$.
  \item Monotonicity of Supremum (set version).
\end{itemize}
\end{remark*}
