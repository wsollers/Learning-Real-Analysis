\subsubsection[Monotonicity of Supremum and Infimum]{Proof --- Monotonicity of Supremum and Infimum}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Chapter~1; see also notes
\texttt{notes-bounding-monotonicty.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Show that if $A \subseteq B \subseteq \mathbb{R}$ with both bounded above,
then $\sup A \le \sup B$; and if both bounded below, then $\inf A \ge \inf B$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  A \subseteq B \subseteq \mathbb{R},\ A,B \text{ bounded above}
  \;\Rightarrow\;
  \sup A \le \sup B.
\]
\[
  A \subseteq B \subseteq \mathbb{R},\ A,B \text{ bounded below}
  \;\Rightarrow\;
  \inf A \ge \inf B.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Since $A \subseteq B$, every upper bound for $B$ is also an upper bound for
$A$. In particular $\sup B$ is an upper bound for $A$. Since $\sup A$ is
the \emph{least} upper bound for $A$, we get $\sup A \le \sup B$.
Infimum case is symmetric.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim Let $A \subseteq B \subseteq \mathbb{R}$ be nonempty with both bounded
above. Then $\sup A \le \sup B$.

\Given $A \subseteq B$, both nonempty and bounded above.

\Goal Show $\sup A \le \sup B$.

\Strategy Show $\sup B$ is an upper bound for $A$; then appeal to $\sup A$
being the least upper bound.

\begin{remark*}[Proof shape]
Direct: $A \subseteq B$ implies every upper bound for $B$ bounds $A$.
Then minimality of $\sup A$ gives the inequality.
\end{remark*}
