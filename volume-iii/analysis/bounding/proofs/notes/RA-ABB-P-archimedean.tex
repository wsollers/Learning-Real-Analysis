\subsubsection[Archimedean Property]{Proof --- Archimedean Property}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Theorem~1.4.2; see notes
\texttt{notes-completeness.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Show that $\mathbb{R}$ satisfies the Archimedean property:
$\forall x \in \mathbb{R},\; \exists n \in \mathbb{N},\; n > x$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \forall x \in \mathbb{R},\; \exists n \in \mathbb{N},\; n > x.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Suppose for contradiction $\mathbb{N}$ is bounded above by $x$. By the
Completeness Axiom $\alpha = \sup \mathbb{N}$ exists. Then $\alpha - 1$
is not an upper bound, so $\exists n \in \mathbb{N}$ with $n > \alpha - 1$,
i.e.\ $n + 1 > \alpha$. But $n + 1 \in \mathbb{N}$, contradicting $\alpha$
being an upper bound.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $\forall x \in \mathbb{R},\; \exists n \in \mathbb{N},\; n > x$.

\Given $x \in \mathbb{R}$ arbitrary.

\Goal Find $n \in \mathbb{N}$ with $n > x$.

\Strategy Suppose $\mathbb{N}$ were bounded above; derive a contradiction
via the Completeness Axiom and the $\varepsilon$-characterization of $\sup$.

\medskip

% --- assume for contradiction: x is an upper bound for N ---

% --- let α = sup N (Completeness) ---

% --- use ε = 1: ∃ n ∈ N with n > α - 1, so n+1 > α ---

% --- n+1 ∈ N contradicts α = sup N ---

\end{proof}

\begin{remark*}[Proof shape]
Proof by contradiction: a bounded $\mathbb{N}$ would have a supremum, but
the successor of any near-supremum natural number exceeds the supremum.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Completeness Axiom.
  \item $\varepsilon$-Characterization of Supremum (with $\varepsilon = 1$).
  \item Closure of $\mathbb{N}$ under successor: $n \in \mathbb{N} \Rightarrow n + 1 \in \mathbb{N}$.
\end{itemize}
\end{remark*}
