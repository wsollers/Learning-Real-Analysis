\subsection*{Minimum Equals Infimum}
\label{prf:min-is-inf}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:sup-unique]{Back to Notes}
\end{remark*}

\begin{proof}
\Claim Let $A \subseteq \mathbb{R}$.
If $A$ has a minimum $m = \min A$, then $m = \inf A$.

\Given $m = \min A$: $m \in A$ and $m \le a$ for all $a \in A$.
\Goal $m = \inf A$, i.e.\ $m$ is the greatest lower bound of $A$.

\ByDef minimum, $m \le a$ for all $a \in A$, so $m$ is a lower bound of $A$.

\Let $\ell$ be any lower bound of $A$.
Then $\ell \le a$ for all $a \in A$.
Since $m \in A$, taking $a = m$ gives $\ell \le m$.

\Therefore $\ell \le m$ for every lower bound $\ell$, so $m$ is the greatest lower bound. \AsReq
\end{proof}

\begin{remark*}[Proof shape]
Mirror image of max-is-sup: (1) $m$ is a lower bound; (2) $m$ is the greatest
such bound because $m \in A$ witnesses $\ell \le m$ for any lower bound $\ell$.
\end{remark*}

\begin{remark*}[Dependencies]
Definition of minimum; definition of infimum (greatest lower bound).
\end{remark*}
