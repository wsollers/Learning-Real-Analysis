\begin{theorem*}
Let $A \subseteq \mathbb{R}$. If $A$ has a minimum, then
\[
\min A = \inf A .
\]
\end{theorem*}

\begin{proof}
Assume $A$ has a minimum and let
\[
m = \min A .
\]

By definition of minimum, we have

\begin{enumerate}
\item $m \in A$,
\item $m \le a$ for all $a \in A$.
\end{enumerate}

From (2), $m$ is a lower bound of $A$.

Now let $\ell$ be any lower bound of $A$. Then
\[
\ell \le a \quad \text{for all } a \in A .
\]

Since $m \in A$, substituting $a=m$ gives
\[
\ell \le m .
\]

Thus $m$ is greater than or equal to every lower bound of $A$. Therefore $m$ is the greatest lower bound of $A$, and hence
\[
m = \inf A .
\]

Since $m = \min A$, we conclude
\[
\min A = \inf A .
\]
\end{proof}
