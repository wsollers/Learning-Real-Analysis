\subsubsection[$\varepsilon$-Approximation]{Proof --- $\varepsilon$-Approximation}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}; see \texttt{notes-bounds-extremal-values.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $s = \sup A$ then $\forall \varepsilon > 0,\; \exists a \in A$ with $s - \varepsilon < a \le s$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  s = \sup A \Rightarrow \forall \varepsilon > 0,\; \exists a \in A,\; s - \varepsilon < a \le s.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
This is the downward half of the $\varepsilon$-characterization of $\sup$: apply condition~(b) of Proposition~\ref{prop:eps-char} to get $a > s - \varepsilon$, and condition~(a) gives $a \le s$.

\vspace{0.75em}
