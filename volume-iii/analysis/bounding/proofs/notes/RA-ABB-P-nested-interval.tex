\subsubsection[Nested Interval Property]{Proof --- Nested Interval Property}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Theorem~1.4.1; see notes
\texttt{notes-completeness.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Let $\{[a_n, b_n]\}$ be a sequence of nested closed bounded intervals,
i.e.\ $[a_{n+1}, b_{n+1}] \subseteq [a_n, b_n]$ for all $n$.
Show $\bigcap_{n=1}^{\infty} [a_n, b_n] \neq \varnothing$.
If additionally $b_n - a_n \to 0$, the intersection is exactly one point.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \forall n,\; a_n \le a_{n+1} \le b_{n+1} \le b_n
  \;\Rightarrow\;
  \exists x \in \mathbb{R},\; \forall n,\; a_n \le x \le b_n.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Let $A = \{a_n : n \in \mathbb{N}\}$. Show $A$ is bounded above (by $b_1$,
in fact by any $b_m$). By the Completeness Axiom, $s = \sup A$ exists.
Argue $a_n \le s$ (clear) and $s \le b_n$ for all $n$ (since each $b_n$
is an upper bound for $A$). Hence $s \in \bigcap [a_n, b_n]$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim The intersection of a nested sequence of nonempty closed bounded
intervals is nonempty.

\Given $[a_1, b_1] \supseteq [a_2, b_2] \supseteq \cdots$, each nonempty
and closed.

\Goal Exhibit $x \in \bigcap [a_n, b_n]$.

\Strategy Show $A = \{a_n\}$ is bounded above; take $s = \sup A$; verify
$s \in [a_n, b_n]$ for all $n$.

\medskip

% --- show A is bounded above by b_1 ---

% --- let s = sup A (Completeness Axiom) ---

% --- show s ≤ b_n for all n ---

% --- conclude s ∈ ∩[a_n, b_n] ---

\medskip
% --- uniqueness when b_n - a_n → 0 ---

\end{proof}

\begin{remark*}[Proof shape]
Take the sup of left endpoints; the nesting forces it into every interval.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Completeness Axiom (Least Upper Bound Property).
  \item The nesting condition implies each $b_m$ is an upper bound for
        $\{a_n\}$.
\end{itemize}
\end{remark*}
