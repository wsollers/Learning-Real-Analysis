\subsubsection[Existence of Square Roots]{Proof --- Existence of Square Roots}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Theorem~1.4.5; see notes
\texttt{notes-completeness.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
For every $a \ge 0$ there exists a unique $x \ge 0$ with $x^2 = a$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \forall a \ge 0,\; \exists!\, x \ge 0,\; x^2 = a.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
\textbf{Existence:} Let $S = \{t \ge 0 : t^2 \le a\}$. $S$ is nonempty
(contains $0$) and bounded above (by $a + 1$ or similar). Set $x = \sup S$.
Show $x^2 < a$ and $x^2 > a$ both lead to contradiction (using the
Archimedean property to move $x$ slightly and stay in $S$ or outside it).
Hence $x^2 = a$.
\textbf{Uniqueness:} If $x^2 = y^2 = a$ with $x, y \ge 0$, then
$(x-y)(x+y) = 0$. Since $x,y \ge 0$, $x + y \ge 0$; if $a > 0$ then
$x + y > 0$, forcing $x = y$. If $a = 0$, then $x = y = 0$.

\vspace{0.75em}

\noindent
\textbf{Note.}
This is one of the more involved proofs in the chapter. The existence part
requires a careful $\varepsilon$-style argument at the supremum.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $\forall a \ge 0,\; \exists!\, x \ge 0,\; x^2 = a$.

\Given $a \ge 0$.

\Goal Find unique $x \ge 0$ with $x^2 = a$.

\Strategy \textbf{Existence:} Supremum of $S = \{t \ge 0 : t^2 \le a\}$;
show the sup squares to $a$ by ruling out $x^2 < a$ and $x^2 > a$.
\textbf{Uniqueness:} Factor $x^2 - y^2 = (x-y)(x+y)$.

\begin{remark*}[Proof shape]
Completeness-driven existence: define $S$ as a level set, take $\sup$,
then rule out both $x^2 < a$ and $x^2 > a$ by perturbing.
Uniqueness is algebraic.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Completeness Axiom.
  \item Archimedean Property (to find small $\varepsilon$ perturbing $x$).
  \item $\varepsilon$-Characterization of Supremum.
\end{itemize}
\end{remark*}
