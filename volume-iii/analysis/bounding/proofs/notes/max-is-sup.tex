\subsection*{Maximum Equals Supremum}
\label{prf:max-is-sup}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:sup-unique]{Back to Notes}
\end{remark*}

\begin{proof}
\Claim Let $A \subseteq \mathbb{R}$.
If $A$ has a maximum $m = \max A$, then $m = \sup A$.

\Given $m = \max A$: $m \in A$ and $a \le m$ for all $a \in A$.
\Goal $m = \sup A$, i.e.\ $m$ is the least upper bound of $A$.

\ByDef maximum, $a \le m$ for all $a \in A$, so $m$ is an upper bound of $A$.

\Let $u$ be any upper bound of $A$.
Then $a \le u$ for all $a \in A$.
Since $m \in A$, taking $a = m$ gives $m \le u$.

\Therefore $m \le u$ for every upper bound $u$, so $m$ is the least upper bound. \AsReq
\end{proof}

\begin{remark*}[Proof shape]
Two-step upper-bound argument: (1) $m$ is an upper bound (from the definition
of maximum); (2) $m$ is the least such bound (because $m \in A$, it witnesses
$m \le u$ for any upper bound $u$).
\end{remark*}

\begin{remark*}[Dependencies]
Definition of maximum; definition of supremum (least upper bound).
\end{remark*}
