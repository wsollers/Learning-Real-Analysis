\begin{theorem*}
Let $A \subseteq \mathbb{R}$. If $A$ has a maximum, then
\[
\max A = \sup A .
\]
\end{theorem*}

\begin{proof}
Assume $A$ has a maximum and let
\[
m = \max A .
\]

By definition of maximum, we have

\begin{enumerate}
\item $m \in A$,
\item $a \le m$ for all $a \in A$.
\end{enumerate}

From (2), $m$ is an upper bound of $A$.

Now let $u$ be any upper bound of $A$. Then
\[
a \le u \quad \text{for all } a \in A .
\]

Since $m \in A$, substituting $a=m$ gives
\[
m \le u .
\]

Thus $m$ is less than or equal to every upper bound of $A$. Therefore $m$ is the least upper bound of $A$, and hence
\[
m = \sup A .
\]

Since $m = \max A$, we conclude
\[
\max A = \sup A .
\]
\end{proof}
