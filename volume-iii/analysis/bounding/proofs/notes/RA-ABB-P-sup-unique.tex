\subsubsection[Uniqueness of Supremum and Infimum]{Proof --- Uniqueness of Supremum and Infimum}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Chapter~1; see also notes
\texttt{notes-bounds-extremal-values.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Show that if $\sup A$ exists then it is unique, and if $\inf A$ exists then
it is unique.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \bigl(\exists s,\, s = \sup A\bigr)
  \;\Rightarrow\;
  \bigl(\forall s',\, s' = \sup A \Rightarrow s' = s\bigr),
\]
and symmetrically for $\inf$.

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Suppose $s$ and $s'$ are both suprema of $A$.
Each must be $\le$ the other (since each is the \emph{least} upper bound,
and the other is an upper bound). Hence $s = s'$.
The infimum case is symmetric.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $\sup A$ exists it is unique; if $\inf A$ exists it is unique.

\Given $A \subseteq \mathbb{R}$ nonempty. Suppose $s$ and $s'$ both satisfy
the definition of $\sup A$.

\Goal Show $s = s'$.

\Strategy Both $s$ and $s'$ are upper bounds for $A$.
Since $s = \sup A$ (least upper bound), $s \le s'$.
Since $s' = \sup A$, $s' \le s$.
Hence $s = s'$.

\medskip

% --- your proof here ---

\end{proof}

\begin{remark*}[Proof shape]
Antisymmetry argument: each of two least upper bounds must be $\le$ the
other, forcing equality.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of supremum (least upper bound): $s = \sup A$ iff $s$
        is an upper bound and $s \le u$ for every upper bound $u$.
  \item Antisymmetry of $\le$ on $\mathbb{R}$.
\end{itemize}
\end{remark*}
