\subsubsection[Irrationals Are Dense]{Proof --- Irrationals Are Dense}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Corollary~1.4.4; see \texttt{notes-completeness.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Between any two distinct reals $a < b$ there exists an irrational $t$ with $a < t < b$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \forall a < b \in \mathbb{R},\; \exists t \notin \mathbb{Q},\; a < t < b.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Apply Density of $\mathbb{Q}$ to the interval $(a/\sqrt{2},\, b/\sqrt{2})$ to get a rational $q$. Then $t = q\sqrt{2}$ is irrational and $a < t < b$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim Between any two distinct reals $a < b$ there exists an irrational $t$ with $a < t < b$.

\Given 

\Goal 

\Strategy Apply Density of $\mathbb{Q}$ to the interval $(a/\sqrt{2},\, b/\sqrt{2})$ to get a rational $q$. Then $t = q\sqrt{2}$ is irrational and $a < t < b$.

\medskip

% --- observe a/√2 < b/√2 (since √2 > 0) ---
% --- by Density of Q: ∃ q ∈ Q with a/√2 < q < b/√2 ---
% --- set t = q√2; since q ≠ 0 (as a < b ensures a/√2 ≠ b/√2) and √2 irrational, t ∉ Q ---
% --- multiply a/√2 < q < b/√2 by √2 > 0: a < t < b ---

\end{proof}

\begin{remark*}[Proof shape]
Scale the interval by $1/\sqrt{2}$, find a rational inside it, then scale back by $\sqrt{2}$ to get an irrational.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Density of $\mathbb{Q}$ in $\mathbb{R}$.
  \item Irrationality of $\sqrt{2}$ (proved in an earlier chapter).
  \item $q \in \mathbb{Q},\; q \ne 0,\; \sqrt{2} \notin \mathbb{Q} \Rightarrow q\sqrt{2} \notin \mathbb{Q}$.
\end{itemize}
\end{remark*}
