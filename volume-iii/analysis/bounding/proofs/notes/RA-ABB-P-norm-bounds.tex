\subsubsection[Norm Bounds]{Proof --- Norm Bounds}

\bigskip

\noindent
\textbf{Source.}
See notes \texttt{notes-bounding-monotonicty.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Let $A \subseteq B \subseteq \mathbb{R}^n$.
Show $\sup_{x \in A} \|x\| \le \sup_{x \in B} \|x\|$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  A \subseteq B \subseteq \mathbb{R}^n
  \;\Rightarrow\;
  \sup_{x \in A}\|x\| \le \sup_{x \in B}\|x\|.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Immediate from Monotonicity for Functions: set $f(x) = \|x\|$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $A \subseteq B \subseteq \mathbb{R}^n \Rightarrow \sup_{A}\|\cdot\| \le \sup_{B}\|\cdot\|$.

\Given $A \subseteq B \subseteq \mathbb{R}^n$.

\Goal Show the sup-norm inequality.

\Strategy Define $f : \mathbb{R}^n \to \mathbb{R}$ by $f(x) = \|x\|$.
Apply Monotonicity for Functions.

\medskip

% --- define f(x) = ||x||, apply Monotonicity for Functions ---

\end{proof}

One-line reduction to Monotonicity for Functions.
