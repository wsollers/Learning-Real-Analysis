\subsubsection[Density of $\mathbb{Q}$ in $\mathbb{R}$]{Proof --- Density of $\mathbb{Q}$ in $\mathbb{R}$}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Theorem~1.4.3; see notes
\texttt{notes-completeness.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Show $\mathbb{Q}$ is dense in $\mathbb{R}$:
for every $a < b$ in $\mathbb{R}$, there exists $q \in \mathbb{Q}$ with
$a < q < b$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \forall a, b \in \mathbb{R},\; a < b
  \;\Rightarrow\;
  \exists q \in \mathbb{Q},\; a < q < b.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
By the Archimedean Property, choose $n \in \mathbb{N}$ with $n > 1/(b-a)$,
i.e.\ $nb - na > 1$. By the Floor Lemma, let $m = \lfloor na \rfloor + 1$.
Then $m > na$, and because $nb > na + 1 \ge m$, we get $a < m/n < b$.
The rational $q = m/n$ is the witness.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $\forall a < b \in \mathbb{R},\; \exists q \in \mathbb{Q},\; a < q < b$.

\Given $a, b \in \mathbb{R}$ with $a < b$.

\Goal Exhibit $q = m/n \in \mathbb{Q}$ with $a < q < b$.

\Strategy Use Archimedean to find $n$ large enough that the interval
$(na, nb)$ has length $> 1$, hence contains an integer $m$.

\medskip

\end{proof}

\begin{remark*}[Proof shape]
Constructive: build the rational $m/n$ explicitly using the Archimedean
Property and Floor Lemma.
\end{remark*}

  \item Floor Lemma.
\end{itemize}
\end{remark*}
