\subsubsection[Density of Irrationals]{Proof --- Density of Irrationals}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Corollary~1.4.4; see notes
\texttt{notes-completeness.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Between any two distinct real numbers lies an irrational number.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \forall a < b \in \mathbb{R},\;
  \exists t \in \mathbb{R} \setminus \mathbb{Q},\;
  a < t < b.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Apply Density of $\mathbb{Q}$ to the interval $(a/\sqrt{2},\, b/\sqrt{2})$:
find a rational $q$ in that interval. Then $t = q\sqrt{2}$ is irrational
(product of a nonzero rational and an irrational), and $a < t < b$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $\forall a < b \in \mathbb{R},\; \exists t \notin \mathbb{Q}$ with $a < t < b$.

\Given $a < b$ in $\mathbb{R}$.

\Goal Find irrational $t \in (a, b)$.

\Strategy Shift the interval by $1/\sqrt{2}$, apply density of $\mathbb{Q}$,
then multiply back by $\sqrt{2}$.

\medskip

% --- apply density of Q to (a/√2, b/√2): get rational q ---

% --- set t = q√2, verify a < t < b ---

% --- verify t is irrational (q ≠ 0 and √2 irrational) ---

\end{proof}

\begin{remark*}[Proof shape]
Reduction to Density of $\mathbb{Q}$ via scaling by $1/\sqrt{2}$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Density of $\mathbb{Q}$ in $\mathbb{R}$.
  \item $\sqrt{2}$ is irrational (proved elsewhere or taken as known).
  \item A nonzero rational times an irrational is irrational.
\end{itemize}
\end{remark*}
