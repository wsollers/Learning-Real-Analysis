\subsubsection[Monotonicity for Functions]{Proof --- Monotonicity for Functions}

\bigskip

\noindent
\textbf{Source.}
See notes \texttt{notes-bounding-monotonicty.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Let $f : X \to \mathbb{R}$ and $A \subseteq B \subseteq X$.
Show $\sup_{x \in A} f(x) \le \sup_{x \in B} f(x)$, and
$\inf_{x \in A} f(x) \ge \inf_{x \in B} f(x)$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  A \subseteq B
  \;\Rightarrow\;
  \sup_{x \in A} f(x) \le \sup_{x \in B} f(x)
  \quad\text{and}\quad
  \inf_{x \in A} f(x) \ge \inf_{x \in B} f(x).
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Define image sets $f(A) = \{f(x) : x \in A\}$ and $f(B) = \{f(x) : x \in B\}$.
Observe $A \subseteq B$ implies $f(A) \subseteq f(B)$. Then apply the
Monotonicity of Supremum theorem to these image sets.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim For $A \subseteq B \subseteq X$ and $f: X \to \mathbb{R}$,
$\sup_{A} f \le \sup_{B} f$ and $\inf_{A} f \ge \inf_{B} f$.

\Given $A \subseteq B$, $f : X \to \mathbb{R}$.

\Goal Show the sup and inf inequalities.

\Strategy Reduce to the set-monotonicity result via image sets.

\end{proof}

\begin{remark*}[Proof shape]
Reduction: translate function suprema to set suprema of image sets, then
apply the already-proved set-monotonicity result.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Monotonicity of Supremum and Infimum (set version).
  \item Definition: $\sup_{x \in A} f(x) = \sup f(A)$ where
        $f(A) := \{f(x) : x \in A\}$.
\end{itemize}
\end{remark*}
