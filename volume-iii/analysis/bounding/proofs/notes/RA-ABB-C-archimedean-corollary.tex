\subsubsection[Archimedean Property Corollary]{Proof --- Archimedean Property Corollary}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Corollary~1.4.3; see \texttt{notes-completeness.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $x > 0$ and $y \in \mathbb{R}$, then $\exists n \in \mathbb{N}$ such that $nx > y$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \forall x > 0,\; \forall y \in \mathbb{R},\; \exists n \in \mathbb{N},\; nx > y.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Apply the Archimedean property to the real number $y/x$; the resulting $n$ satisfies $n > y/x$, i.e.\ $nx > y$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $x > 0$ and $y \in \mathbb{R}$, then $\exists n \in \mathbb{N}$ such that $nx > y$.

\Given 

\Goal 

\Strategy Apply the Archimedean property to the real number $y/x$; the resulting $n$ satisfies $n > y/x$, i.e.\ $nx > y$.

\medskip

\end{proof}

\begin{remark*}[Proof shape]
Direct reduction to the Archimedean property by rescaling.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Archimedean Property ($\forall z \in \mathbb{R},\; \exists n \in \mathbb{N},\; n > z$).
  \item Positivity of $x$ (to multiply both sides and preserve inequality).
\end{itemize}
\end{remark*}
