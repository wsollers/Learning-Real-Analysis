\subsubsection[Floor Lemma]{Proof --- Floor Lemma}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}; see notes \texttt{notes-completeness.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
For every $x \in \mathbb{R}$, there exists a unique $m \in \mathbb{Z}$ with
$m \le x < m + 1$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \forall x \in \mathbb{R},\; \exists!\, m \in \mathbb{Z},\; m \le x < m+1.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
\textbf{Existence:} By the Archimedean property there exists $k \in \mathbb{N}$
with $k > -x$, so the set $S = \{n \in \mathbb{Z} : n \le x\}$ is bounded
above. By the well-ordering / Archimedean argument, $S$ has a largest element
$m$; then $m \le x < m + 1$.
\textbf{Uniqueness:} If $m$ and $m'$ both satisfy the condition, then
$|m - m'| < 1$ and $m - m' \in \mathbb{Z}$, forcing $m = m'$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $\forall x \in \mathbb{R},\; \exists!\, m \in \mathbb{Z},\; m \le x < m+1$.

\Given $x \in \mathbb{R}$.

\Goal Find and show unique $m \in \mathbb{Z}$ with $m \le x < m+1$.

\Strategy Existence via Archimedean Property and the structure of $\mathbb{Z}$.
Uniqueness via $|m - m'| < 1$ and integrality.

\medskip

% --- Existence ---

% --- Uniqueness ---

\end{proof}

\begin{remark*}[Proof shape]
Two-part: construct the floor via the Archimedean Property; prove uniqueness
by showing two candidates differ by less than $1$ in absolute value yet are integers.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Archimedean Property.
  \item Well-ordering of $\mathbb{N}$ (or equivalently: there is no infinite
        strictly decreasing sequence of integers).
\end{itemize}
\end{remark*}
