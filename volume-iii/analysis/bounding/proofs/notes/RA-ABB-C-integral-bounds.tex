\subsubsection[Integral Bounds]{Proof --- Integral Bounds}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-bounding-monotonicty.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $f$ is bounded on $[a,b]$ with $\inf f([a,b]) \ge m$ and $\sup f([a,b]) \le M$, then $m(b-a) \le \int_a^b f \le M(b-a)$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  m(b-a) \le \int_a^b f \le M(b-a).
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
The constant functions $m$ and $M$ bound $f$ pointwise. Monotonicity of the integral then gives the bounds.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $f$ is bounded on $[a,b]$ with $\inf f([a,b]) \ge m$ and $\sup f([a,b]) \le M$, then $m(b-a) \le \int_a^b f \le M(b-a)$.

\Given 

\Goal 

\Strategy The constant functions $m$ and $M$ bound $f$ pointwise. Monotonicity of the integral then gives the bounds.

\medskip

% --- state: m ≤ f(x) ≤ M for all x ∈ [a,b] ---
% --- integrate all three sides over [a,b]: m(b-a) ≤ ∫f ≤ M(b-a) ---
% --- cite monotonicity of integration ---

\end{proof}

\begin{remark*}[Proof shape]
Pointwise bounds on $f$ integrate to bounds on $\int f$ by monotonicity of the Riemann integral.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Monotonicity of the Riemann integral (if $f \le g$ pointwise then $\int f \le \int g$).
  \item Definition of $\inf$ and $\sup$ applied to the image $f([a,b])$.
\end{itemize}
\end{remark*}
