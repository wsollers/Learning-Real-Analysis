\begin{theorem*}[Monotonicity of the Infimum]
Let $A,B \subseteq \mathbb{R}$ be nonempty and bounded below.
If $A \subseteq B$, then
\[
\inf B \le \inf A .
\]
\end{theorem*}

\begin{proof}
Since $A$ is bounded below, $\inf A$ exists and is a lower bound of $A$.
Thus
\[
\inf A \le a \quad \text{for all } a \in A .
\]

Because $A \subseteq B$, every lower bound of $B$ is also a lower bound of $A$.
In particular, $\inf B$ is a lower bound of $A$.

Since $\inf A$ is the greatest lower bound of $A$, it follows that
\[
\inf B \le \inf A .
\]
\end{proof}
