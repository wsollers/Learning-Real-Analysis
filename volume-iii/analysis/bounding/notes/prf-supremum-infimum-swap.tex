\subsection*{Negation Rule and Scalar Multiplication ($c < 0$)}
\label{prf:supremum-infimum-swap}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:neg-rule]{Back to Theorem in Notes}
\end{remark*}

\begin{theorem}
Let $A \subset \mathbb{R}$ be bounded above and let $c<0$.
Then
\[
\sup(cA)=c\,\inf A .
\]
\end{theorem}

\begin{proof}
Let
\[
s_A = \sup(cA), \qquad i_A = \inf A .
\]

Since $i_A=\inf A$, we have
\[
i_A \le a \qquad \forall a \in A.
\]

Multiplying by $c$ and noting that $c<0$ reverses the inequality gives
\[
c i_A \ge c a \qquad \forall a \in A.
\]

Thus $c i_A$ is an upper bound for the set $cA=\{ca : a\in A\}$.
Since $s_A=\sup(cA)$ is the least upper bound, it follows that
\[
s_A \le c i_A.
\]

Now let $\varepsilon>0$ be arbitrary. Because $i_A=\inf A$, there exists
$a\in A$ such that
\[
a < i_A + \frac{\varepsilon}{|c|}.
\]

Multiplying by $c$ (again noting $c<0$ reverses the inequality) yields
\[
c a > c\!\left(i_A + \frac{\varepsilon}{|c|}\right)
     = c i_A - \varepsilon .
\]

Hence
\[
c i_A - \varepsilon < c a \le s_A .
\]

Therefore
\[
c i_A - \varepsilon < s_A .
\]

Since $\varepsilon>0$ was arbitrary, we conclude
\[
c i_A \le s_A .
\]

Combining the two inequalities gives
\[
s_A = c i_A .
\]

Thus
\[
\sup(cA)=c\,\inf A.
\]
\end{proof}

\begin{remark*}[Proof shape]
Two-part argument: exhibit the candidate value as an upper (lower) bound,
then show no smaller (larger) value qualifies using the
$\varepsilon$-characterization of the supremum (infimum).
\end{remark*}

\begin{remark*}[Dependencies]
Definition of supremum (infimum); \hyperref[prop:eps-char]{$\varepsilon$-Characterization, Proposition~\ref*{prop:eps-char}}.
\end{remark*}
