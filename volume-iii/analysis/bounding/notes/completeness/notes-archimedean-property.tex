\subsection{Archimedean Property}

\begin{remark*}[Why this requires proof]
The Archimedean Property states that the natural numbers are unbounded in
$\mathbb{R}$. That conclusion is not a formal consequence of the ordered-field
axioms alone. There exist ordered fields satisfying all ordered-field axioms in
which the natural numbers are bounded above, so the statement must be justified
from stronger structure.

The needed structure is completeness. If $\mathbb{N}$ were bounded above in
$\mathbb{R}$, then the Least Upper Bound Property would produce a supremum
$\alpha \in \mathbb{R}$ for $\mathbb{N}$. But then $\alpha - 1$ cannot be an
upper bound, so there exists $n \in \mathbb{N}$ with $n > \alpha - 1$. Hence
$n+1 > \alpha$, contradicting the fact that $\alpha$ is an upper bound of
$\mathbb{N}$.

\smallskip
\noindent
\hyperref[prf:archimedean-property]{\textit{Go to proof.}}
\end{remark*}




\begin{theorem}[Archimedean Property]
\label{thm:archimedean-property}
For every real number $x$, there exists a natural number $n$ such that
\[
n > x.
\]
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
\forall x \in \mathbb{R}\; \exists n \in \mathbb{N}\; (n > x).
\]
\end{remark*}

\begin{remark*}[Theorem predicate reading]
\[
ArchimedeanProperty.
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists x \in \mathbb{R}\; \forall n \in \mathbb{N}\; (n \le x).
\]
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\neg ArchimedeanProperty
\quad\Longleftrightarrow\quad
\underbrace{\exists x \in \mathbb{R}\; \forall n \in \mathbb{N}\; (n \le x)}_{\text{the natural numbers are bounded above in }\mathbb{R}}.
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\neg ArchimedeanProperty
\quad\Longleftrightarrow\quad
\exists x \in \mathbb{R}\; \forall n \in \mathbb{N}\; (n \le x).
\]
\end{remark*}

\begin{remark*}[Failure modes]\hfill
\begin{itemize}
  \item \textbf{Failure of unboundedness.} There exists a real number $x$ that bounds every natural number from above.
  \item \textbf{Failure of Archimedean behavior.} The copy of $\mathbb{N}$ inside $\mathbb{R}$ would stop growing past a fixed real number.
\end{itemize}
\end{remark*}

\begin{remark*}[Interpretation]
The Archimedean Property says that the natural numbers are unbounded in $\mathbb{R}$.
No matter which real number is chosen, some natural number lies beyond it.

The point is that this is not a formal consequence of the ordered-field axioms alone.
It is a completeness consequence: the real line contains no infinitely large upper bound
for $\mathbb{N}$.
\end{remark*}















\begin{corollary}[Archimedean Reciprocal Corollary]
\label{cor:archimedean-reciprocal}
For every $x > 0$ and every $y \in \mathbb{R}$ there exists $n \in \mathbb{N}$ such that
\[
nx > y.
\]
\smallskip
\noindent
\hyperref[prf:archimedean-reciprocal]{\textit{Go to proof.}}
\end{corollary}

\begin{remark*}[Standard quantified statement]
\[
\forall x \in \mathbb{R}\; \forall y \in \mathbb{R}\;
\bigl(
x > 0 \rightarrow \exists n \in \mathbb{N}\; (nx > y)
\bigr).
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\exists x \in \mathbb{R}\; \exists y \in \mathbb{R}\;
\bigl(
x > 0 \land \forall n \in \mathbb{N}\; (nx \le y)
\bigr).
\]
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\neg\Bigl(
\forall x \in \mathbb{R}\; \forall y \in \mathbb{R}\;
\bigl(
x > 0 \rightarrow \exists n \in \mathbb{N}\; (nx > y)
\bigr)
\Bigr)
\]
\[
\Longleftrightarrow\quad
\underbrace{\exists x \in \mathbb{R}\; \exists y \in \mathbb{R}\;
\bigl(
x > 0 \land \forall n \in \mathbb{N}\; (nx \le y)
\bigr)}_{\text{a positive real number has all of its natural multiples bounded above}}.
\]
\end{remark*}

\begin{remark*}[Failure modes]\hfill
\begin{itemize}
  \item \textbf{Failure of growth under repeated addition.} There exists a positive real number $x$ whose natural multiples never exceed some fixed real number $y$.
  \item \textbf{Failure of Archimedean behavior.} Repeatedly adding a positive amount would not eventually pass every prescribed real threshold.
\end{itemize}
\end{remark*}

\begin{remark*}[Interpretation]
This corollary says that every positive real number, no matter how small, eventually exceeds
any prescribed real number when multiplied by a sufficiently large natural number.

The point is that positivity alone forces unbounded growth under repeated addition. A fixed
positive step size cannot remain trapped below a real bound forever.
\end{remark*}
