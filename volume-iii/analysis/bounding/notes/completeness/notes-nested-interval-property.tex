\subsection{Nested Interval Property}









\begin{theorem}[Nested Interval Property]
\label{thm:nested-interval-theorem}
Let $(I_n)$ be a sequence of closed intervals in $\mathbb{R}$. If $(I_n)$ is a nested
interval sequence, then
\[
\bigcap_{n=1}^{\infty} I_n \neq \varnothing.
\]
\smallskip
\noindent
\hyperref[prf:nested-interval-theorem]{\textit{Go to proof.}}
\end{theorem}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\forall (a_n) \subseteq \mathbb{R}\; \forall (b_n) \subseteq \mathbb{R}\;
\Biggl[
\Bigl(\forall n \in \mathbb{N}\; (a_n \le b_n)\Bigr)
\\
&\qquad\land
\Bigl(\forall n \in \mathbb{N}\; (I_n = [a_n,b_n])\Bigr)
\land
\Bigl(\forall n \in \mathbb{N}\; (I_{n+1} \subseteq I_n)\Bigr)
\\
&\qquad\rightarrow
\exists x \in \mathbb{R}\; \forall n \in \mathbb{N}\; (x \in I_n)
\Biggr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Theorem predicate reading]
\[
\forall (I_n)\;
\Bigl[
\underbrace{\operatorname{NestedIntervalSequence}(I_n)}_{\text{the intervals are nested}}
\rightarrow
\underbrace{\operatorname{NonemptySet}\!\left(\bigcap_{n=1}^{\infty} I_n\right)}_{\text{their total intersection is nonempty}}
\Bigr].
\]
In this theorem, each $I_n$ is additionally assumed to be a closed interval in $\mathbb{R}$.
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\exists (a_n) \subseteq \mathbb{R}\; \exists (b_n) \subseteq \mathbb{R}\;
\Biggl[
\Bigl(\forall n \in \mathbb{N}\; (a_n \le b_n)\Bigr)
\\
&\qquad\land
\Bigl(\forall n \in \mathbb{N}\; (I_n = [a_n,b_n])\Bigr)
\land
\Bigl(\forall n \in \mathbb{N}\; (I_{n+1} \subseteq I_n)\Bigr)
\\
&\qquad\land
\forall x \in \mathbb{R}\; \exists n \in \mathbb{N}\; (x \notin I_n)
\Biggr].
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\exists (I_n)\;
\Bigl[
\underbrace{\operatorname{NestedIntervalSequence}(I_n)}_{\text{the intervals remain nested}}
\land
\underbrace{\neg \operatorname{NonemptySet}\!\left(\bigcap_{n=1}^{\infty} I_n\right)}_{\text{the total intersection is empty}}
\Bigr].
\]
\end{remark*}

\begin{remark*}[Negation predicate reading]
\[
\exists (I_n)\;
\Bigl[
\operatorname{NestedIntervalSequence}(I_n)
\land
\neg \operatorname{NonemptySet}\!\left(\bigcap_{n=1}^{\infty} I_n\right)
\Bigr].
\]
Equivalently, there exists a nested sequence of closed intervals in $\mathbb{R}$ whose total
intersection is empty.
\end{remark*}

\begin{remark*}[Failure modes]\hfill
\begin{itemize}
  \item \textbf{Failure of common containment.} The intervals remain nested, but no single real number lies in all of them.
  \item \textbf{Failure of completeness behavior.} The nesting construction keeps restricting location, yet the real line would still fail to retain a common point.
\end{itemize}
\end{remark*}

\begin{remark*}[Interpretation]
The Nested Interval Property says that a decreasing sequence of closed intervals in
$\mathbb{R}$ cannot shrink in such a way that every point is eventually lost. If each interval
lies inside the previous one, then at least one real number remains in all of them.

The important point is that nesting prevents the intervals from wandering apart, while
closedness prevents limiting endpoint behavior from dropping the common point out of the
intersection.
\end{remark*}
