\subsection*{Scalar Multiplication ($c > 0$)}
\label{prf:infimum-positive-scalar-multiple}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:scale-pos]{Back to Theorem in Notes}
\end{remark*}

\begin{theorem*}[Infimum of a Scalar Multiple]
Let $A \subseteq \mathbb{R}$ be nonempty and bounded below, and let $c>0$.
Define
\[
cA := \{ca : a \in A\}.
\]
Then
\[
\inf(cA) = c\,\inf A.
\]
\end{theorem*}

\begin{proof}
Let
\[
\alpha := \inf A.
\]

First we show that $c\alpha$ is a lower bound for $cA$.

Let $x \in cA$. Then $x = ca$ for some $a \in A$. Since $\alpha = \inf A$, we have
\[
\alpha \le a.
\]
Multiplying by $c>0$ preserves the inequality, giving
\[
c\alpha \le ca = x.
\]

Thus $c\alpha$ is a lower bound for $cA$, and therefore
\[
c\alpha \le \inf(cA).
\]

Now let $\varepsilon > 0$ be arbitrary. Because $\alpha = \inf A$, there exists
$a_\varepsilon \in A$ such that
\[
a_\varepsilon < \alpha + \frac{\varepsilon}{c}.
\]

Multiplying by $c$ gives
\[
c a_\varepsilon < c\alpha + \varepsilon.
\]

Since $c a_\varepsilon \in cA$, the definition of infimum implies
\[
\inf(cA) \le c a_\varepsilon.
\]

Therefore
\[
\inf(cA) < c\alpha + \varepsilon.
\]

Because $\varepsilon > 0$ was arbitrary, it follows that
\[
\inf(cA) \le c\alpha.
\]

Combining the inequalities
\[
c\alpha \le \inf(cA)
\quad\text{and}\quad
\inf(cA) \le c\alpha,
\]
we conclude that
\[
\inf(cA) = c\alpha = c\,\inf A.
\]
\end{proof}

\begin{remark*}[Proof shape]
Two-part argument: exhibit the candidate value as an upper (lower) bound,
then show no smaller (larger) value qualifies using the
$\varepsilon$-characterization of the supremum (infimum).
\end{remark*}

\begin{remark*}[Dependencies]
Definition of supremum (infimum); \hyperref[prop:eps-char]{$\varepsilon$-Characterization, Proposition~\ref*{prop:eps-char}}.
\end{remark*}
