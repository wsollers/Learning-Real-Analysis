\begin{remark*}[Task]
Let
\[
A=\left\{\frac{1}{n}:n\in\mathbb{N}\right\}.
\]
Determine \(\sup A\) and \(\inf A\).
\end{remark*}

\begin{remark*}[Technique]
\TechniqueEntry{tech:unpack-definition},
\TechniqueEntry{tech:choose-N},
\TechniqueEntry{tech:supremum-characterization}
\end{remark*}

\begin{remark*}[Construction]
We claim that
\[
\sup A = 1
\qquad\text{and}\qquad
\inf A = 0.
\]
\end{remark*}

\begin{proof}
First we show that
\[
\lim_{n\to\infty}\frac{1}{n}=0.
\]

Let \(\varepsilon>0\) be arbitrary. By the Archimedean property, there exists
\(N\in\mathbb{N}\) such that
\[
N>\frac{1}{\varepsilon}.
\]
Then for every \(n\ge N\), we have
\[
n\ge N>\frac{1}{\varepsilon}.
\]
Taking reciprocals gives
\[
\frac{1}{n}\le \frac{1}{N}<\varepsilon.
\]
Hence
\[
\left|\frac{1}{n}-0\right|=\frac{1}{n}<\varepsilon.
\]
Since this holds for every \(\varepsilon>0\), it follows that
\[
\frac{1}{n}\to 0 \qquad \text{as } n\to\infty.
\]

Now
\[
A=\left\{1,\frac12,\frac13,\dots\right\}.
\]

Because \(1\in A\) and \(\frac1n\le 1\) for all \(n\in\mathbb{N}\), the set \(A\)
has maximum \(1\). Therefore
\[
\sup A = 1.
\]

Also, for every \(n\in\mathbb{N}\),
\[
0<\frac1n,
\]
so \(0\) is a lower bound of \(A\).

To show that \(0\) is the greatest lower bound, let \(\varepsilon>0\) be given.
Since \(\frac1n\to 0\), there exists \(N\in\mathbb{N}\) such that
\[
\frac1N<\varepsilon.
\]
Then \(\frac1N\in A\) and
\[
0\le \frac1N < \varepsilon.
\]
Thus no positive number can be a lower bound of \(A\). Hence
\[
\inf A = 0.
\]

Therefore
\[
\sup A = 1
\qquad\text{and}\qquad
\inf A = 0.
\]
\end{proof}

\begin{remark*}[Dependencies]
We used the definition of convergence, the Archimedean property,
the fact that a maximum is the supremum, and the definition of infimum.
\end{remark*}

\begin{remark*}[Proof shape]
First prove \(\frac1n\to 0\) with an \(\varepsilon\)–\(N\) argument. Then identify
the maximum of the set to get the supremum, and finally use the limit together
with the definition of greatest lower bound to compute the infimum.
\end{remark*}

\begin{remark*}[Reflection]
The key structural idea is that the supremum is easy because the set actually
has a largest element, while the infimum is not attained and must instead be
shown by approximation from above.
\end{remark*}
