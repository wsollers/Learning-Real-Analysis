\subsection{Bounds and Extremal Values --- Reference to Bounding Chapter}

\begin{tcolorbox}[colback=gray!6, colframe=gray!40, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definitions and results from the Bounding chapter}},
  fonttitle=\small\bfseries]
\begin{tabular}{@{}p{0.30\textwidth}p{0.65\textwidth}@{}}
\textbf{Upper/lower bound} &
  $u$ is an upper bound for $A$ iff $\forall a \in A,\, a \le u$. \\[0.3em]
\textbf{Supremum} &
  $s = \sup A$ iff (i) $s$ is an upper bound, and
  (ii) $\forall u$ upper bound, $s \le u$. \\[0.3em]
\textbf{$\varepsilon$-criterion} &
  $s = \sup A$ iff $s$ is an upper bound and
  $\forall \varepsilon > 0,\, \exists a \in A,\, s - \varepsilon < a$. \\[0.3em]
\textbf{Infimum} & symmetric to supremum. \\[0.3em]
\textbf{Max/Min} & extrema that belong to the set itself. \\
\end{tabular}
