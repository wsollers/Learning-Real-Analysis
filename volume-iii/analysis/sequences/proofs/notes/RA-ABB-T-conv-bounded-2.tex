\subsubsection[Convergent Sequence Is Bounded]{Proof --- Convergent Sequence Is Bounded}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Theorem~2.3.2; see \texttt{notes-convergence.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $(x_n)$ converges, then $(x_n)$ is bounded.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  x_n \to L \Rightarrow \exists M > 0,\; \forall n,\; |x_n| \le M.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Use $\varepsilon = 1$: beyond some $N$, all terms lie in $(L-1, L+1)$. The first $N-1$ terms are finitely many; take $M = \max\{|x_1|,\ldots,|x_{N-1}|, |L|+1\}$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $(x_n)$ converges, then $(x_n)$ is bounded.

\Given 

\Goal 

\Strategy Use $\varepsilon = 1$: beyond some $N$, all terms lie in $(L-1, L+1)$. The first $N-1$ terms are finitely many; take $M = \max\{|x_1|,\ldots,|x_{N-1}|, |L|+1\}$.

\medskip

% --- so |x_n| ≤ |L| + 1 for all n ≥ N ---
% --- let M = max{|x_1|, ..., |x_{N-1}|, |L|+1} ---
% --- M is finite (finitely many terms); |x_n| ≤ M for all n ---

\end{proof}

\begin{remark*}[Proof shape]
Use $\varepsilon = 1$ to control the tail; handle the finite head by taking a maximum.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item $\varepsilon$-$N$ definition of convergence.
  \item Triangle inequality: $|x_n| = |x_n - L + L| \le |x_n - L| + |L|$.
  \item Finite maximum is well-defined.
\end{itemize}
\end{remark*}
