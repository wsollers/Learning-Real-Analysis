\subsubsection[Fekete's Lemma (Superadditive Sequences)]{Proof --- Fekete's Lemma (Superadditive Sequences)}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-recurrence-relations.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $a_{m+n} \ge a_m + a_n$ (superadditive), then $\lim_{n\to\infty} a_n/n = \sup_{n \ge 1} a_n/n$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  a_{m+n} \ge a_m + a_n \Rightarrow \lim_{n\to\infty} \frac{a_n}{n} = \sup_{n\ge 1} \frac{a_n}{n}.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Apply Fekete's Subadditive Lemma to $b_n = -a_n$, which satisfies $b_{m+n} \le b_m + b_n$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $a_{m+n} \ge a_m + a_n$ (superadditive), then $\lim_{n\to\infty} a_n/n = \sup_{n \ge 1} a_n/n$.

\Given 

\Goal 

\Strategy Apply Fekete's Subadditive Lemma to $b_n = -a_n$, which satisfies $b_{m+n} \le b_m + b_n$.

\medskip

% --- set b_n = -a_n; check b_{m+n} = -a_{m+n} ≤ -(a_m + a_n) = b_m + b_n (subadditive) ---
% --- apply Fekete subadditive: lim b_n/n = inf b_n/n ---
% --- negate: lim a_n/n = sup a_n/n ---

\end{proof}

\begin{remark*}[Proof shape]
Reduction to the subadditive case by negation.
\end{remark*}
