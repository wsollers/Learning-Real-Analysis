\subsubsection[Convergence is a Tail Property]{Proof --- Convergence is a Tail Property}

\bigskip

\noindent
\textbf{Source.}
Tao, \emph{Analysis I}; see notes \texttt{notes-k-tails.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
$(x_n) \to L$ if and only if the $k$-tail $(x_{n+k})_{n \ge 1}$ converges
to $L$, for any fixed $k \ge 0$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \forall k \ge 0,\;
  (x_n) \to L
  \;\iff\;
  (x_{n+k}) \to L.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Given $\varepsilon > 0$: for $(\Rightarrow)$, the $N$ witnessing convergence
of $(x_n)$ also works for the tail (since $n \ge N$ implies $n + k \ge N$).
For $(\Leftarrow)$, use the tail's $N$ and observe $n \ge N + k$ suffices.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $(x_n) \to L \iff (x_{n+k}) \to L$ for any fixed $k \ge 0$.

\Given A real sequence $(x_n)$, a limit $L$, a fixed $k \ge 0$.

\Goal Prove the biconditional.

\Strategy Both directions involve unfolding the definition of convergence
and adjusting the threshold index by $k$.

\medskip

% --- (⇒): given N for (x_n), show same N works for (x_{n+k}) ---

% --- (⇐): given N for tail, shift by k for (x_n) ---

\end{proof}

\begin{remark*}[Proof shape]
Both directions are direct $\varepsilon$-$N$ arguments where the witness
threshold shifts by $k$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of convergence.
  \item If $n \ge N$ and $k \ge 0$ then $n + k \ge N$.
\end{itemize}
\end{remark*}
