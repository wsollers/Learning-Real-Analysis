\subsubsection[Limit Respects Upper Bounds]{Proof --- Limit Respects Upper Bounds}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}; see \texttt{notes-algebra-of-sequences.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $x_n \le b$ for all $n$ and $x_n \to x$, then $x \le b$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  (\forall n,\; x_n \le b) \wedge (x_n \to x) \Rightarrow x \le b.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Apply the Order Limit Theorem with $(y_n) = (b, b, b, \ldots)$ (constant sequence converging to $b$).

\vspace{0.75em}

\noindent
