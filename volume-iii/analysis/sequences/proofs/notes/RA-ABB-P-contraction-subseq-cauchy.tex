\subsubsection[Contraction Subsequence Is Cauchy]{Proof --- Contraction Subsequence Is Cauchy}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-recurrence-relations.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $(x_{n_k})$ is a contractive subsequence of $(x_n)$ and the full sequence is such that $|x_{n+1} - x_n| \le c |x_n - x_{n-1}|$, then $(x_n)$ is Cauchy.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \text{Contractive recurrence} \Rightarrow (x_n) \text{ Cauchy}.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} The recurrence $|x_{n+1} - x_n| \le c^n |x_1 - x_0|$ follows by the Iterative Propagation Lemma, then apply Contraction Sequence Is Cauchy.
