\subsubsection[Monotonicity of Tail Suprema and Infima]{Proof --- Monotonicity of Tail Suprema and Infima}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}; see \texttt{notes-limsup-liminf.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $s_n = \sup\{x_k : k \ge n\}$ and $i_n = \inf\{x_k : k \ge n\}$, then $(s_n)$ is decreasing and $(i_n)$ is increasing.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  (s_n) \text{ decreasing},\; (i_n) \text{ increasing}.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Use subset monotonicity of $\sup$ and $\inf$: $T_{n+1} \subseteq T_n$ so $\sup T_{n+1} \le \sup T_n$ and $\inf T_{n+1} \ge \inf T_n$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $s_n = \sup\{x_k : k \ge n\}$ and $i_n = \inf\{x_k : k \ge n\}$, then $(s_n)$ is decreasing and $(i_n)$ is increasing.

\Given 

\Goal 

\Strategy Use subset monotonicity of $\sup$ and $\inf$: $T_{n+1} \subseteq T_n$ so $\sup T_{n+1} \le \sup T_n$ and $\inf T_{n+1} \ge \inf T_n$.

\medskip

% --- T_{n+1} ⊆ T_n (tail containment) ---
% --- A ⊆ B ⇒ sup A ≤ sup B (Monotonicity of Supremum) ---
% --- A ⊆ B ⇒ inf A ≥ inf B (Monotonicity of Infimum) ---

\end{proof}

\begin{remark*}[Proof shape]
Nested tail sets $\Rightarrow$ monotone tail suprema/infima by the subset monotonicity of $\sup$ and $\inf$.
\end{remark*}
