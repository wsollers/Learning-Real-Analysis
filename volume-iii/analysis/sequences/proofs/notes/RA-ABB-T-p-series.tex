\subsubsection[p-Series]{Proof --- p-Series}

\bigskip

\noindent
\textbf{Source.} Abbott, \emph{Understanding Analysis}; see \texttt{notes-series-tests.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} $\sum_{n=1}^\infty 1/n^p < \infty$ if and only if $p > 1$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \sum_{n=1}^\infty \frac{1}{n^p} < \infty \Longleftrightarrow p > 1.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Apply the Integral Test with $f(x) = x^{-p}$. The integral $\int_1^\infty x^{-p}\,dx$ converges iff $p > 1$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $\sum_{n=1}^\infty 1/n^p < \infty$ if and only if $p > 1$.
