\subsubsection[Order Preservation for Limsup and Liminf]{Proof --- Order Preservation for Limsup and Liminf}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-limsup-liminf.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $x_n \le y_n$ for all $n$, then $\limsup x_n \le \limsup y_n$ and $\liminf x_n \le \liminf y_n$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  x_n \le y_n \Rightarrow \limsup x_n \le \limsup y_n.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Pointwise domination gives $\sup T_k(x) \le \sup T_k(y)$ for each $k$; take limits.

\vspace{0.75em}

\noindent
