\subsubsection[Radius of Convergence Theorem]{Proof --- Radius of Convergence Theorem}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-power-series.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} For a power series $\sum a_n x^n$, there exists $R \in [0,\infty]$ such that the series converges absolutely for $|x| < R$ and diverges for $|x| > R$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \exists R \in [0,\infty],\; \sum a_n x^n \text{ abs.\ conv.\ for } |x|<R,\; \text{div.\ for } |x|>R.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Apply the Root Test to $\sum a_n x^n$: $\limsup |a_n x^n|^{1/n} = |x| \limsup |a_n|^{1/n}$. Set $1/R = \limsup |a_n|^{1/n}$ (Cauchy-Hadamard). Root Test gives convergence iff $|x|/R < 1$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim For a power series $\sum a_n x^n$, there exists $R \in [0,\infty]$ such that the series converges absolutely for $|x| < R$ and diverges for $|x| > R$.

\Given 

\Goal 

\Strategy Apply the Root Test to $\sum a_n x^n$: $\limsup |a_n x^n|^{1/n} = |x| \limsup |a_n|^{1/n}$. Set $1/R = \limsup |a_n|^{1/n}$ (Cauchy-Hadamard). Root Test gives convergence iff $|x|/R < 1$.

\end{proof}

\begin{remark*}[Proof shape]
Root Test applied to power series; the key is that $|x|$ factors out of the $\limsup$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Root Test for series.
  \item Scalar multiplication for $\limsup$: $\limsup |cx_n| = |c| \limsup |x_n|$.
\end{itemize}
\end{remark*}
