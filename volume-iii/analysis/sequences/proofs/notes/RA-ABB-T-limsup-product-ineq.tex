\subsubsection[Product Inequality for Nonnegative Sequences]{Proof --- Product Inequality for Nonnegative Sequences}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-limsup-liminf.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $x_n, y_n \ge 0$, then $\limsup(x_n y_n) \le (\limsup x_n)(\limsup y_n)$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  x_n, y_n \ge 0 \Rightarrow \limsup(x_n y_n) \le (\limsup x_n)(\limsup y_n).
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} $\sup(AB) \le (\sup A)(\sup B)$ for nonneg $A, B$: each $x_k \le \sup A$ and $y_k \le \sup B$, so $x_k y_k \le (\sup A)(\sup B)$. Apply to tail sets and take limits.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $x_n, y_n \ge 0$, then $\limsup(x_n y_n) \le (\limsup x_n)(\limsup y_n)$.

\Given 

\Goal 

\Strategy $\sup(AB) \le (\sup A)(\sup B)$ for nonneg $A, B$: each $x_k \le \sup A$ and $y_k \le \sup B$, so $x_k y_k \le (\sup A)(\sup B)$. Apply to tail sets and take limits.

\medskip

% --- lim(s_n(x)·s_n(y)) = limsup x_n · limsup y_n ---
% --- so limsup(x_n y_n) ≤ (limsup x_n)(limsup y_n) ---

\begin{remark*}[Proof shape]
Pointwise bound on products $\Rightarrow$ bound on tail sups $\Rightarrow$ bound on $\limsup$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item $x_k \le \sup_{k \ge n} x_k$ and $y_k \le \sup_{k \ge n} y_k$ (nonneg, so product dominates).
  \item Order Limit Theorem.
\end{itemize}
\end{remark*}
