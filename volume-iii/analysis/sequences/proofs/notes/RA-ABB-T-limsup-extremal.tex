\subsubsection[Extremal Subsequences for Limsup and Liminf]{Proof --- Extremal Subsequences for Limsup and Liminf}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-limsup-liminf.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
For a bounded sequence $(x_n)$, there exist subsequences converging to $\limsup x_n$ and to $\liminf x_n$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \exists (x_{n_k}) \to \limsup x_n,\; \exists (x_{m_j}) \to \liminf x_n.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Use the subsequence characterization: $\limsup x_n$ is a subsequential limit (the largest one), and similarly for $\liminf$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim For a bounded sequence $(x_n)$, there exist subsequences converging to $\limsup x_n$ and to $\liminf x_n$.

\Given 

\Goal 

\Strategy Use the subsequence characterization: $\limsup x_n$ is a subsequential limit (the largest one), and similarly for $\liminf$.

\medskip

% --- limsup = L: apply characterization → ∃ subsequence (x_{n_k}) → L ---
% --- liminf = l: apply characterization with inf → ∃ subsequence (x_{m_j}) → l ---

\end{proof}

\begin{remark*}[Proof shape]
Direct consequence of the subsequence characterization of $\limsup$ and $\liminf$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Characterization of $\limsup$ via subsequences.
  \item Symmetric version for $\liminf$.
\end{itemize}
\end{remark*}
