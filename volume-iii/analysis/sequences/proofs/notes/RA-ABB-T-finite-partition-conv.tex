\subsubsection[Finite Partition Convergence Principle]{Proof --- Finite Partition Convergence Principle}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-subsequence-toolkit.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $\mathbb{N} = E_1 \cup \cdots \cup E_k$ (disjoint infinite partition) and each subsequence $(a_n)_{n \in E_i}$ converges to $L$, then $(a_n) \to L$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \text{Each piece converges to } L \Rightarrow (a_n) \to L.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Given $\varepsilon > 0$, for each $i$ get $N_i$ past which $(a_n)_{n \in E_i}$ is within $\varepsilon$ of $L$. Take $N = \max(N_1, \ldots, N_k)$; every $n \ge N$ belongs to some $E_i$ and so satisfies the $\varepsilon$ bound.

\vspace{0.75em}

\noindent
\textbf{Proof.}
