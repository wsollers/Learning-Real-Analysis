\subsubsection[Iterative Propagation Lemma]{Proof --- Iterative Propagation Lemma}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-recurrence-relations.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $|x_{n+1}| \le c|x_n|$ for all $n \ge N$ and $0 \le c < 1$, then $|x_n| \le c^{n-N}|x_N|$ for all $n \ge N$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  |x_n| \le c^{n-N}|x_N| \text{ for all } n \ge N.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Induction on $n - N$. Each step multiplies by $c$, so after $n - N$ steps the bound is $c^{n-N}|x_N|$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $|x_{n+1}| \le c|x_n|$ for all $n \ge N$ and $0 \le c < 1$, then $|x_n| \le c^{n-N}|x_N|$ for all $n \ge N$.

\Given 

\Goal 

\Strategy Induction on $n - N$. Each step multiplies by $c$, so after $n - N$ steps the bound is $c^{n-N}|x_N|$.

\medskip

% --- base n = N: |x_N| ≤ c^0 |x_N| = |x_N| ✓ ---
% --- inductive step: |x_{n+1}| ≤ c|x_n| ≤ c · c^{n-N}|x_N| = c^{n+1-N}|x_N| ---

\end{proof}

\begin{remark*}[Proof shape]
Induction on the number of steps, each multiplying the bound by $c$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Hypothesis: $|x_{n+1}| \le c|x_n|$ for $n \ge N$.
  \item Induction on $n$.
\end{itemize}
\end{remark*}
