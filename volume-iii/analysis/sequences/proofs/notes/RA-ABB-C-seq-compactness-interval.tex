\subsubsection[Sequential Compactness of Closed Intervals]{Proof --- Sequential Compactness of Closed Intervals}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}; see \texttt{notes-subsequences.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Every sequence in $[a,b]$ has a convergent subsequence with limit in $[a,b]$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \forall (a_n) \subseteq [a,b],\; \exists (a_{n_k}) \to L \in [a,b].
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
$(a_n)$ is bounded (by $a$ and $b$), so by BW it has a convergent subsequence $(a_{n_k}) \to L$. Since $[a,b]$ is closed, $L \in [a,b]$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim Every sequence in $[a,b]$ has a convergent subsequence with limit in $[a,b]$.

\Given 

\Goal 

\Strategy $(a_n)$ is bounded (by $a$ and $b$), so by BW it has a convergent subsequence $(a_{n_k}) \to L$. Since $[a,b]$ is closed, $L \in [a,b]$.

\medskip

% --- (a_n) ⊆ [a,b] ⇒ (a_n) bounded ---
% --- BW: ∃ convergent subsequence (a_{n_k}) → L ---
% --- [a,b] is closed and contains all a_{n_k}, so L ∈ [a,b] ---

\end{proof}

\begin{remark*}[Proof shape]
BW gives the subsequence; closedness of $[a,b]$ ensures the limit stays inside.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Bolzano--Weierstrass Theorem.
  \item Closed sets contain their limit points (proved in Metric Spaces chapter).
\end{itemize}
\end{remark*}
