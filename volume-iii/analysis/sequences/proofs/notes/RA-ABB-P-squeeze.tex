\subsubsection[Squeeze Theorem]{Proof --- Squeeze Theorem}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Theorem~2.3.3.
See notes \texttt{notes-convergence.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $x_n \le y_n \le z_n$ for all $n$, and $(x_n) \to L$ and $(z_n) \to L$,
then $(y_n) \to L$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  x_n \le y_n \le z_n,\;
  (x_n) \to L,\;
  (z_n) \to L
  \;\Rightarrow\; (y_n) \to L.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Given $\varepsilon > 0$, get $N$ so that for $n \ge N$:
$|x_n - L| < \varepsilon$ and $|z_n - L| < \varepsilon$, i.e.
$L - \varepsilon < x_n$ and $z_n < L + \varepsilon$.
Then $L - \varepsilon < x_n \le y_n \le z_n < L + \varepsilon$, so
$|y_n - L| < \varepsilon$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim Under the given hypotheses, $(y_n) \to L$.

\Given $x_n \le y_n \le z_n$ for all $n$; $(x_n) \to L$; $(z_n) \to L$.

\Goal For any $\varepsilon > 0$, find $N$ with $|y_n - L| < \varepsilon$ for $n \ge N$.

\Strategy The bounds on $x_n$ and $z_n$ together constrain $y_n$ to the
interval $(L - \varepsilon, L + \varepsilon)$.

\medskip

% --- let ε > 0 ---

% --- get N₁ with x_n > L - ε for n ≥ N₁ ---

% --- get N₂ with z_n < L + ε for n ≥ N₂ ---

% --- for n ≥ max(N₁, N₂): L - ε < x_n ≤ y_n ≤ z_n < L + ε ---

% --- hence |y_n - L| < ε ---

\end{proof}

\begin{remark*}[Proof shape]
Direct chain of inequalities: the two outer sequences control the interval
containing $y_n$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of convergence.
  \item Pointwise inequalities $x_n \le y_n \le z_n$.
\end{itemize}
\end{remark*}
