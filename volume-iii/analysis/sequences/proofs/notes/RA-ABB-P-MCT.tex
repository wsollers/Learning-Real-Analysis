\subsubsection[Monotone Convergence Theorem]{Proof --- Monotone Convergence Theorem}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Theorem~2.4.2.
See notes \texttt{notes-monotone-sequences.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
A monotone bounded sequence converges. Specifically:
\begin{itemize}
  \item If $(x_n)$ is increasing and bounded above, then $(x_n) \to \sup\{x_n\}$.
  \item If $(x_n)$ is decreasing and bounded below, then $(x_n) \to \inf\{x_n\}$.
\end{itemize}

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  (x_n) \text{ increasing, bounded above}
  \;\Rightarrow\;
  (x_n) \to \sup_{n} x_n.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Let $L = \sup\{x_n : n \in \mathbb{N}\}$ (exists by Completeness Axiom).
Given $\varepsilon > 0$, by the $\varepsilon$-characterization of $\sup$, there
exists $N$ with $x_N > L - \varepsilon$. Since $(x_n)$ is increasing,
$x_n \ge x_N > L - \varepsilon$ for all $n \ge N$. Also $x_n \le L$.
Hence $|x_n - L| < \varepsilon$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim A monotone bounded sequence converges to its extremum.

\Given $(x_n)$ increasing and bounded above.

\Goal Show $(x_n) \to L := \sup\{x_n\}$.

\Strategy Use the $\varepsilon$-characterization of $\sup$ to find $N$ where
the tail is within $\varepsilon$ of $L$; monotonicity keeps it there.

\medskip

% --- let L = sup{x_n}, exists by Completeness Axiom ---

% --- let ε > 0; by ε-char of sup: ∃ N with x_N > L - ε ---

% --- for n ≥ N: x_n ≥ x_N > L - ε (by monotonicity) ---

% --- also x_n ≤ L (L is upper bound) ---

% --- hence L - ε < x_n ≤ L, so |x_n - L| < ε ---

\end{proof}

\begin{remark*}[Proof shape]
Supremum is the natural limit candidate: monotonicity ensures the sequence
never retreats, and the $\varepsilon$-characterization ensures it gets close.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Completeness Axiom (LUB property).
  \item $\varepsilon$-Characterization of Supremum.
  \item Monotonicity: $m \le n \Rightarrow x_m \le x_n$.
\end{itemize}
\end{remark*}
