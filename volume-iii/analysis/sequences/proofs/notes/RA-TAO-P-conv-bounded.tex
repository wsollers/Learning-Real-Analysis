\subsubsection[Convergent Implies Bounded]{Proof --- Every Convergent Sequence is Bounded}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Theorem~2.3.2; Tao, \emph{Analysis I}.
See notes \texttt{notes-convergence.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $(x_n) \to L$ then $(x_n)$ is bounded.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  (x_n) \to L
  \;\Rightarrow\;
  \exists M > 0,\; \forall n \in \mathbb{N},\; |x_n| \le M.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Use $\varepsilon = 1$: get $N$ such that $|x_n - L| < 1$ for $n \ge N$.
By the reverse triangle inequality, $|x_n| < |L| + 1$ for $n \ge N$.
The finite initial segment $\{x_1, \ldots, x_{N-1}\}$ is bounded by its
maximum. Take $M = \max(|x_1|, \ldots, |x_{N-1}|, |L| + 1)$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $(x_n) \to L \Rightarrow (x_n)$ is bounded.

\Given $(x_n)$ a real sequence, $(x_n) \to L$.

\Goal Find $M > 0$ with $|x_n| \le M$ for all $n$.

\Strategy Apply the definition of convergence with $\varepsilon = 1$; handle
the initial finite segment separately.

\medskip

% --- use ε = 1: get N with |x_n - L| < 1 for n ≥ N ---

% --- deduce |x_n| < |L| + 1 for n ≥ N (reverse triangle ineq) ---

% --- set M = max(|x_1|, ..., |x_{N-1}|, |L| + 1) ---

% --- conclude |x_n| ≤ M for all n ---

\end{proof}

\begin{remark*}[Proof shape]
Split the sequence at the threshold $N$: beyond $N$ the tail is controlled
by convergence; before $N$ a finite maximum handles the finitely many terms.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of convergence.
  \item Reverse triangle inequality: $|x_n| - |L| \le |x_n - L|$.
  \item Every finite set of real numbers has a maximum.
\end{itemize}
\end{remark*}
