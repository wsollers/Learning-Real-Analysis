\subsubsection[Algebra of Limits]{Proof --- Algebra of Limits}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Theorem~2.3.3.
See notes \texttt{notes-algebra-of-sequences.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Let $(x_n) \to x$ and $(y_n) \to y$. Then:
\begin{enumerate}
  \item $(cx_n) \to cx$ for any $c \in \mathbb{R}$;
  \item $(x_n + y_n) \to x + y$;
  \item $(x_n y_n) \to xy$;
  \item $(x_n / y_n) \to x/y$, provided $y \ne 0$.
\end{enumerate}

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  (x_n) \to x,\; (y_n) \to y
  \;\Rightarrow\;
  \text{(limits of sums, products, quotients as above).}
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Each part uses $\varepsilon$-$N$: for sums, split error $\varepsilon/2$ to each sequence.
For products, write $x_n y_n - xy = x_n(y_n - y) + y(x_n - x)$ and bound
each term using boundedness of $(x_n)$. Quotients reduce to showing
$1/y_n \to 1/y$ (which requires $y_n \ne 0$ eventually and a bounded-below
argument).

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim The algebraic limit laws hold for convergent sequences.

\Given $(x_n) \to x$, $(y_n) \to y$.

\Goal Prove each of the four parts.

\Strategy Split $\varepsilon$ appropriately for each algebraic operation.

\medskip

\textbf{Part (1): Scalar multiple.}
% --- |cx_n - cx| = |c||x_n - x| < ε ---

\medskip

\textbf{Part (2): Sum.}
% --- |x_n + y_n - (x+y)| ≤ |x_n - x| + |y_n - y| < ε/2 + ε/2 ---

\medskip

\textbf{Part (3): Product.}
% --- write x_n y_n - xy = x_n(y_n - y) + y(x_n - x) ---
% --- bound |x_n| ≤ M (convergent implies bounded) ---
% --- choose N so both |x_n - x| < ε/(2(|y|+1)) and |y_n - y| < ε/(2M) ---

\medskip

\textbf{Part (4): Quotient.}
% --- first show 1/y_n → 1/y when y ≠ 0 ---
% --- use |1/y_n - 1/y| = |y - y_n|/|y_n||y|; bound 1/|y_n| below ---
% --- then part (4) follows from part (3) ---

\end{proof}

\begin{remark*}[Proof shape]
Each part is a direct $\varepsilon$-$N$ argument; the product uses the
algebraic identity $x_n y_n - xy = x_n(y_n - y) + y(x_n - x)$ to split
into two manageable terms.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of convergence.
  \item Every convergent sequence is bounded (for the product case).
  \item Triangle inequality.
\end{itemize}
\end{remark*}
