\subsubsection[Subsequence of a Subsequence]{Proof --- Subsequence of a Subsequence}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-subsequence-toolkit.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
A subsequence of a subsequence is a subsequence of the original sequence.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \text{If } (a_{n_k}) \text{ is a subseq of } (a_n) \text{ and } (a_{n_{k_j}}) \text{ is a subseq of } (a_{n_k}),\text{ then } (a_{n_{k_j}}) \text{ is a subseq of } (a_n).
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
The composition of two strictly increasing functions $j \mapsto k_j$ and $k \mapsto n_k$ is strictly increasing, so $j \mapsto n_{k_j}$ is a valid index sequence.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim A subsequence of a subsequence is a subsequence of the original sequence.

\Given 

\Goal 

\Strategy The composition of two strictly increasing functions $j \mapsto k_j$ and $k \mapsto n_k$ is strictly increasing, so $j \mapsto n_{k_j}$ is a valid index sequence.

\medskip

% --- j → k_j strictly increasing (subseq of subseq) ---
% --- k → n_k strictly increasing (subseq of original) ---
% --- composition j → n_{k_j} is strictly increasing ---
% --- therefore (a_{n_{k_j}}) is a valid subsequence of (a_n) ---

\end{proof}

\begin{remark*}[Proof shape]
Composition of strictly increasing functions is strictly increasing.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of subsequence.
  \item Composition of strictly increasing functions is strictly increasing.
\end{itemize}
\end{remark*}
