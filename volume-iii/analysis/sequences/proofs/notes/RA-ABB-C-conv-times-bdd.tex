\subsubsection[Convergent Times Bounded Is Bounded]{Proof --- Convergent Times Bounded Is Bounded}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-algebra-of-sequences.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $(x_n)$ converges and $(y_n)$ is bounded, then $(x_n y_n)$ is bounded.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  x_n \to x,\; |y_n| \le M \Rightarrow \exists K,\; |x_n y_n| \le K.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
$(x_n)$ converges so is bounded by some $B$. Then $|x_n y_n| \le BM$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $(x_n)$ converges and $(y_n)$ is bounded, then $(x_n y_n)$ is bounded.

\Given 

\Goal 

\Strategy $(x_n)$ converges so is bounded by some $B$. Then $|x_n y_n| \le BM$.

\medskip

% --- convergent ⇒ bounded: |x_n| ≤ B ---
% --- |x_n y_n| = |x_n||y_n| ≤ B · M ---

\end{proof}

\begin{remark*}[Proof shape]
Product of two bounds: boundedness of $(x_n)$ from convergence, boundedness of $(y_n)$ from hypothesis.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Convergent sequences are bounded.
  \item $|ab| = |a||b|$.
\end{itemize}
\end{remark*}
