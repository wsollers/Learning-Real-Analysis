\subsubsection[Convergence is Inherited by Subsequences]{Proof --- Convergence is Inherited by Subsequences}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-subsequence-toolkit.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $(a_n) \to L$, then every subsequence $(a_{n_k}) \to L$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  a_n \to L \Rightarrow a_{n_k} \to L \text{ for every strictly increasing } (n_k).
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
This is precisely the statement of ``Subsequences Inherit Limits'' (see \texttt{notes-subsequences.tex}), restated here as a toolkit proposition.

\vspace{0.75em}
