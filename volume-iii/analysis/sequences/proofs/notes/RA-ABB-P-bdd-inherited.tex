\subsubsection[Boundedness is Inherited by Subsequences]{Proof --- Boundedness is Inherited by Subsequences}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-subsequence-toolkit.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $(a_n)$ is bounded, then every subsequence $(a_{n_k})$ is bounded by the same bound.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \exists M,\; \forall n,\; |a_n| \le M \Rightarrow \forall (n_k),\; \forall k,\; |a_{n_k}| \le M.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
The bound $M$ applies to all terms $a_n$ without exception; subsequence terms are a subset of all terms.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $(a_n)$ is bounded, then every subsequence $(a_{n_k})$ is bounded by the same bound.

\Given 

\Goal 

\Strategy The bound $M$ applies to all terms $a_n$ without exception; subsequence terms are a subset of all terms.

\medskip

% --- every term a_{n_k} is a term of (a_n), so |a_{n_k}| ≤ M ---

\end{proof}

\begin{remark*}[Proof shape]
Immediate: the bound applies to every term regardless of index.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of bounded sequence.
  \item Subsequence terms are terms of the original sequence.
\end{itemize}
\end{remark*}
