\subsubsection[Fundamental Asymptotic Limits]{Proof --- Fundamental Asymptotic Limits}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-growth-and-asymptotics.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} The following limits hold: $n^{1/n} \to 1$, $c^{1/n} \to 1$ for $c > 0$, $n!/n^n \to 0$, $n^k/c^n \to 0$ for $c > 1$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  n^{1/n} \to 1,\quad c^{1/n} \to 1 \;(c>0),\quad \frac{n!}{n^n} \to 0,\quad \frac{n^k}{c^n} \to 0 \;(c>1).
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Each requires a different technique. $n^{1/n}$: set $n^{1/n} = 1 + \varepsilon_n$, use binomial; $c^{1/n}$: take logs; $n^k/c^n$: Ratio Test; $n!/n^n$: direct estimate.

\vspace{0.75em}

\noindent
\textbf{Proof.}
