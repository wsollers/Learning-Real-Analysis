\subsubsection[Uniqueness of Limits]{Proof --- Uniqueness of Limits}

\bigskip

\noindent
\textbf{Source.}
Tao, \emph{Analysis I}; Abbott, \emph{Understanding Analysis}, Proposition~2.2.7.
See notes \texttt{notes-convergence.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
A convergent sequence has exactly one limit.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  (x_n) \to L \;\wedge\; (x_n) \to L'
  \;\Rightarrow\; L = L'.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Let $\varepsilon > 0$. Both limits give tails within $\varepsilon/2$ of $L$ and $L'$
respectively. For large $n$, the triangle inequality gives
$|L - L'| \le |L - x_n| + |x_n - L'| < \varepsilon$.
Since $\varepsilon > 0$ was arbitrary, $|L - L'| = 0$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $(x_n) \to L$ and $(x_n) \to L'$ then $L = L'$.

\Given $(x_n)$ a real sequence with $(x_n) \to L$ and $(x_n) \to L'$.

\Goal Show $L = L'$.

\Strategy Suppose $L \ne L'$; derive a contradiction using $\varepsilon = |L - L'|/2$.

\medskip

% --- let ε = |L - L'| / 2 > 0 (assuming L ≠ L') ---

% --- get N₁ with |x_n - L| < ε for n ≥ N₁ ---

% --- get N₂ with |x_n - L'| < ε for n ≥ N₂ ---

% --- for n ≥ max(N₁, N₂): |L - L'| ≤ |L - x_n| + |x_n - L'| < 2ε = |L - L'| ---

% --- contradiction ---

\end{proof}

\begin{remark*}[Proof shape]
Classical $\varepsilon/2$ triangle inequality argument. The key move is
that $|L - L'|$ is a fixed positive number that must be strictly less
than itself — an impossibility.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of convergence: $(x_n) \to L$ iff
        $\forall \varepsilon > 0,\, \exists N,\, \forall n \ge N,\, |x_n - L| < \varepsilon$.
  \item Triangle inequality.
\end{itemize}
\end{remark*}
