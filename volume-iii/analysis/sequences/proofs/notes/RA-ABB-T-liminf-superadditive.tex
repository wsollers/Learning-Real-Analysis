\subsubsection[Superadditivity of Liminf]{Proof --- Superadditivity of Liminf}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-limsup-liminf.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} $\liminf(x_n + y_n) \ge \liminf x_n + \liminf y_n$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \liminf(x_n + y_n) \ge \liminf x_n + \liminf y_n.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Apply Sign Symmetry to the Subadditivity of $\limsup$: $\liminf(x_n+y_n) = -\limsup(-(x_n+y_n)) \ge -\limsup(-x_n) - \limsup(-y_n) = \liminf x_n + \liminf y_n$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $\liminf(x_n + y_n) \ge \liminf x_n + \liminf y_n$.

\Given 

\Goal 

\Strategy Apply Sign Symmetry to the Subadditivity of $\limsup$: $\liminf(x_n+y_n) = -\limsup(-(x_n+y_n)) \ge -\limsup(-x_n) - \limsup(-y_n) = \liminf x_n + \liminf y_n$.

\medskip

% --- liminf(x_n+y_n) = -limsup(-(x_n+y_n)) ---
% --- = -limsup(-x_n - y_n) ---
% --- ≥ -limsup(-x_n) - limsup(-y_n)  (by subadditivity of limsup) ---
% --- = liminf x_n + liminf y_n  (by sign symmetry) ---

\end{proof}

\begin{remark*}[Proof shape]
Reduction to Subadditivity of $\limsup$ via Sign Symmetry.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Subadditivity of $\limsup$.
  \item Sign Symmetry of $\limsup$ and $\liminf$.
\end{itemize}
\end{remark*}
