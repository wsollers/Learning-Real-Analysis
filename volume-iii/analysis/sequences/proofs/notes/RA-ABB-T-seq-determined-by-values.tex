\subsubsection[A Sequence Is Determined by Its Values]{Proof --- A Sequence Is Determined by Its Values}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-sequences.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Two sequences are equal if and only if they agree on every term.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  (a_n) = (b_n) \Longleftrightarrow \forall n \in \mathbb{N},\; a_n = b_n.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
A sequence is a function $\mathbb{N} \to \mathbb{R}$; two functions are equal iff they agree at every point.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim Two sequences are equal if and only if they agree on every term.

\Given 

\Goal 

\Strategy A sequence is a function $\mathbb{N} \to \mathbb{R}$; two functions are equal iff they agree at every point.

\medskip

% --- sequences are functions f: N → R ---
% --- function equality: f = g ⟺ f(n) = g(n) for all n ---
% --- direct translation ---

\end{proof}

\begin{remark*}[Proof shape]
Definitional: equality of sequences is equality of functions.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of sequence as a function $\mathbb{N} \to \mathbb{R}$.
  \item Extensionality of functions.
\end{itemize}
\end{remark*}
