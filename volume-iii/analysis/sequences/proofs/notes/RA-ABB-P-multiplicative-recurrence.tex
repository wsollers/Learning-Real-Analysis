\subsubsection[Multiplicative Recurrence Bound]{Proof --- Multiplicative Recurrence Bound}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-recurrence-relations.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $x_{n+1} = c x_n$ (with $|c| < 1$), then $x_n = c^n x_0 \to 0$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  x_{n+1} = c x_n,\; |c| < 1 \Rightarrow x_n = c^n x_0 \to 0.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Unravel the recurrence: $x_n = c^n x_0$ by induction. Since $|c| < 1$, $c^n \to 0$ (geometric sequence), so $x_n \to 0$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $x_{n+1} = c x_n$ (with $|c| < 1$), then $x_n = c^n x_0 \to 0$.

\Given 

\Goal 

\Strategy Unravel the recurrence: $x_n = c^n x_0$ by induction. Since $|c| < 1$, $c^n \to 0$ (geometric sequence), so $x_n \to 0$.

\medskip

\begin{remark*}[Proof shape]
Closed-form solution by induction; geometric decay gives convergence to 0.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Geometric sequence: $|c| < 1 \Rightarrow c^n \to 0$.

\end{itemize}
\end{remark*}
