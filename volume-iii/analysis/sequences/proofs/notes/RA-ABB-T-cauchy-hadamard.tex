\subsubsection[Cauchy--Hadamard Formula]{Proof --- Cauchy--Hadamard Formula}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-power-series.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} The radius of convergence of $\sum a_n x^n$ satisfies $1/R = \limsup_{n\to\infty} |a_n|^{1/n}$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \frac{1}{R} = \limsup_{n\to\infty} |a_n|^{1/n}.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} This is the formula extracted from the Radius of Convergence Theorem: $R = 1/(\limsup |a_n|^{1/n})$, with the conventions $1/0 = \infty$ and $1/\infty = 0$.

\vspace{0.75em}

\noindent
