\subsubsection[Order Limit Theorem]{Proof --- Order Limit Theorem}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Theorem~2.3.4.
See notes \texttt{notes-convergence.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Let $(x_n) \to x$ and $(y_n) \to y$ in $\mathbb{R}$. If $x_n \le y_n$ for
all $n$, then $x \le y$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  (x_n) \to x,\; (y_n) \to y,\; \forall n,\; x_n \le y_n
  \;\Rightarrow\; x \le y.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Suppose for contradiction $x > y$. Set $\varepsilon = (x - y)/2 > 0$.
For large $n$, $x_n > x - \varepsilon = (x + y)/2$ and $y_n < y + \varepsilon = (x + y)/2$,
giving $x_n > y_n$ — contradicting $x_n \le y_n$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $(x_n) \to x$, $(y_n) \to y$, $x_n \le y_n$ for all $n$ $\Rightarrow$ $x \le y$.

\Given $(x_n) \to x$, $(y_n) \to y$, $x_n \le y_n$ for all $n \in \mathbb{N}$.

\Goal Show $x \le y$.

\Strategy Proof by contradiction: assume $x > y$ and derive $x_n > y_n$ for
large $n$.

\medskip

% --- assume x > y; set ε = (x - y)/2 > 0 ---

% --- get N₁ with |x_n - x| < ε for n ≥ N₁: so x_n > x - ε = (x+y)/2 ---

% --- get N₂ with |y_n - y| < ε for n ≥ N₂: so y_n < y + ε = (x+y)/2 ---

% --- for n ≥ max(N₁, N₂): x_n > (x+y)/2 > y_n — contradicts x_n ≤ y_n ---

\end{proof}

\begin{remark*}[Proof shape]
Proof by contradiction with $\varepsilon = (x - y)/2$: the two tails must
eventually straddle the midpoint, violating the pointwise ordering.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of convergence.
  \item Assumption $x_n \le y_n$ for all $n$.
\end{itemize}
\end{remark*}
