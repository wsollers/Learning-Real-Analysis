\subsubsection[Basic Inequalities for Limsup and Liminf]{Proof --- Basic Inequalities for Limsup and Liminf}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-limsup-liminf.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} $\liminf x_n \le \limsup x_n$; if $(x_n)$ is bounded then both are finite.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \liminf x_n \le \limsup x_n.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Each tail infimum $i_k \le$ tail supremum $s_k$; take limits.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $\liminf x_n \le \limsup x_n$; if $(x_n)$ is bounded then both are finite.
