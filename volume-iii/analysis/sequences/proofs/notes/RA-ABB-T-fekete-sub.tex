\subsubsection[Fekete's Lemma (Subadditive Sequences)]{Proof --- Fekete's Lemma (Subadditive Sequences)}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-recurrence-relations.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $a_{m+n} \le a_m + a_n$ for all $m, n \ge 1$ (subadditive), then $\lim_{n\to\infty} a_n/n = \inf_{n\ge 1} a_n/n$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  a_{m+n} \le a_m + a_n \Rightarrow \lim_{n\to\infty} \frac{a_n}{n} = \inf_{n\ge 1} \frac{a_n}{n}.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Fix $m$, write $n = qm + r$ (division algorithm), bound $a_n/n \le a_m/m + a_r/(qm+r)$. Let $n \to \infty$: $q \to \infty$ while $r$ stays bounded, giving $\limsup a_n/n \le a_m/m$. Take inf over $m$.

\vspace{0.75em}

\noindent

\Given

\begin{remark*}[Proof shape]
Fix $m$; Euclidean division expresses large $n$ in terms of $m$; subadditivity bounds $a_n$ by $q a_m + a_r$; pass to $\limsup$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Division algorithm for integers.
  \item Subadditivity: $a_{m+n} \le a_m + a_n$.
  \item $\limsup a_n/n$ and $\liminf a_n/n$ exist for bounded-below sequences.
\end{itemize}
\end{remark*}
