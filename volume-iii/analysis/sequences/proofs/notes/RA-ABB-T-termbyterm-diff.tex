\subsubsection[Term-by-Term Differentiation of Power Series]{Proof --- Term-by-Term Differentiation of Power Series}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-power-series.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $f(x) = \sum_{n=0}^\infty a_n x^n$ for $|x| < R$, then $f'(x) = \sum_{n=1}^\infty n a_n x^{n-1}$, and this series has the same radius of convergence $R$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  f(x) = \sum a_n x^n \Rightarrow f'(x) = \sum_{n=1}^\infty n a_n x^{n-1} \text{ on } |x| < R.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Show the differentiated series has radius $R$ (since $\limsup |n a_n|^{1/n} = \limsup |a_n|^{1/n}$ as $n^{1/n} \to 1$). Then show the derivative of the partial sum limit equals the limit of the derivatives (requires uniform convergence on compact subsets).

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $f(x) = \sum_{n=0}^\infty a_n x^n$ for $|x| < R$, then $f'(x) = \sum_{n=1}^\infty n a_n x^{n-1}$, and this series has the same radius of convergence $R$.

\Given

\begin{remark*}[Proof shape]
Two-part argument: (1) Cauchy-Hadamard shows same radius; (2) uniform convergence justifies swapping limit and derivative.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Cauchy--Hadamard Formula.
  \item $n^{1/n} \to 1$ (Fundamental Asymptotic Limits).
  \item Weierstrass M-test for uniform convergence.
  \item Uniform convergence allows term-by-term differentiation.
\end{itemize}
\end{remark*}
