\subsubsection[Sign Symmetry of Limsup and Liminf]{Proof --- Sign Symmetry of Limsup and Liminf}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-limsup-liminf.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} $\limsup(-x_n) = -\liminf x_n$ and $\liminf(-x_n) = -\limsup x_n$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \limsup(-x_n) = -\liminf x_n.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} $\sup(-T_k) = -\inf(T_k)$ pointwise; take limits.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $\limsup(-x_n) = -\liminf x_n$ and $\liminf(-x_n) = -\limsup x_n$.

\Given 

\Goal 

\Strategy $\sup(-T_k) = -\inf(T_k)$ pointwise; take limits.

\medskip

% --- sup{-x : x ∈ T_k} = -inf{x : x ∈ T_k} ---
% --- so s_k(-x) = -i_k(x) for each k ---
% --- take limits: limsup(-x_n) = lim(-i_k(x)) = -liminf x_n ---

\end{proof}

\begin{remark*}[Proof shape]
Negation swaps sup and inf at each tail level; the limit inherits this symmetry.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item $\sup(-A) = -\inf(A)$ for any bounded set $A$.
  \item Limit of $(-i_k) = -\lim i_k$.
\end{itemize}
\end{remark*}
