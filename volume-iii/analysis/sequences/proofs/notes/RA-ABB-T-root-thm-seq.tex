\subsubsection[Root Theorem for Sequences]{Proof --- Root Theorem for Sequences}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-growth-and-asymptotics.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $a_n > 0$ and $a_{n+1}/a_n \to L < 1$, then $a_n \to 0$; if $L > 1$ then $a_n \to \infty$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  a_{n+1}/a_n \to L < 1 \Rightarrow a_n \to 0.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Pick $r$ with $L < r < 1$. Eventually $a_{n+1} < r a_n$, so $a_n < a_N r^{n-N} \to 0$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $a_n > 0$ and $a_{n+1}/a_n \to L < 1$, then $a_n \to 0$; if $L > 1$ then $a_n \to \infty$.

\Given 

\Goal 

\Strategy Pick $r$ with $L < r < 1$. Eventually $a_{n+1} < r a_n$, so $a_n < a_N r^{n-N} \to 0$.

\medskip

% --- L < 1: pick r with L < r < 1 ---
% --- get N: for n ≥ N, a_{n+1}/a_n < r ---
% --- a_n ≤ a_N r^{n-N} → 0 ---

\end{proof}

\begin{remark*}[Proof shape]
Same geometric domination argument as in the Ratio Test: once ratios are below $r < 1$, terms decay geometrically.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Iterative Propagation Lemma (for the geometric decay).
  \item Geometric sequence: $r^n \to 0$ for $0 \le r < 1$.
\end{itemize}
\end{remark*}
