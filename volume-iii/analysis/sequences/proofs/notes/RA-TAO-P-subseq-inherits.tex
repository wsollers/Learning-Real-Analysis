\subsubsection[Subsequences Inherit Limits]{Proof --- Subsequences Inherit Limits}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Theorem~2.5.2; Tao, \emph{Analysis I}.
See notes \texttt{notes-subsequences.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $(x_n) \to L$ and $(x_{n_k})$ is any subsequence, then
$(x_{n_k}) \to L$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  (x_n) \to L
  \;\wedge\;
  n_1 < n_2 < n_3 < \cdots
  \;\Rightarrow\;
  (x_{n_k}) \to L.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Given $\varepsilon > 0$, get $N$ so that $|x_n - L| < \varepsilon$ for $n \ge N$.
Key fact: the index function satisfies $n_k \ge k$ (by induction: $n_1 \ge 1$
and strictly increasing). So for $k \ge N$, $n_k \ge k \ge N$, giving
$|x_{n_k} - L| < \varepsilon$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $(x_n) \to L \Rightarrow (x_{n_k}) \to L$ for any subsequence.

\Given $(x_n) \to L$; strictly increasing $n : \mathbb{N} \to \mathbb{N}$,
$k \mapsto n_k$.

\Goal For any $\varepsilon > 0$, find $K$ with $|x_{n_k} - L| < \varepsilon$ for
$k \ge K$.

\Strategy Use the index growth lemma $n_k \ge k$; then the threshold $N$
for $(x_n)$ serves as a threshold $K = N$ for the subsequence.

\medskip

% --- cite Index Growth: n_k ≥ k (by induction) ---

% --- given ε > 0, get N with |x_n - L| < ε for n ≥ N ---

% --- for k ≥ N: n_k ≥ k ≥ N, so |x_{n_k} - L| < ε ---

\end{proof}

\begin{remark*}[Proof shape]
The index growth lemma $n_k \ge k$ bridges the subsequence index $k$ to
the main sequence index $n_k$; convergence of the main sequence then
applies directly.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of convergence and subsequence.
  \item Index Growth: $n_k \ge k$ for any strictly increasing
        $\sigma : \mathbb{N} \to \mathbb{N}$ (proved by induction).
\end{itemize}
\end{remark*}
