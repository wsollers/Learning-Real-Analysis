\subsubsection[Integral Test]{Proof --- Integral Test}

\bigskip

\noindent
\textbf{Source.} Abbott, \emph{Understanding Analysis}; see \texttt{notes-series-tests.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $f$ is positive, continuous, and decreasing on $[1,\infty)$ and $a_n = f(n)$, then $\sum a_n < \infty \Leftrightarrow \int_1^\infty f < \infty$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \sum_{n=1}^\infty f(n) < \infty \Longleftrightarrow \int_1^\infty f(x)\,dx < \infty.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Compare using the staircase estimate: $f(n+1) \le \int_n^{n+1} f(x)\,dx \le f(n)$ (since $f$ is decreasing). Sum over $n$ and compare the series with the integral.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $f$ is positive, continuous, and decreasing on $[1,\infty)$ and $a_n = f(n)$, then $\sum a_n < \infty \Leftrightarrow \int_1^\infty f < \infty$.

\Given 

\Goal 

\Strategy Compare using the staircase estimate: $f(n+1) \le \int_n^{n+1} f(x)\,dx \le f(n)$ (since $f$ is decreasing). Sum over $n$ and compare the series with the integral.

\medskip

% --- f decreasing: f(n+1) ≤ ∫_n^{n+1} f(x)dx ≤ f(n) ---
% --- sum from n=1 to N: ∑_{n=2}^{N+1} f(n) ≤ ∫_1^{N+1} f ≤ ∑_{n=1}^N f(n) ---
% --- (⇒) ∑f(n) < ∞ ⇒ ∫ f bounded above ⇒ ∫_1^∞ f < ∞ (since f ≥ 0) ---
% --- (⇐) ∫ f < ∞ ⇒ partial sums of ∑f(n+1) are bounded ⇒ ∑f(n) < ∞ ---

\end{proof}

\begin{remark*}[Proof shape]
Staircase comparison: monotonicity of $f$ lets us sandwich the integral between left and right Riemann sums.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item $f$ positive, continuous, decreasing on $[1,\infty)$.
  \item Monotone Convergence Theorem (for nonneg series).
  \item Left and right Riemann sum bounds from monotonicity.
\end{itemize}
\end{remark*}
