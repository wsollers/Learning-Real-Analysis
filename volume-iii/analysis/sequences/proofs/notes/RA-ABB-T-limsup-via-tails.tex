\subsubsection[Limit Superior via Tails]{Proof --- Limit Superior via Tails}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-k-tails.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $(x_n)$ is bounded and $s_k = \sup T_k$, then $\limsup x_n = \lim_{k \to \infty} s_k$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \limsup x_n = \lim_{k \to \infty} \sup\{x_n : n \ge k\}.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
The sequence $(s_k)$ is decreasing (Tail Suprema are Monotone) and bounded below, so it converges by MCT. The limit is defined to be $\limsup x_n$.

\vspace{0.75em}

\noindent
\textbf{Proof.}

\begin{remark*}[Proof shape]
$(s_k)$ is a bounded decreasing sequence; MCT guarantees its limit, which is $\limsup x_n$ by definition.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Tail Suprema are Monotone.
  \item Monotone Convergence Theorem (MCT).
  \item $(x_n)$ bounded $\Rightarrow$ each $s_k$ is a finite real number.
\end{itemize}
\end{remark*}
