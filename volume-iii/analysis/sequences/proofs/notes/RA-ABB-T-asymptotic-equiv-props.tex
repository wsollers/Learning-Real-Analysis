\subsubsection[Properties of Asymptotic Equivalence]{Proof --- Properties of Asymptotic Equivalence}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-growth-and-asymptotics.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} Asymptotic equivalence $a_n \sim b_n$ (meaning $a_n/b_n \to 1$) is an equivalence relation; if $a_n \sim b_n$ then $\sum a_n$ and $\sum b_n$ have the same convergence behavior.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  a_n \sim b_n \Rightarrow (\sum a_n < \infty \Longleftrightarrow \sum b_n < \infty).
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Reflexivity/symmetry/transitivity follow from ratio arithmetic. Equivalence for series follows from the Limit Comparison Test (since $a_n/b_n \to 1 \in (0,\infty)$).

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim Asymptotic equivalence $a_n \sim b_n$ (meaning $a_n/b_n \to 1$) is an equivalence relation; if $a_n \sim b_n$ then $\sum a_n$ and $\sum b_n$ have the same convergence behavior.

\Given 

\Goal 

\Strategy Reflexivity/symmetry/transitivity follow from ratio arithmetic. Equivalence for series follows from the Limit Comparison Test (since $a_n/b_n \to 1 \in (0,\infty)$).

\medskip

% --- reflexivity: a_n/a_n = 1 → 1 ✓ ---
% --- symmetry: a_n/b_n → 1 ⇒ b_n/a_n = 1/(a_n/b_n) → 1 ✓ ---
% --- transitivity: a_n/b_n → 1 and b_n/c_n → 1 ⇒ a_n/c_n = (a_n/b_n)(b_n/c_n) → 1 ✓ ---
% --- series: a_n/b_n → 1 ∈ (0,∞) ⇒ apply Limit Comparison Test ---

\end{proof}

\begin{remark*}[Proof shape]
Equivalence relation: routine ratio arithmetic. Series equivalence: Limit Comparison Test.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Algebra of limits (ratios).
  \item Limit Comparison Test.
\end{itemize}
\end{remark*}
