\subsubsection[Index Growth Lemma]{Proof --- Index Growth Lemma}

\bigskip

\noindent
\textbf{Source.} Abbott, \emph{Understanding Analysis}; see \texttt{notes-subsequences.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $(n_k)_{k=1}^\infty$ is a strictly increasing sequence of natural numbers, then $n_k \ge k$ for all $k \ge 1$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  n_k \text{ strictly increasing in } \mathbb{N} \Rightarrow n_k \ge k \text{ for all } k.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Induction on $k$. Base: $n_1 \ge 1$. Inductive step: $n_{k+1} > n_k \ge k \Rightarrow n_{k+1} \ge k+1$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $(n_k)_{k=1}^\infty$ is a strictly increasing sequence of natural numbers, then $n_k \ge k$ for all $k \ge 1$.

\Given 

\Goal 

\Strategy Induction on $k$. Base: $n_1 \ge 1$. Inductive step: $n_{k+1} > n_k \ge k \Rightarrow n_{k+1} \ge k+1$.

\medskip

% --- base k=1: n_1 ∈ N so n_1 ≥ 1 = k ---
% --- inductive step: assume n_k ≥ k ---
% --- n_{k+1} > n_k (strictly increasing) and n_{k+1} ∈ N ---
% --- so n_{k+1} ≥ n_k + 1 ≥ k + 1 ---
% --- conclusion: n_k ≥ k for all k ---

\end{proof}

\begin{remark*}[Proof shape]
Induction: each step the index grows by at least 1.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of strictly increasing sequence of natural numbers.
  \item Well-ordering of $\mathbb{N}$: if $m > n$ in $\mathbb{N}$ then $m \ge n+1$.
\end{itemize}
\end{remark*}
