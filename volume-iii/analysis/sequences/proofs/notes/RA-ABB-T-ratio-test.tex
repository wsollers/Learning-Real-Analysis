\subsubsection[Ratio Test]{Proof --- Ratio Test}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}; see \texttt{notes-series-tests.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $L = \limsup |a_{n+1}/a_n|$: $L < 1 \Rightarrow \sum a_n$ converges absolutely; $L > 1 \Rightarrow$ diverges.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  L = \limsup |a_{n+1}/a_n| < 1 \Rightarrow \sum |a_n| < \infty.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
If $L < 1$, pick $r$ with $L < r < 1$. Eventually $|a_{n+1}| < r|a_n|$, giving $|a_n| \le Cr^n$. Comparison with geometric series gives convergence.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $L = \limsup |a_{n+1}/a_n|$: $L < 1 \Rightarrow \sum a_n$ converges absolutely; $L > 1 \Rightarrow$ diverges.

\begin{remark*}[Proof shape]
Domination by a geometric series: once the ratio is below $r < 1$, terms decay geometrically.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Geometric series: $\sum r^n < \infty$ for $|r| < 1$.
  \item Direct Comparison Test.
  \item $\limsup$ characterization (eventually bounded below $r$).
\end{itemize}
\end{remark*}
