\subsubsection[Convergence of Absolute Values]{Proof --- Convergence of Absolute Values}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-algebra-of-sequences.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $x_n \to x$ then $|x_n| \to |x|$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  x_n \to x \Rightarrow |x_n| \to |x|.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Use the reverse triangle inequality: $\bigl||x_n| - |x|\bigr| \le |x_n - x|$. Given $\varepsilon > 0$, the $\varepsilon$-$N$ condition on $(x_n)$ gives the same condition on $(|x_n|)$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $x_n \to x$ then $|x_n| \to |x|$.

\Given 

\Goal 

\Strategy Use the reverse triangle inequality: $\bigl||x_n| - |x|\bigr| \le |x_n - x|$. Given $\varepsilon > 0$, the $\varepsilon$-$N$ condition on $(x_n)$ gives the same condition on $(|x_n|)$.

\medskip

% --- reverse triangle inequality: ||x_n| - |x|| ≤ |x_n - x| ---
% --- given ε > 0 get N with |x_n - x| < ε for n ≥ N ---
% --- then ||x_n| - |x|| ≤ |x_n - x| < ε for n ≥ N ---

\end{proof}

\begin{remark*}[Proof shape]
The reverse triangle inequality transfers the $\varepsilon$-$N$ bound on $(x_n - x)$ directly to $(|x_n| - |x|)$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Reverse triangle inequality: $\bigl||a| - |b|\bigr| \le |a - b|$.
  \item $\varepsilon$-$N$ definition of convergence.
\end{itemize}
\end{remark*}
