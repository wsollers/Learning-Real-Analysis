\subsubsection[Monotonicity is Inherited by Subsequences]{Proof --- Monotonicity is Inherited by Subsequences}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-subsequence-toolkit.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $(a_n)$ is monotone increasing, then every subsequence $(a_{n_k})$ is monotone increasing.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  (a_n) \text{ increasing} \Rightarrow (a_{n_k}) \text{ increasing for every strictly increasing } (n_k).
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Since $(n_k)$ is strictly increasing, $k < j \Rightarrow n_k < n_j$. Since $(a_n)$ is increasing, $n_k < n_j \Rightarrow a_{n_k} \le a_{n_j}$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $(a_n)$ is monotone increasing, then every subsequence $(a_{n_k})$ is monotone increasing.

\Given 

\Goal 

\Strategy Since $(n_k)$ is strictly increasing, $k < j \Rightarrow n_k < n_j$. Since $(a_n)$ is increasing, $n_k < n_j \Rightarrow a_{n_k} \le a_{n_j}$.

\medskip

% --- k < j ⇒ n_k < n_j (strictly increasing index sequence) ---
% --- n_k < n_j ⇒ a_{n_k} ≤ a_{n_j} (monotone increasing original) ---
% --- therefore k < j ⇒ a_{n_k} ≤ a_{n_j}: the subsequence is increasing ---

\end{proof}

\begin{remark*}[Proof shape]
Chain: strict index growth $+$ monotone original $\Rightarrow$ monotone subsequence.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of strictly increasing sequence of indices.
  \item Definition of monotone increasing sequence.
\end{itemize}
\end{remark*}
