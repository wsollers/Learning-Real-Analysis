\subsubsection[Every Subsequence Is a Sequence]{Proof --- Every Subsequence Is a Sequence}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-sequences.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $(a_n)$ is a sequence and $(n_k)$ is strictly increasing, then $(a_{n_k})$ is a sequence.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  (n_k) \text{ strictly increasing in } \mathbb{N} \Rightarrow k \mapsto a_{n_k} \text{ is a function } \mathbb{N} \to \mathbb{R}.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
The map $k \mapsto a_{n_k}$ is a well-defined function $\mathbb{N} \to \mathbb{R}$, hence a sequence by definition.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $(a_n)$ is a sequence and $(n_k)$ is strictly increasing, then $(a_{n_k})$ is a sequence.

\Given 

\Goal 

\Strategy The map $k \mapsto a_{n_k}$ is a well-defined function $\mathbb{N} \to \mathbb{R}$, hence a sequence by definition.

\medskip

% --- k ↦ n_k is a function N → N ---
% --- n ↦ a_n is a function N → R ---
% --- composition k ↦ a_{n_k} is a function N → R ---
% --- therefore (a_{n_k}) is a sequence ---

\end{proof}

\begin{remark*}[Proof shape]
Composition of well-defined functions is well-defined.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of sequence.
  \item Definition of subsequence.
  \item Composition of functions.
\end{itemize}
\end{remark*}
