\subsubsection[Subsequences Are Iterated Tails]{Proof --- Subsequences Are Iterated Tails}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-k-tails.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Every subsequence $(x_{n_k})$ is obtained by passing through progressively later tails of $(x_n)$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \forall k,\; x_{n_k} \in T_{n_k} \subseteq T_k.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Since $n_k \ge k$ (Index Growth Lemma), the $k$th subsequence term $x_{n_k}$ lies in the $k$-th tail $T_k$ (or later).

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim Every subsequence $(x_{n_k})$ is obtained by passing through progressively later tails of $(x_n)$.

\Given 

\Goal 

\Strategy Since $n_k \ge k$ (Index Growth Lemma), the $k$th subsequence term $x_{n_k}$ lies in the $k$-th tail $T_k$ (or later).

\medskip

% --- by Index Growth: n_k ≥ k for all k ---
% --- x_{n_k} ∈ {x_n : n ≥ n_k} = T_{n_k} ⊆ T_k ---

\end{proof}

\begin{remark*}[Proof shape]
A consequence of the Index Growth Lemma: subsequence indices grow at least as fast as their own position.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Index Growth Lemma: if $(n_k)$ strictly increasing in $\mathbb{N}$ then $n_k \ge k$.
\end{itemize}
\end{remark*}
