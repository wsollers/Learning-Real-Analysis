\subsubsection[Diagonal Subsequence Lemma]{Proof --- Diagonal Subsequence Lemma}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-subsequence-toolkit.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Given sequences $(a_n^{(m)}) \to L_m$ and $L_m \to L$, there exists a diagonal subsequence $(a_{n_m}^{(m)}) \to L$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \exists \text{ strictly increasing } (n_m) \text{ s.t.\ } a_{n_m}^{(m)} \to L.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
At step $m$, choose $n_m$ large enough (and $> n_{m-1}$) so that $|a_{n_m}^{(m)} - L_m| < 1/m$ and $|L_m - L| < 1/m$; then $|a_{n_m}^{(m)} - L| < 2/m$ by triangle inequality.

\vspace{0.75em}

\noindent
\textbf{Proof.}

\Claim Given sequences $(a_n^{(m)}) \to L_m$ and $L_m \to L$, there exists a diagonal subsequence $(a_{n_m}^{(m)}) \to L$.

\begin{remark*}[Proof shape]
Diagonal argument: use $1/m$ bounds at each step and the triangle inequality to bound total error by $2/m$.
\end{remark*}
