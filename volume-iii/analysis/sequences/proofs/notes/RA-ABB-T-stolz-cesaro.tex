\subsubsection[Stolz--Ces{\`a}ro Theorem]{Proof --- Stolz--Ces{\`a}ro Theorem}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-growth-and-asymptotics.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $(b_n)$ is strictly increasing and unbounded, and $\lim (a_{n+1} - a_n)/(b_{n+1} - b_n) = L$, then $\lim a_n/b_n = L$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \lim_{n\to\infty} \frac{a_{n+1}-a_n}{b_{n+1}-b_n} = L \Rightarrow \lim_{n\to\infty} \frac{a_n}{b_n} = L.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} The discrete analogue of L'Hôpital's rule. Fix $\varepsilon$; for large $n$ the ratio of differences is within $\varepsilon$ of $L$; telescope $a_n$ and $b_n$ in terms of these differences and estimate.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $(b_n)$ is strictly increasing and unbounded, and $\lim (a_{n+1} - a_n)/(b_{n+1} - b_n) = L$, then $\lim a_n/b_n = L$.

\Given 

\Goal 

\Strategy The discrete analogue of L'Hôpital's rule. Fix $\varepsilon$; for large $n$ the ratio of differences is within $\varepsilon$ of $L$; telescope $a_n$ and $b_n$ in terms of these differences and estimate.

\medskip

% --- given ε, get N: |(a_{k+1}-a_k)/(b_{k+1}-b_k) - L| < ε for k ≥ N ---
% --- write a_n - a_N = ∑_{k=N}^{n-1} (a_{k+1}-a_k), similarly b_n - b_N ---
% --- bound |(a_n - a_N)/(b_n - b_N) - L| < ε ---
% --- use b_n → ∞ to show a_N/(b_n) → 0 and b_N/(b_n) → 0 ---
% --- combine: |a_n/b_n - L| → 0 ---

\end{proof}

\begin{remark*}[Proof shape]
Telescope both sequences; the difference-ratio bound then translates into a ratio bound once unboundedness of $(b_n)$ washes out the finite initial segment.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item $(b_n)$ strictly increasing and unbounded.
  \item Algebra of limits.
  \item Telescoping sums.
\end{itemize}
\end{remark*}
