\subsubsection[Scalar Multiplication for Limsup]{Proof --- Scalar Multiplication for Limsup}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-limsup-liminf.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} For $c > 0$: $\limsup(c x_n) = c \limsup x_n$. For $c < 0$: $\limsup(c x_n) = c \liminf x_n$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  c > 0 \Rightarrow \limsup(cx_n) = c\limsup x_n.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} $\sup(cA) = c \sup A$ for $c > 0$ (since multiplying by $c > 0$ preserves order). Apply to each tail and pass to the limit.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim For $c > 0$: $\limsup(c x_n) = c \limsup x_n$. For $c < 0$: $\limsup(c x_n) = c \liminf x_n$.

\Given 

\Goal 

\Strategy $\sup(cA) = c \sup A$ for $c > 0$ (since multiplying by $c > 0$ preserves order). Apply to each tail and pass to the limit.

\medskip

% --- c > 0: sup{cx_k : k≥n} = c·sup{x_k : k≥n} = c·s_n ---
% --- lim(c·s_n) = c·lim s_n = c·limsup x_n ---
% --- c < 0: sup{cx_k : k≥n} = c·inf{x_k : k≥n} (order reversal) ---
% --- = c·i_n; take limit ---

\end{proof}

\begin{remark*}[Proof shape]
Scaling a set's sup scales by $c$ (for $c>0$); negation swaps sup and inf.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item $\sup(cA) = c \sup A$ for $c > 0$.
  \item Sign Symmetry of $\limsup$ and $\liminf$ (for $c < 0$).
\end{itemize}
\end{remark*}
