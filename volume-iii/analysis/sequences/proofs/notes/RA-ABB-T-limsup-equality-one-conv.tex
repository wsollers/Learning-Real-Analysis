\subsubsection[Equality When One Sequence Converges]{Proof --- Equality When One Sequence Converges}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-limsup-liminf.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $y_n \to y$, then $\limsup(x_n + y_n) = \limsup x_n + y$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  y_n \to y \Rightarrow \limsup(x_n + y_n) = \limsup x_n + y.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} By subadditivity: $\limsup(x_n + y_n) \le \limsup x_n + \limsup y_n = \limsup x_n + y$. Apply same to $x_n = (x_n+y_n) - y_n$ for the reverse inequality.

\vspace{0.75em}

\noindent

\begin{remark*}[Proof shape]
Two applications of subadditivity give both inequalities.
\end{remark*}
