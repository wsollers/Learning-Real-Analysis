\subsubsection[Tails of a Convergent Sequence Converge]{Proof --- Tails of a Convergent Sequence Converge}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-k-tails.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $(x_n) \to L$, then every $k$-tail converges to $L$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  x_n \to L \Rightarrow \forall k \in \mathbb{N},\; (x_n)_{n \ge k} \to L.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
This is the forward direction of ``Convergence is a Tail Property'' restated as a corollary.

\vspace{0.75em}
