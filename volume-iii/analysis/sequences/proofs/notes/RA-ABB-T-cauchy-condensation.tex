\subsubsection[Cauchy Condensation Test]{Proof --- Cauchy Condensation Test}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-series.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $f(n)$ is positive and decreasing, then $\sum f(n)$ converges iff $\sum 2^n f(2^n)$ converges.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \sum_{n=1}^\infty f(n) < \infty \Longleftrightarrow \sum_{n=0}^\infty 2^n f(2^n) < \infty.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Group terms in blocks of length $2^n$: in each block all terms are within a factor of 2 of the block's first/last term, allowing comparison between the original series and the condensed series.

\vspace{0.75em}
