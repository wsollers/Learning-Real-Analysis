\subsubsection[Limit Comparison Test]{Proof --- Limit Comparison Test}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}; see \texttt{notes-series-tests.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $a_n, b_n > 0$ and $\lim a_n/b_n = L$ with $0 < L < \infty$, then $\sum a_n$ and $\sum b_n$ either both converge or both diverge.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  0 < L = \lim (a_n/b_n) < \infty \Rightarrow (\sum a_n < \infty \Longleftrightarrow \sum b_n < \infty).
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
From $a_n/b_n \to L$, eventually $L/2 < a_n/b_n < 2L$, i.e.\ $(L/2)b_n < a_n < 2L b_n$. Apply Direct Comparison in both directions.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $a_n, b_n > 0$ and $\lim a_n/b_n = L$ with $0 < L < \infty$, then $\sum a_n$ and $\sum b_n$ either both converge or both diverge.

\Given 

\Goal 

\Strategy From $a_n/b_n \to L$, eventually $L/2 < a_n/b_n < 2L$, i.e.\ $(L/2)b_n < a_n < 2L b_n$. Apply Direct Comparison in both directions.

\medskip

% --- get N with L/2 < a_n/b_n < 2L for n ≥ N ---
% --- (L/2) b_n < a_n < 2L b_n for n ≥ N ---
% --- (⇒) if ∑b_n < ∞: ∑a_n < 2L ∑b_n < ∞ (direct comparison) ---
% --- (⇐) if ∑a_n < ∞: b_n < (2/L)a_n, so ∑b_n < ∞ ---

\end{proof}

\begin{remark*}[Proof shape]
The ratio being bounded away from 0 and $\infty$ allows two-sided comparison.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Direct Comparison Test.
  \item Algebra of limits: ratio converges $\Rightarrow$ eventually bounded above and below.
\end{itemize}
\end{remark*}
