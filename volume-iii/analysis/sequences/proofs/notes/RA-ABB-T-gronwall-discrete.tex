\subsubsection[Discrete Grönwall Inequality]{Proof --- Discrete Grönwall Inequality}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-recurrence-relations.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $u_n \le \alpha + \sum_{k=0}^{n-1} \beta_k u_k$ with $\alpha \ge 0$ and $\beta_k \ge 0$, then $u_n \le \alpha \prod_{k=0}^{n-1}(1 + \beta_k)$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  u_n \le \alpha \prod_{k=0}^{n-1}(1+\beta_k).
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Induction: at each step, use the bound on $u_k$ to control the sum. The product form accumulates the $\beta_k$ factors.

\vspace{0.75em}

\noindent
