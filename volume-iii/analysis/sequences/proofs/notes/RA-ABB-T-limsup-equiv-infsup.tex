\subsubsection[Equivalent Formulation of Limsup as Inf-Sup]{Proof --- Equivalent Formulation of Limsup as Inf-Sup}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-limsup-liminf.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
$\limsup x_n = \inf_{n \ge 1} \sup_{k \ge n} x_k$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \limsup x_n = \inf_{n \ge 1} s_n \text{ where } s_n = \sup_{k \ge n} x_k.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
$(s_n)$ is decreasing (Monotonicity), so its infimum equals its limit (MCT applied to the decreasing sequence).

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $\limsup x_n = \inf_{n \ge 1} \sup_{k \ge n} x_k$.

\Given 

\Goal 

\Strategy $(s_n)$ is decreasing (Monotonicity), so its infimum equals its limit (MCT applied to the decreasing sequence).

\medskip

% --- (s_n) decreasing and bounded below ---
% --- MCT: lim s_n = inf{s_n : n ≥ 1} ---
% --- limsup x_n := lim s_n = inf_{n≥1} s_n ---

\end{proof}

\begin{remark*}[Proof shape]
Decreasing bounded sequence: limit $=$ infimum. This turns the limit definition of $\limsup$ into the inf-sup expression.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Monotonicity of Tail Suprema.
  \item Monotone Convergence Theorem.
  \item For decreasing bounded sequence: $\lim s_n = \inf_n s_n$.
\end{itemize}
\end{remark*}
