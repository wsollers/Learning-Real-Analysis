\subsubsection[Reflection via Universal Properties]{Proof --- Reflection via Universal Properties}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-subsequence-toolkit.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $\mathcal{P}$ is a universal subsequence property then $\mathcal{P}$ is inherited and reflected by the collection of all subsequences.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \mathcal{P} \text{ universal} \Rightarrow (a_n) \text{ satisfies } \mathcal{P} \Leftrightarrow \forall (a_{n_k}),\; (a_{n_k}) \text{ satisfies } \mathcal{P}.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
(⇒) $\mathcal{P}$ universal $\Rightarrow$ every subsequence satisfies it (inheritance). (⇐) $(a_n)$ is a subsequence of itself, so if every subsequence satisfies $\mathcal{P}$ then so does $(a_n)$.

\vspace{0.75em}

\noindent
\textbf{Proof.}

\Claim If $\mathcal{P}$ is a universal subsequence property then $\mathcal{P}$ is inherited and reflected by the collection of all subsequences.

\begin{remark*}[Proof shape]
$(a_n)$ is its own subsequence (identity index), so the universal quantifier over subsequences includes the sequence itself.
\end{remark*}
