\subsubsection[Convergent Implies Cauchy]{Proof --- Every Convergent Sequence is Cauchy}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}, Theorem~2.6.2; Tao, \emph{Analysis I}.
See notes \texttt{notes-cauchy.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $(x_n)$ converges, then $(x_n)$ is a Cauchy sequence.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \exists L \in \mathbb{R},\; (x_n) \to L
  \;\Rightarrow\;
  \forall \varepsilon > 0,\; \exists N,\; \forall m, n \ge N,\;
  |x_m - x_n| < \varepsilon.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Given $\varepsilon > 0$, get $N$ with $|x_n - L| < \varepsilon/2$ for $n \ge N$.
For $m, n \ge N$: $|x_m - x_n| \le |x_m - L| + |L - x_n| < \varepsilon$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $(x_n) \to L \Rightarrow (x_n)$ is Cauchy.

\Given $(x_n) \to L$.

\Goal Show $\forall \varepsilon > 0,\; \exists N,\; \forall m, n \ge N,\;
|x_m - x_n| < \varepsilon$.

\Strategy Triangle inequality via the common limit: $|x_m - x_n| \le
|x_m - L| + |L - x_n|$.

\medskip

% --- let ε > 0; get N with |x_n - L| < ε/2 for n ≥ N ---

% --- for m, n ≥ N: |x_m - x_n| ≤ |x_m - L| + |L - x_n| < ε/2 + ε/2 = ε ---

\end{proof}

\begin{remark*}[Proof shape]
Straightforward $\varepsilon/2$ argument: pass through the limit as an
intermediate.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of convergence and Cauchy sequence.
  \item Triangle inequality.
\end{itemize}
\end{remark*}
