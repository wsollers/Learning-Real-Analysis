\subsubsection[Root Test]{Proof --- Root Test}

\bigskip

\noindent
\textbf{Source.}
Abbott, \emph{Understanding Analysis}; see \texttt{notes-series-tests.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
If $L = \limsup |a_n|^{1/n}$: $L < 1 \Rightarrow \sum a_n$ abs.\ converges; $L > 1 \Rightarrow$ diverges.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  L = \limsup |a_n|^{1/n} < 1 \Rightarrow \sum |a_n| < \infty.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Pick $r$ with $L < r < 1$. Eventually $|a_n|^{1/n} < r$, so $|a_n| < r^n$. Geometric series comparison.

\vspace{0.75em}
