\subsubsection[Absolute Convergence is Stable Under Rearrangement]{Proof --- Absolute Convergence is Stable Under Rearrangement}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-series-rearrangements.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $\sum a_n$ converges absolutely then every rearrangement also converges absolutely to the same sum.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \sum |a_n| < \infty \Rightarrow \sum |a_{\sigma(n)}| < \infty \text{ and } \sum a_{\sigma(n)} = \sum a_n.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} This combines Absolute Convergence is Rearrangement Invariant (same sum) with the observation that $\sum |a_{\sigma(n)}| = \sum |a_n|$ (just a rearrangement of nonneg terms, which cannot change the sum of a nonneg convergent series).

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $\sum a_n$ converges absolutely then every rearrangement also converges absolutely to the same sum.

\Given

\medskip

% --- ∑|a_{σ(n)}| = ∑|a_n| < ∞ (rearrangement of nonneg terms, covered by the nonneg rearrangement theorem) ---

\end{proof}

\begin{remark*}[Proof shape]
Absolute value part: nonneg rearrangement theorem. Sum part: Rearrangement Invariant theorem.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Absolute Convergence is Rearrangement Invariant.
  \item For nonneg series, any rearrangement has the same sum (can be proved from MCT and monotonicity).
\end{itemize}
\end{remark*}
