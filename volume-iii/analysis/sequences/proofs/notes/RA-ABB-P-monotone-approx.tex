\subsubsection[Monotone Approximation of Bounds]{Proof --- Monotone Approximation of Bounds}

\bigskip

\noindent
\textbf{Source.} See \texttt{monotone-approximation.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} Every bounded sequence $(x_n)$ satisfies $\liminf x_n \le \limsup x_n$, with equality iff the sequence converges.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \liminf x_n \le \limsup x_n,\quad \liminf x_n = \limsup x_n \Longleftrightarrow \lim x_n \text{ exists}.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} The tail infima are $\le$ the tail suprema pointwise ($i_k \le s_k$), so their limits satisfy $\liminf \le \limsup$. Equality follows from the Convergence Criterion.

\vspace{0.75em}

\noindent

\begin{remark*}[Proof shape]
Pointwise inequality on tail sup/inf $\Rightarrow$ inequality on limits. Equality gives the Convergence Criterion.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}

  \item Convergence Criterion via $\limsup$ and $\liminf$.

\end{remark*}
