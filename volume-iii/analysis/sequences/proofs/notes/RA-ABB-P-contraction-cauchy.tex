\subsubsection[Contraction Sequence Is Cauchy]{Proof --- Contraction Sequence Is Cauchy}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-recurrence-relations.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $|x_{n+1} - x_n| \le c^n M$ for some $c \in [0,1)$ and $M > 0$, then $(x_n)$ is Cauchy.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  |x_{n+1} - x_n| \le c^n M,\; 0 \le c < 1 \Rightarrow (x_n) \text{ Cauchy}.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Telescope: $|x_m - x_n| \le \sum_{k=n}^{m-1} c^k M \le Mc^n/(1-c)$. Since $c^n \to 0$, this goes to 0.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $|x_{n+1} - x_n| \le c^n M$ for some $c \in [0,1)$ and $M > 0$, then $(x_n)$ is Cauchy.

\Given 

\Goal 

\Strategy Telescope: $|x_m - x_n| \le \sum_{k=n}^{m-1} c^k M \le Mc^n/(1-c)$. Since $c^n \to 0$, this goes to 0.

\medskip

% --- WLOG m > n; telescope: |x_m - x_n| ≤ ∑_{k=n}^{m-1} |x_{k+1} - x_k| ---
% --- ≤ ∑_{k=n}^{m-1} c^k M = Mc^n · (1 - c^{m-n})/(1-c) ≤ Mc^n/(1-c) ---
% --- given ε, pick N with Mc^N/(1-c) < ε (possible since c^n → 0) ---
% --- for m > n ≥ N: |x_m - x_n| < ε ---

\end{proof}

\begin{remark*}[Proof shape]
Telescope and geometric series bound; the tail decays geometrically.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Triangle inequality (telescoping sum).
  \item Geometric series: $\sum_{k=n}^\infty c^k = c^n/(1-c)$ for $|c| < 1$.
  \item $c^n \to 0$ for $|c| < 1$.
\end{itemize}
\end{remark*}
