\subsubsection[Affine Recurrence Bound]{Proof --- Affine Recurrence Bound}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-recurrence-relations.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $x_{n+1} = c x_n + d$ (with $|c| < 1$), then $(x_n)$ converges to the fixed point $L = d/(1-c)$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  x_{n+1} = c x_n + d,\; |c| < 1 \Rightarrow x_n \to \frac{d}{1-c}.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} Let $y_n = x_n - L$ where $L = d/(1-c)$. Then $y_{n+1} = c y_n$, so $y_n = c^n y_0 \to 0$, giving $x_n \to L$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim If $x_{n+1} = c x_n + d$ (with $|c| < 1$), then $(x_n)$ converges to the fixed point $L = d/(1-c)$.

\Given 

\Goal 

\Strategy Let $y_n = x_n - L$ where $L = d/(1-c)$. Then $y_{n+1} = c y_n$, so $y_n = c^n y_0 \to 0$, giving $x_n \to L$.

\medskip
