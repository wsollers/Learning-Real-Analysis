\subsubsection[Comparison via Absolute Values]{Proof --- Comparison via Absolute Values}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-absolute-convergence.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $|a_n| \le b_n$ eventually and $\sum b_n < \infty$, then $\sum a_n$ converges absolutely.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  |a_n| \le b_n,\; \sum b_n < \infty \Rightarrow \sum |a_n| < \infty.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} This is just the Direct Comparison Test applied to $|a_n|$ and $b_n$, both nonneg.

\vspace{0.75em}

\noindent
