\subsubsection[Subadditivity of Limsup]{Proof --- Subadditivity of Limsup}

\bigskip

\noindent
\textbf{Source.} See \texttt{notes-limsup-liminf.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} $\limsup(x_n + y_n) \le \limsup x_n + \limsup y_n$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \limsup(x_n + y_n) \le \limsup x_n + \limsup y_n.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} $\sup(A+B) \le \sup A + \sup B$ applied to each tail set; pass to limits.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim $\limsup(x_n + y_n) \le \limsup x_n + \limsup y_n$.

\Strategy $\sup(A+B) \le \sup A + \sup B$ applied to each tail set; pass to limits.

\begin{remark*}[Proof shape]
Pointwise sub-additivity of sup $+$ limit.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item $\sup(A+B) \le \sup A + \sup B$ for bounded sets.
  \item Monotone Convergence of tail suprema.
  \item Order Limit Theorem.
\end{itemize}
\end{remark*}
