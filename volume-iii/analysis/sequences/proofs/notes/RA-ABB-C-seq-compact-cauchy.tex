\subsubsection[Sequential Compactness of Closed Intervals (via Cauchy)]{Proof --- Sequential Compactness of Closed Intervals (via Cauchy)}

\bigskip

\noindent
\textbf{Source.}
See \texttt{notes-cauchy.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Every sequence in $[a,b]$ has a convergent subsequence with limit in $[a,b]$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  \forall (a_n) \subseteq [a,b],\; \exists (a_{n_k}) \to L \in [a,b].
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Same as the version in notes-subsequences. Apply BW to get the convergent subsequence, use closedness of $[a,b]$ for the limit.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
\Claim Every sequence in $[a,b]$ has a convergent subsequence with limit in $[a,b]$.

\Given 

\Goal 

\Strategy Same as the version in notes-subsequences. Apply BW to get the convergent subsequence, use closedness of $[a,b]$ for the limit.

\medskip

% --- (a_n) bounded ⇒ BW gives convergent (a_{n_k}) → L ---
% --- [a,b] closed: limit of sequence in [a,b] is in [a,b] ---

\end{proof}

\begin{remark*}[Proof shape]
BW $+$ closedness. See also the identical corollary in notes-subsequences.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Bolzano--Weierstrass Theorem.
  \item Closed sets contain their limit points.
\end{itemize}
\end{remark*}
