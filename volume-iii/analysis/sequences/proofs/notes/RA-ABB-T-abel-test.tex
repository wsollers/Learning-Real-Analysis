\subsubsection[Abel Test]{Proof --- Abel Test}

\bigskip

\noindent
\textbf{Source.} Abbott, \emph{Understanding Analysis}; see \texttt{notes-series-tests.tex}.

\vspace{0.75em}

\noindent
\textbf{Goal.} If $(a_n)$ is monotone and bounded, and $\sum b_n$ converges, then $\sum a_n b_n$ converges.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
\[
  (a_n) \text{ monotone bounded},\; \sum b_n < \infty \Rightarrow \sum a_n b_n < \infty.
\]

\vspace{0.75em}

\noindent
\textbf{Proof strategy.} $(a_n)$ is monotone bounded so converges to some $L$. Write $a_n = L + c_n$ where $c_n \to 0$ decreasingly. Apply Dirichlet Test to $\sum c_n b_n$ (since $\sum b_n$ converges, its partial sums are bounded). Add $L \sum b_n$ which converges.

\vspace{0.75em}
