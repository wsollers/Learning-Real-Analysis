\subsection*{6. Finite Product Rule}
\textbf{Claim:} If $f_1, f_2, \dots, f_n$ are functions on an interval $I$ to $\mathbb{R}$ that are differentiable at $c \in I$, then:
\[ \left( \prod_{i=1}^{n} f_i \right)'(c) = \sum_{i=1}^{n} \left( f_i'(c) \prod_{j \neq i} f_j(c) \right) \]

\begin{proof}
We proceed by mathematical induction on $n$.

\textbf{Base Case ($n=1$):} 
For $n=1$, $(f_1)'(c) = f_1'(c)$, which is trivially true. 

\textbf{Inductive Hypothesis:} 
Assume the rule holds for $n=k$:
\[ (f_1 f_2 \cdots f_k)'(c) = \sum_{i=1}^{k} \left( f_i'(c) \prod_{j \neq i, j=1}^{k} f_j(c) \right) \]

\textbf{Inductive Step:} 
Consider $n = k+1$. Let $g = f_1 f_2 \cdots f_k$. Then the product is $g \cdot f_{k+1}$. By the **Product Rule for two functions**:
\[ (g \cdot f_{k+1})'(c) = g'(c)f_{k+1}(c) + g(c)f_{k+1}'(c) \]
Substituting the inductive hypothesis for $g'(c)$ and the definition of $g(c)$:
\[ \left[ \sum_{i=1}^{k} \left( f_i'(c) \prod_{j \neq i, j=1}^{k} f_j(c) \right) \right] f_{k+1}(c) + \left[ \prod_{i=1}^{k} f_i(c) \right] f_{k+1}'(c) \]
Distributing $f_{k+1}(c)$ into the sum:
\[ \sum_{i=1}^{k} \left( f_i'(c) \prod_{j \neq i, j=1}^{k+1} f_j(c) \right) + \left( f_{k+1}'(c) \prod_{j=1}^{k} f_j(c) \right) \]
This combines into a single sum from $i=1$ to $k+1$:
\[ \sum_{i=1}^{k+1} \left( f_i'(c) \prod_{j \neq i, j=1}^{k+1} f_j(c) \right) \]
Thus, the statement holds for all $n \in \mathbb{N}$ by induction.
\end{proof}
