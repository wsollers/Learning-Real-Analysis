\subsection*{5. Finite Sum Rule}
\textbf{Claim:} If $f_1, f_2, \dots, f_n$ are functions on an interval $I$ to $\mathbb{R}$ that are differentiable at $c \in I$, then:
\[ \left( \sum_{i=1}^{n} f_i \right)'(c) = \sum_{i=1}^{n} f_i'(c) \]

\begin{proof}
We proceed by mathematical induction on $n$.

\textbf{Base Case ($n=1$):} 
For $n=1$, the statement is $(f_1)'(c) = f_1'(c)$, which is clearly true. 

\textbf{Inductive Hypothesis:} 
Assume the statement holds for some $k \in \mathbb{N}$, such that:
\[ \left( \sum_{i=1}^{k} f_i \right)'(c) = \sum_{i=1}^{k} f_i'(c) \]

\textbf{Inductive Step:} 
We must show the statement holds for $n = k+1$. Let $g = \sum_{i=1}^{k} f_i$. Then:
\begin{align*}
\left( \sum_{i=1}^{k+1} f_i \right)'(c) &= \left( \left( \sum_{i=1}^{k} f_i \right) + f_{k+1} \right)'(c) \\
&= \left( g + f_{k+1} \right)'(c)
\end{align*}
By the **Addition Rule for Derivatives** (which we proved previously for two functions):
\[ (g + f_{k+1})'(c) = g'(c) + f_{k+1}'(c) \]
Substituting the inductive hypothesis for $g'(c)$:
\[ \left( \sum_{i=1}^{k} f_i \right)'(c) + f_{k+1}'(c) = \sum_{i=1}^{k} f_i'(c) + f_{k+1}'(c) = \sum_{i=1}^{k+1} f_i'(c) \]
Thus, by the principle of mathematical induction, the statement holds for all $n \in \mathbb{N}$.
\end{proof}
