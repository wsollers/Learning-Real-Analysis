\begin{theorem}[Closed Sets via Derived Set]
Let $(X,d)$ be a metric space and let $E \subseteq X$.
Then $E$ is closed if and only if
\[
E' \subseteq E.
\]
\end{theorem}


% =========================================================
% Proof: Closed iff E contains its Derived Set
% =========================================================

\subsection*{Closed Sets via Derived Set}
\label{prf:closed-derived-set}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and let $E \subseteq X$.
Then $E$ is closed if and only if $E' \subseteq E$.

\Given A metric space $(X,d)$ and a subset $E \subseteq X$.

\Goal To prove
\[
E \text{ is closed }
\;\Longleftrightarrow\;
E' \subseteq E.
\]

\Strategy Prove both directions:

\medskip

\noindent
($\Rightarrow$) Assume $E$ is closed.
Use the definition of closed and the definition of limit point
to show that every limit point of $E$ must lie in $E$.

\medskip

\noindent
($\Leftarrow$) Assume $E' \subseteq E$.
Use the sequential characterization of closure (or the ball definition)
to show that $E$ contains all its limit points,
and therefore is closed.

\end{proof}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of limit point.
  \item Definition of derived set $E'$.
  \item Characterization of closure.
\end{itemize}
\end{remark*}


\begin{theorem}[Sequential Characterization of the Derived Set]
Let $(X,d)$ be a metric space, let $E \subseteq X$, and let $x \in X$.
Then
\[
x \in E'
\]
if and only if there exists a sequence of distinct points in $E$
that converges to $x$.
\end{theorem}


% =========================================================
% Proof: Sequential Characterization of Derived Set
% =========================================================

\subsection*{Sequential Characterization of the Derived Set}
\label{prf:derived-set-sequential}

\begin{proof}
\Claim Let $(X,d)$ be a metric space, $E \subseteq X$, and $x \in X$.
Then $x \in E'$ if and only if there exists a sequence of distinct
points in $E$ converging to $x$.

\Given A metric space $(X,d)$, a subset $E \subseteq X$, and a point $x \in X$.

\Goal To prove
\[
x \in E'
\;\Longleftrightarrow\;
\exists (x_n) \subseteq E
\text{ with distinct terms such that }
x_n \to x.
\]

\Strategy Prove both directions:

\medskip

\noindent
($\Rightarrow$) Assume $x \in E'$.
Use the ball definition of limit point to construct a sequence
of distinct points of $E$ in shrinking balls around $x$.

\medskip

\noindent
($\Leftarrow$) Assume there exists a sequence of distinct points
of $E$ converging to $x$.
Use the definition of limit point to show that every ball
around $x$ intersects $E \setminus \{x\}$.

\end{proof}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of limit point.
  \item Definition of derived set.
  \item Sequential characterization of closure.
\end{itemize}
\end{remark*}
