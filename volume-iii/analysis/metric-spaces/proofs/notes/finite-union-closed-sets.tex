\subsection*{Finite Union of Closed Sets is Closed}
\label{prf:finite-union-closed}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and let $F_1,\dots,F_n \subseteq X$ be closed.
Then $\bigcup_{k=1}^n F_k$ is closed.

\Given A metric space $(X,d)$ and closed sets $F_1,\dots,F_n \subseteq X$.

\Goal To show that $F := \bigcup_{k=1}^n F_k$ is closed.

\Strategy Use your chosen definition of closed:
either argue sequentially (a convergent sequence in the union must eventually
lie in one of the finitely many closed sets), or use complements and De Morgan's laws.
\end{proof}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of closed set (sequential or complement-open).
  \item (Optional) De Morgan's law: $X \setminus \bigcup_{k=1}^n F_k = \bigcap_{k=1}^n (X \setminus F_k)$.
\end{itemize}
\end{remark*}
