\subsection*{Reflections are Isometries}
\label{prf:reflections-are-isometries}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:reflections-are-isometries]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Reflection across a hyperplane in $\mathbb{R}^n$ is an isometry.

\Given Reflection $H : \mathbb{R}^n \to \mathbb{R}^n$ across a hyperplane
with unit normal $\hat{n}$, so $H(x) = x - 2(\hat{n} \cdot x)\hat{n}$.

\Goal To show $\|H(x) - H(y)\|_2 = \|x - y\|_2$.

\Strategy The matrix of $H$ is $I - 2\hat{n}\hat{n}^\top$.
Verify this matrix is orthogonal (it satisfies $H^\top H = I$),
then apply Proposition~\ref{prop:rotations-are-isometries}.

\end{proof}

\begin{remark*}[Proof shape]
Compute $H^\top H$ explicitly, show it equals $I$, then appeal to
the rotations-are-isometries result since both share the orthogonality property.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Proposition~\ref{prop:rotations-are-isometries}: orthogonal maps are isometries.
  \item Linear algebra: $(I - 2\hat{n}\hat{n}^\top)^\top(I - 2\hat{n}\hat{n}^\top) = I$.
\end{itemize}
\end{remark*}
