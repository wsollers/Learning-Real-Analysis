\subsection*{Injectivity of Isometries}
\label{prf:isometry-injective}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:isometry-injective]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Every isometry $f : X \to Y$ is injective.

\Given Metric spaces $(X,d_X)$ and $(Y,d_Y)$ and an isometry $f$.

\Goal To show $f(x) = f(y) \Rightarrow x = y$.

\Strategy If $f(x) = f(y)$ then $d_X(x,y) = d_Y(f(x),f(y)) = d_Y(f(x),f(x)) = 0$,
so $x = y$ by M2.

\end{proof}

\begin{remark*}[Proof shape]
One line: the isometry condition converts equality of outputs to distance zero,
which forces equality of inputs via M2.
\end{remark*}

\begin{itemize}

  \item Definition~\ref{def:metric}: M2 (identity of indiscernibles).
\end{itemize}
\end{remark*}
