\subsection*{Convergence in the Product Metric}
\label{prf:product-convergence-multiple-metrics}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:product-convergence]{Back to Proposition in Notes
(Multiple Metrics)}
\end{remark*}

\begin{proof}
\Claim In the product metric on $X \times Y$, a sequence
$(x_n, y_n) \to (x, y)$ if and only if $x_n \to x$ and $y_n \to y$.

\Given Metric spaces $(X, d_X)$, $(Y, d_Y)$ with product metric
$d_\infty = \max(d_X, d_Y)$.

\Goal To prove the component-wise characterization.

\Strategy $d_\infty((x_n,y_n),(x,y)) = \max(d_X(x_n,x), d_Y(y_n,y)) \to 0$
iff both components $\to 0$.
See also Proposition~\ref{prop:product-convergence} for the full proof.

\end{proof}

  \item Real analysis: $\max(a_n, b_n) \to 0 \iff a_n \to 0$ and $b_n \to 0$.
\end{itemize}
