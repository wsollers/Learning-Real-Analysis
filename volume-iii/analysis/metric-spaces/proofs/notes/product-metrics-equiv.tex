\subsection*{The Three Product Metrics are Equivalent}
\label{prf:product-metrics-equiv}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:product-metrics-equiv]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim For any $(x_1,y_1), (x_2,y_2) \in X \times Y$:
\[
d_\infty \;\le\; d_2 \;\le\; d_1 \;\le\; 2\, d_\infty.
\]

\Given Metric spaces $(X,d_X)$ and $(Y,d_Y)$.
Write $a = d_X(x_1,x_2)$ and $b = d_Y(y_1,y_2)$.

\Goal To prove the three-way inequality chain
$\max(a,b) \le \sqrt{a^2+b^2} \le a+b \le 2\max(a,b)$.

\Strategy Each inequality is a short calculation with non-negative reals.
Work through each step: $\max(a,b)^2 \le a^2+b^2$; then $a^2+b^2 \le (a+b)^2$;
then $a+b \le 2\max(a,b)$.

\end{proof}

\begin{remark*}[Proof shape]
Three separate inequalities on non-negative reals $a, b$.
Each is a one- or two-line calculation; no metric axioms needed
beyond unpacking the definitions.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:product-metrics}: $d_1$, $d_2$, $d_\infty$ on $X \times Y$.
  \item Algebraic fact: $2ab \ge 0$, so $a^2 + b^2 \le (a+b)^2$.
  \item Definition of equivalent metrics: $c_1 d \le d' \le c_2 d$ for constants $c_1, c_2 > 0$.
\end{itemize}
\end{remark*}
