\subsection*{Closure Formula}
\label{prf:closure-formula}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:closure-formula]{Back to Theorem in Notes}
\end{remark*}

\begin{proof}
\Claim For any $E \subseteq X$: $\overline{E} = E \cup E'$.

\Given A metric space $(X,d)$ and $E \subseteq X$.

\Goal To show $\overline{E} = E \cup E'$.

\Strategy
$(\supseteq)$: $E \subseteq \overline{E}$ always (a set is contained in its closure).
$E' \subseteq \overline{E}$ because limit points are adherent points.

$(\subseteq)$: If $x \in \overline{E}$ then either $x \in E$ or every ball
around $x$ meets $E \setminus \{x\}$ (making $x$ a limit point of $E$).
So $x \in E \cup E'$.

\end{proof}

\begin{remark*}[Proof shape]
Each direction is a short unfolding of definitions.
The key step is: adherent but not in $E$ means limit point.
\end{remark*}
