\subsection*{Convergence is Preserved by Isometries}
\label{prf:convergence-preserved-by-isometry}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[lem:convergence-preserved]{Back to Lemma in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d_X)$ and $(Y,d_Y)$ be metric spaces and let $f : X \to Y$
be an isometry. If $x_n \to x$ in $X$, then $f(x_n) \to f(x)$ in $Y$.

\Given Metric spaces $(X,d_X)$ and $(Y,d_Y)$, an isometry $f : X \to Y$,
and a sequence $(x_n)$ in $X$ with $x_n \to x$.

\Goal To show $d_Y(f(x_n), f(x)) \to 0$.

\Strategy Since $f$ is an isometry, $d_Y(f(x_n), f(x)) = d_X(x_n, x)$ for all $n$.
Convergence $x_n \to x$ in $X$ says exactly that $d_X(x_n, x) \to 0$.

\medskip
\Let $\varepsilon > 0$.

\ByAss $x_n \to x$ in $X$: $\exists N \in \mathbb{N}$ such that $n \ge N \Rightarrow d_X(x_n, x) < \varepsilon$.

\ByDef{isometry} $d_Y(f(x_n), f(x)) = d_X(x_n, x)$ for all $n$.

\Therefore for all $n \ge N$:
\[
d_Y(f(x_n), f(x)) = d_X(x_n, x) < \varepsilon.
\]

Since $\varepsilon > 0$ was arbitrary, $f(x_n) \to f(x)$ in $Y$. \AsReq
\end{proof}

\begin{remark*}[Proof shape]
One line: the isometry condition $d_Y(f(x_n), f(x)) = d_X(x_n, x)$ converts
the convergence statement in $Y$ directly into the given convergence statement in $X$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:isometry}: $d_Y(f(u), f(v)) = d_X(u,v)$ for all $u,v$.
  \item Definition~\ref{def:convergence-metric}: $x_n \to x$ iff $d_X(x_n,x) \to 0$.
\end{itemize}
\end{remark*}
