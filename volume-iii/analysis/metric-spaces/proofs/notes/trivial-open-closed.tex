\subsection*{Trivial Open and Closed Sets}
\label{prf:trivial-open-closed}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:trivial-open-closed]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space.
Then $\varnothing$ and $X$ are both open and both closed.

\Given A metric space $(X,d)$.

\Goal To verify all four claims: $\varnothing$ open, $X$ open,
$\varnothing$ closed, $X$ closed.

\Strategy Openness of $\varnothing$ is vacuous (no points to check).
Openness of $X$ is trivial (every ball is contained in $X$).
Closedness follows from the complement characterization:
$X \setminus \varnothing = X$ (open) and $X \setminus X = \varnothing$ (open).

\end{proof}

\begin{remark*}[Proof shape]
Two lines for openness; two lines for closedness via complements.
The vacuous truth step for $\varnothing$ is worth writing out explicitly.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:open-set}: $U$ open iff every point has a ball inside $U$.
  \item Definition~\ref{def:closed-set}: $F$ closed iff $X \setminus F$ is open.
  \item Vacuous truth: universal quantification over $\varnothing$ is true.
\end{itemize}
\end{remark*}
