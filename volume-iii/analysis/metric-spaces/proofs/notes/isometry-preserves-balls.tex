\subsection*{Preservation of Open Balls}
\label{prf:isometry-preserves-balls}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:isometry-preserves-balls]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim If $f : X \to Y$ is an isometry then
$f(B_r(x)) = B_r(f(x))$ for all $x \in X$, $r > 0$.

\Given An isometry $f : X \to Y$, centre $x \in X$, radius $r > 0$.

\Goal To show $f(B_r(x)) = B_r(f(x))$ (set equality).

\Strategy
$(\subseteq)$: If $y \in B_r(x)$ then $d_Y(f(y),f(x)) = d_X(y,x) < r$.
$(\supseteq)$: If $v \in B_r(f(x))$ write $v = f(y)$ using injectivity
and compute $d_X(y,x) = d_Y(f(y),f(x)) < r$.

\end{proof}

\begin{remark*}[Proof shape]
Two-way containment; each direction is one application of the isometry
condition.
\end{remark*}
