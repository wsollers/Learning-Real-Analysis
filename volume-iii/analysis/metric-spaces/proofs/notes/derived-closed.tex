\subsection*{Derived Set is Closed}
\label{prf:derived-closed}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:derived-closed]{Back to Theorem in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and $E \subseteq X$.
The derived set $E' = \{p \in X : p \text{ is a limit point of } E\}$
is closed.

\Given A metric space $(X,d)$ and $E \subseteq X$.

\Goal To show $E'$ is closed, i.e.\ $E'$ contains all its limit points.

\Strategy Let $p$ be a limit point of $E'$.
By the sequential characterization, there exist $p_n \in E'$ with $p_n \to p$.
Each $p_n$ is a limit point of $E$, so nearby it there are points of $E$.
Use a diagonal construction to build a sequence in $E$ converging to $p$,
showing $p \in E'$.

\begin{remark*}[Proof shape]
Diagonal argument: combine convergence $p_n \to p$ with the limit-point
property of each $p_n$ to produce points of $E$ converging to $p$.
\end{remark*}

\begin{itemize}

  \item Triangle inequality to control $d(p, x_n)$ via $d(p, p_n) + d(p_n, x_n)$.
