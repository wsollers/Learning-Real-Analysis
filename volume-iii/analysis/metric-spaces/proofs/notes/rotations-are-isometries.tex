\subsection*{Rotations are Isometries}
\label{prf:rotations-are-isometries}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:rotations-are-isometries]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $R : \mathbb{R}^n \to \mathbb{R}^n$ be orthogonal ($R^\top R = I$).
Then $R$ is an isometry.

\Given An orthogonal matrix $R$ and points $x, y \in \mathbb{R}^n$.

\Goal To show $\|Rx - Ry\|_2 = \|x - y\|_2$.

\Strategy $\|Rx - Ry\|^2 = \|R(x-y)\|^2 = (R(x-y))^\top R(x-y)
= (x-y)^\top R^\top R (x-y) = (x-y)^\top I (x-y) = \|x-y\|^2$.

% -------------------------------------------------------

\end{proof}

\begin{remark*}[Proof shape]
One computation using $R^\top R = I$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of orthogonal matrix: $R^\top R = I$.
  \item Euclidean norm: $\|v\|^2 = v^\top v$.
  \item Linearity: $R(x-y) = Rx - Ry$.
\end{itemize}
\end{remark*}
