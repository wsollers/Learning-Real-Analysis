\subsection*{Boundary via Closure}
\label{prf:boundary-closure}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:boundary-closure]{Back to Theorem in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and $A \subseteq X$.
Then $\partial A = \overline{A} \cap \overline{X \setminus A}$.

\Given A metric space $(X,d)$ and $A \subseteq X$.

\Goal To show $x \in \partial A \iff x \in \overline{A} \cap \overline{X \setminus A}$.

\Strategy Unpack both sides using the geometric characterization of closure:
$x \in \overline{S}$ iff every ball around $x$ meets $S$.
So $x \in \overline{A} \cap \overline{X \setminus A}$ iff every ball meets $A$
\emph{and} every ball meets $X \setminus A$, which is exactly the definition of $\partial A$.

\end{proof}

\begin{remark*}[Proof shape]
Definitional unfolding using the geometric characterization of closure.
Both sides reduce to the same condition on balls.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:boundary}: $\partial A$ via balls meeting both $A$ and $X \setminus A$.
  \item Definition~\ref{def:closure}: $x \in \overline{S}$ iff every ball meets $S$.
  \item Intersection of two conditions via $\cap$.
\end{itemize}
\end{remark*}
