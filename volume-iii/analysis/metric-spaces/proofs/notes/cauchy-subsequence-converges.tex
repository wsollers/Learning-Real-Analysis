\subsection*{Cauchy Sequence with a Convergent Subsequence Converges}
\label{prf:cauchy-subsequence-converges}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:cauchy-subsequence]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space.
If $(x_n)$ is Cauchy and has a subsequence $(x_{n_k}) \to x$,
then $x_n \to x$.

\Given A Cauchy sequence $(x_n)$ and a subsequence $x_{n_k} \to x$.

\Goal To show $x_n \to x$, i.e.\ $d(x_n, x) \to 0$.

\Strategy Route $d(x_n, x)$ through a subsequence term $x_{n_k}$
using the triangle inequality. Choose $n_k$ large enough that
both the Cauchy gap $d(x_n, x_{n_k})$ and the subsequence gap
$d(x_{n_k}, x)$ are below $\varepsilon/2$.

\end{proof}

\begin{remark*}[Proof shape]
Triangle inequality through a well-chosen subsequence term:
Cauchy controls $d(x_n, x_{n_k})$; convergence controls $d(x_{n_k}, x)$.
\end{remark*}
