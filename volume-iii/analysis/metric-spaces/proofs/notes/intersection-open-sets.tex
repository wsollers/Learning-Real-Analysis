\subsection*{Finite Intersections of Open Sets are Open}
\label{prf:finite-intersection-open}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and let $U_1, \dots, U_n$ be open subsets of $X$.
Then $\bigcap_{k=1}^{n} U_k$ is open.

\Given Open sets $U_1, \dots, U_n \subseteq X$.

\Goal To show $U := \bigcap_{k=1}^{n} U_k$ is open.

\Strategy For a point in the intersection, each $U_k$ supplies its own ball.
Take the smallest radius; that ball fits inside all $n$ open sets simultaneously.
This step requires \emph{finite}ly many sets: the minimum of finitely many positive
numbers is positive; a minimum over infinitely many could be zero.

\medskip
\Let $x \in U$, so $x \in U_k$ for each $k = 1, \dots, n$.

Since each $U_k$ is open and $x \in U_k$: $\exists r_k > 0$ with $B_{r_k}(x) \subseteq U_k$.

\Let $r := \min\{r_1, r_2, \dots, r_n\} > 0$ (positive since it is the minimum of finitely many positive numbers).

For each $k$: $B_r(x) \subseteq B_{r_k}(x) \subseteq U_k$.

\Therefore $B_r(x) \subseteq \bigcap_{k=1}^n U_k = U$.

Since $x \in U$ was arbitrary, $U$ is open. \AsReq
\end{proof}

\begin{remark*}[Why finiteness is essential]
For an \emph{infinite} intersection the argument fails.
The sets $U_n = (-1/n, 1/n)$ in $\mathbb{R}$ are all open, but
$\bigcap_{n=1}^\infty U_n = \{0\}$ is not open.
The minimum radius would be $\inf_n r_n = 0$, which is not positive.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:open-closed-set}: open set definition.
  \item The fact that the minimum of finitely many positive numbers is positive.
\end{itemize}
\end{remark*}
