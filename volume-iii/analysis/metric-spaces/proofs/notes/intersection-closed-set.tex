\subsection*{Intersection of Closed Sets is Closed}
\label{prf:intersection-closed}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and let $\{F_i\}_{i \in I}$ be a collection
of closed subsets of $X$. Then $\bigcap_{i \in I} F_i$ is closed.

\Given A metric space $(X,d)$ and closed sets $F_i \subseteq X$ for each $i \in I$.

\Goal To show that $F := \bigcap_{i \in I} F_i$ is closed.

\Strategy Use your chosen definition of closed:
either the sequential characterization (limits of sequences stay inside),
or the complement-open characterization together with De Morgan's laws.
\end{proof}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of closed set (sequential or complement-open).
  \item (Optional) De Morgan's law: $X \setminus \bigcap_i F_i = \bigcup_i (X \setminus F_i)$.
\end{itemize}
\end{remark*}
