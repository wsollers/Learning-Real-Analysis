\subsection*{Open Ball as a Union of Closed Balls}
\label{prf:open-as-union-closed}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:open-as-union-closed]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space, $x \in X$, $r > 0$.
Then $B_r(x) = \bigcup_{0 < s < r} \overline{B}_s(x)$.

\Given A metric space $(X,d)$, a centre $x \in X$, and radius $r > 0$.

\Goal To show the two-way set containment.

\Strategy
$(\supseteq)$: Any point in some $\overline{B}_s(x)$ with $s < r$
satisfies $d(x,y) \le s < r$, so it is in $B_r(x)$.

$(\subseteq)$: Given $y \in B_r(x)$, find an $s$ with $d(x,y) < s < r$
and check $y \in \overline{B}_s(x)$.
The midpoint $s = \frac{d(x,y)+r}{2}$ works.

\end{proof}

\begin{remark*}[Proof shape]
Standard two-way containment. The interesting direction is $(\subseteq)$:
exhibit a specific $s$ between $d(x,y)$ and $r$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of open ball $B_r(x)$: $d(x,y) < r$.
  \item Definition of closed ball $\overline{B}_s(x)$: $d(x,y) \le s$.
\end{itemize}
\end{remark*}
