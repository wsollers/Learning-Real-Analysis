\subsection*{Isometry Group}
\label{prf:isometry-group}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:isometry-group]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim The set $\mathrm{Isom}(X)$ of all bijective isometries $f : X \to X$
forms a group under composition.

\Given A metric space $(X,d)$.

\Goal To verify the four group axioms: closure, associativity, identity, inverses.

\Strategy
\begin{itemize}
  \item \textbf{Closure}: Composition of two isometries is an isometry ---
        check $d(g \circ f(x), g \circ f(y)) = d(x,y)$.
  \item \textbf{Associativity}: Composition of functions is always associative.
  \item \textbf{Identity}: The identity map $\mathrm{id}_X$ is an isometry.
  \item \textbf{Inverses}: By Proposition~\ref{prop:inverse-isometry}.
\end{itemize}
