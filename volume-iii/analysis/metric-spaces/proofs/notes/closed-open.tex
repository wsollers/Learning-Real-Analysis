\subsection*{Closed Sets and Open Complements}
\label{prf:closed-iff-complement-open}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:closed-iff-complement-open]{Back to Theorem
(Closed iff Complement Open) in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and let $F \subseteq X$.
Then $F$ is closed if and only if its complement $X \setminus F$
is open.

\Given A metric space $(X,d)$ and a subset $F \subseteq X$.

\Goal To prove
\[
F \text{ is closed }
\;\Longleftrightarrow\;
X \setminus F \text{ is open.}
\]

\Strategy Prove both directions:

\medskip

\noindent
(\(\Rightarrow\)) Assume $F$ is closed.
Use the definition of closed (sequential or complement-based,
depending on which definition is primary in this section)
to show that for each point in $X \setminus F$,
there exists a ball contained entirely in $X \setminus F$.

\medskip

\noindent
(\(\Leftarrow\)) Assume $X \setminus F$ is open.
Use the definition of open sets to show that any sequence in $F$
cannot converge to a point outside $F$,
or equivalently, that limits of sequences in $F$
remain in $F$.

\end{proof}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of open ball.
  \item Definition of open set.
  \item Definition of closed set.
  \item Definition of convergence in a metric space (if using sequential form).
\end{itemize}
\end{remark*}
