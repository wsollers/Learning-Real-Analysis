\subsection*{Distance Zero to a Set Characterizes Closure}
\label{prf:dist-point-set-closure}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:dist-point-set-closure]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space, $A \subseteq X$ non-empty, and $x \in X$.
Then $d(x, A) = 0$ if and only if $x \in \overline{A}$.

\Given A metric space $(X,d)$, a non-empty set $A \subseteq X$, and $x \in X$.

\Goal To show $\inf_{a \in A} d(x,a) = 0 \iff x \in \overline{A}$.

\Strategy Both sides say the same thing: every ball around $x$ meets $A$.
$(\Rightarrow)$: If the infimum is $0$, for any $r > 0$ there exists $a \in A$
with $d(x,a) < r$, so $B_r(x) \cap A \neq \varnothing$.
By the geometric characterization of closure, $x \in \overline{A}$.
$(\Leftarrow)$: If $x \in \overline{A}$, every ball meets $A$, so no positive
real is a lower bound for $\{d(x,a) : a \in A\}$, hence the infimum is $0$.

\end{proof}

\begin{remark*}[Proof shape]
Translate the infimum condition and the closure condition into the same
statement about open balls: each is equivalent to ``every ball around $x$
meets $A$.''
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:dist-point-set}: $d(x,A) = \inf_{a \in A} d(x,a)$.
  \item Definition of $\overline{A}$ via the geometric characterization:
        $x \in \overline{A}$ iff every ball $B_r(x)$ meets $A$.
  \item Infimum characterization: $\inf S = 0$ iff $\forall\, r > 0,\; \exists\, s \in S$ with $s < r$.
\end{itemize}
\end{remark*}
