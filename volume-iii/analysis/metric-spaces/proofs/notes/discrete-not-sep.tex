\subsection*{An Uncountable Discrete Space is Not Separable}
\label{prf:discrete-not-sep}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:discrete-not-sep]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $X$ be an uncountable set with the discrete metric.
Then $(X, d_{\mathrm{disc}})$ is not separable.

\Given An uncountable set $X$ with $d_{\mathrm{disc}}(x,y) = 1$ for $x \neq y$.

\Goal To show no countable $D \subseteq X$ satisfies $\overline{D} = X$.

\Strategy In the discrete metric, every ball $B_{1/2}(x) = \{x\}$.
For $D$ to be dense, every ball must meet $D$, so every singleton
$\{x\}$ must meet $D$, meaning $x \in D$.
Since $x$ was arbitrary, $D = X$ --- which is uncountable.

\end{proof}

\begin{remark*}[Proof shape]
The discrete metric forces every dense set to be all of $X$;
since $X$ is uncountable, no countable dense set exists.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of the discrete metric: $B_{1/2}(x) = \{x\}$.
  \item Proposition~\ref{prop:dense-equiv}(ii): dense iff every ball meets $D$.
  \item Definition~\ref{def:separable}: separable = countable dense subset exists.
\end{itemize}
\end{remark*}
