\subsection*{Complete Subspace is Closed}
\label{prf:complete-subspace-closed}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:complete-subspace-closed]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and $Y \subseteq X$.
If $(Y, d|_{Y \times Y})$ is complete, then $Y$ is closed in $X$.

\Given A metric space $(X,d)$ and a complete subspace $Y \subseteq X$.

\Goal To show $Y$ is closed in $X$, i.e.\ $Y$ contains all its limit points.

\Strategy Let $(x_n)$ be a sequence in $Y$ converging to some $x \in X$.
Convergent sequences are Cauchy (Proposition~\ref{prop:convergent-implies-cauchy}),
so $(x_n)$ is Cauchy in $Y$.
By completeness of $Y$, $x_n \to y$ for some $y \in Y$.
Limits are unique, so $x = y \in Y$.

\end{proof}

\begin{remark*}[Proof shape]
Convergent in $X$ $\Rightarrow$ Cauchy in $Y$ $\Rightarrow$ converges in $Y$
$\Rightarrow$ same limit $\Rightarrow$ limit is in $Y$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Proposition~\ref{prop:convergent-implies-cauchy}: convergent $\Rightarrow$ Cauchy.
  \item Definition~\ref{def:complete}: complete metric space.
  \item Proposition~\ref{prop:cauchy-unique-limit}: uniqueness of limits.
  \item Theorem~\ref{thm:closed-equiv}: closed iff sequentially closed.
\end{itemize}
\end{remark*}
