\subsection*{Closed Subspace of a Complete Space is Complete}
\label{prf:closed-subspace-complete}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:closed-subspace-complete]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a complete metric space and $F \subseteq X$ closed.
Then $(F, d|_{F \times F})$ is complete.

\Given A complete metric space $(X,d)$ and a closed subset $F \subseteq X$.
A Cauchy sequence $(x_n)$ in $F$.

\Goal To show $(x_n)$ converges to a point in $F$.

\Strategy $(x_n)$ is also Cauchy in $X$. Since $X$ is complete, $x_n \to x$
for some $x \in X$. Since $F$ is closed and $(x_n)$ is a sequence in $F$
converging to $x$, conclude $x \in F$.

\end{proof}

\begin{remark*}[Proof shape]
Lift to $X$, use completeness of $X$, then use closedness of $F$
to pull the limit back into $F$.
\end{remark*}
