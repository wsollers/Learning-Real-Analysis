\subsection*{Sequential Characterization of Limit Points}
\label{prf:seq-char-limit}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:seq-char-limit]{Back to Theorem in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space, $E \subseteq X$, and $p \in X$.
Then $p$ is a limit point of $E$ if and only if there exists a sequence
of points $x_n \in E \setminus \{p\}$ with $x_n \to p$.

\Given A metric space $(X,d)$, a set $E \subseteq X$, and a point $p \in X$.

\Goal To prove the biconditional.

\Strategy
$(\Leftarrow)$: If such a sequence exists, every ball $B_r(p)$ contains
$x_n \neq p$ for large $n$, so $p$ is a limit point.

$(\Rightarrow)$: If $p$ is a limit point, every ball $B_{1/n}(p)$ contains
a point $x_n \in E \setminus \{p\}$. Show $(x_n)$ converges to $p$.

\end{proof}

\begin{remark*}[Proof shape]
$(\Leftarrow)$ is a direct verification.
$(\Rightarrow)$ uses the Archimedean-style construction: build a sequence
by picking a point from each shrinking ball $B_{1/n}(p)$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:limit-point}: $p$ is a limit point of $E$ iff every
        $B_r(p)$ contains a point of $E$ other than $p$.
  \item Definition~\ref{def:convergence-metric}: $x_n \to p$ iff $d(x_n,p) \to 0$.
  \item Archimedean property: for any $r > 0$ there exists $n$ with $1/n < r$.
\end{itemize}
\end{remark*}
