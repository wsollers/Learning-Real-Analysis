\subsection*{Cauchy Sequences are Bounded}
\label{prf:cauchy-bounded}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:cauchy-bounded]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space.
Every Cauchy sequence in $X$ is bounded.

\Given A metric space $(X,d)$ and a Cauchy sequence $(x_n)$.

\Goal To show $\exists\, p \in X$ and $R > 0$ such that $d(x_n, p) \le R$ for all $n$.

\Strategy Apply the Cauchy condition with $\varepsilon = 1$ to get $N$ such
that all terms beyond $N$ lie within a ball of radius $1$ around $x_N$.
The finitely many terms before $N$ contribute a finite set of distances;
take the maximum.

\end{proof}

\begin{remark*}[Proof shape]
Split the sequence at the $N$ from the Cauchy condition with $\varepsilon = 1$.
Tail is bounded by $1$; head is a finite set, so bounded by a finite max.
\end{remark*}

  \item Finite maximum of a finite set of real numbers exists.
