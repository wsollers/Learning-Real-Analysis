\subsection*{Equivalent Characterizations of Closed Sets}
\label{prf:closed-equiv}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:closed-equiv]{Back to Theorem in Notes}
\end{remark*}

\begin{proof}
\Claim The following are equivalent for $F \subseteq (X,d)$:
\begin{enumerate}[(i)]
  \item $F$ is closed (topological definition: $X \setminus F$ is open).
  \item $F$ contains all its limit points: $F' \subseteq F$.
  \item $F$ is sequentially closed: if $x_n \in F$ and $x_n \to p$, then $p \in F$.
\end{enumerate}

\Given A metric space $(X,d)$ and $F \subseteq X$.

\Goal To prove (i) $\Leftrightarrow$ (ii) $\Leftrightarrow$ (iii).

\Strategy
(i) $\Rightarrow$ (iii): If $x_n \in F$ converges to $p \notin F$, then
$p \in X \setminus F$ (open), so some ball around $p$ avoids $F$,
contradicting convergence.

(iii) $\Rightarrow$ (ii): Any limit point is the limit of a sequence in $F$
(sequential characterization), so sequential closure gives it lies in $F$.

(ii) $\Rightarrow$ (i): Use Theorem~\ref{thm:closed-derived}.
