\subsection*{Equivalent Characterizations of Closure}
\label{prf:closure-equivalences}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:closure-equivalences]{Back to Theorem in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space, $A \subseteq X$, $x \in X$.
The following are equivalent:
\begin{enumerate}[label=(\alph*)]
  \item $x \in \overline{A}$.
  \item $\forall r > 0,\; B_r(x) \cap A \neq \varnothing$.
  \item $\exists (a_n) \subseteq A$ with $a_n \to x$.
\end{enumerate}

\Given $(X,d)$, $A \subseteq X$, $x \in X$.

\Goal Prove (a) $\Leftrightarrow$ (b) $\Leftrightarrow$ (c).

\medskip
\noindent\textbf{(a) $\Rightarrow$ (b).}

\ByDef $\overline{A}$ is the smallest closed set containing $A$.
Suppose for contradiction that $\exists r_0 > 0$ with $B_{r_0}(x) \cap A = \varnothing$.
Then $A \subseteq X \setminus B_{r_0}(x)$.
The set $X \setminus B_{r_0}(x)$ is closed (it is the complement of an open ball)
and contains $A$.
So $\overline{A} \subseteq X \setminus B_{r_0}(x)$, meaning $x \notin \overline{A}$.
Contrapositive: $x \in \overline{A} \Rightarrow$ (b).

\medskip
\noindent\textbf{(b) $\Rightarrow$ (c).}

This is exactly the $(\Leftarrow)$ direction of
Proposition~\ref{prf:sequential-characterization-closure}:
for each $n$, pick $a_n \in B_{1/n}(x) \cap A$;
then $d(a_n, x) < 1/n \to 0$, so $a_n \to x$.

\medskip
\noindent\textbf{(c) $\Rightarrow$ (a).}

\Let $(a_n) \subseteq A$ with $a_n \to x$.
\Let $F$ be any closed set with $A \subseteq F$.
Then $(a_n) \subseteq F$ and $a_n \to x$, so $x \in F$
(since $F$ is closed).
Since $\overline{A}$ is one such closed set, $x \in \overline{A}$. \AsReq
\end{proof}

\begin{remark*}[Proof shape]
The cycle (a) $\to$ (b) $\to$ (c) $\to$ (a) is classic.
The key tool for (b) $\to$ (c) is the diagonal sequence construction;
(c) $\to$ (a) uses only the definition of closure.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Proposition~\ref{prop:balls-open-closed}: open balls are open, closed balls are closed.
  \item Definition~\ref{def:closure}: $\overline{A}$ = smallest closed superset.
  \item Definition of closed set (sequential characterization).
  \item Archimedean property: $1/n \to 0$.
\end{itemize}
\end{remark*}
