\subsection*{Symmetry of Set Distance}
\label{prf:dist-sets-symmetric}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:dist-sets-symmetric]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and $A, B \subseteq X$ non-empty.
Then $d(A, B) = d(B, A)$.

\Given A metric space $(X,d)$ and non-empty sets $A, B \subseteq X$.

\Goal To show $\inf_{a \in A, b \in B} d(a,b) = \inf_{b \in B, a \in A} d(b,a)$.

\Strategy The two infima range over the same set of values
$\{d(a,b) : a \in A, b \in B\}$,
since $d(a,b) = d(b,a)$ by metric symmetry M3.
The infimum of a set does not depend on how the set is parameterized.

\end{proof}

\begin{remark*}[Proof shape]
The distance set $\{d(a,b) : a \in A, b \in B\}$ equals
$\{d(b,a) : b \in B, a \in A\}$ by M3. Same set, same infimum.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:dist-between-sets}: $d(A,B) = \inf\{d(a,b) : a \in A, b \in B\}$.
  \item Definition~\ref{def:metric}: M3 (symmetry), $d(a,b) = d(b,a)$.
\end{itemize}
\end{remark*}
