\subsection*{Convergent Implies Cauchy}
\label{prf:convergent-implies-cauchy}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:convergent-implies-cauchy]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space.
Every convergent sequence in $X$ is Cauchy.

\Given A metric space $(X,d)$ and a sequence $(x_n)$ with $x_n \to x$.

\Goal To show: $\forall\, \varepsilon > 0,\; \exists N \in \mathbb{N}$ such that
$m, n \ge N \Rightarrow d(x_m, x_n) < \varepsilon$.

\Strategy Use convergence to find $N$ such that both $x_m$ and $x_n$
are within $\varepsilon/2$ of $x$. Then the triangle inequality via $x$
gives $d(x_m, x_n) < \varepsilon$.

\end{proof}

\begin{remark*}[Proof shape]
Classic $\varepsilon/2$ argument: route the distance $d(x_m, x_n)$ through
the limit $x$ using the triangle inequality.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:convergence-metric}: $x_n \to x$ iff $d(x_n,x) \to 0$.
  \item Definition~\ref{def:cauchy}: Cauchy sequence in a metric space.
  \item Definition~\ref{def:metric}: M4 (triangle inequality).
\end{itemize}
\end{remark*}
