\subsection*{Three Forms of the Triangle Inequality}
\label{prf:triangle-three-forms}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:triangle-three-forms]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and $x,y,z \in X$.
Then $d(x,z) \le d(x,y)+d(y,z)$, \;
$d(x,y) \le d(x,z)+d(z,y)$, \;
$d(y,z) \le d(y,x)+d(x,z)$.

\Given A metric space $(X,d)$ and points $x,y,z \in X$.

\Goal To derive all three symmetric forms from M4 and M3.

\Strategy Axiom M4 gives the first form directly.
The other two come from relabelling the points in M4 and
using symmetry M3 to swap any pair.
There is no calculation; just bookkeeping.

\end{proof}

\begin{remark*}[Proof shape]
M4 gives one form; symmetry M3 turns relabellings into the other two.
All three are literally the same inequality with different names on the vertices.
\end{remark*}
