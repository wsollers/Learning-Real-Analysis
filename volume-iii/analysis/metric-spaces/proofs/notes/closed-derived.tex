\subsection*{Closed Sets and Derived Sets}
\label{prf:closed-derived}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:closed-derived]{Back to Theorem in Notes}
\end{remark*}

\begin{proof}
\Claim $E \subseteq X$ is closed if and only if $E' \subseteq E$
(every limit point of $E$ belongs to $E$).

\Given A metric space $(X,d)$ and $E \subseteq X$.

\Goal To prove: $E$ closed $\iff$ $E' \subseteq E$.

\Strategy
$(\Rightarrow)$: If $E$ is closed and $p \in E'$, use the sequential
characterization to get $x_n \in E \setminus \{p\}$ with $x_n \to p$.
Since $E$ is closed (sequentially), $p \in E$.

$(\Leftarrow)$: If $E' \subseteq E$, take any sequence $x_n \in E$ with
$x_n \to p$. Either $p \in E$ already, or $p$ is a limit point of $E$,
so $p \in E' \subseteq E$.

\end{proof}

\begin{remark*}[Proof shape]
Each direction uses the sequential characterization to translate between
``closed'' and ``absorbs limit points.''
\end{remark*}
