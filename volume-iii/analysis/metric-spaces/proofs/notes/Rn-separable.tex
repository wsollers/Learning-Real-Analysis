\subsection*{$\mathbb{R}^n$ is Separable}
\label{prf:Rn-separable}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:Rn-separable]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim $(\mathbb{R}^n, d_2)$ is separable with countable dense subset $\mathbb{Q}^n$.

\Given The Euclidean space $(\mathbb{R}^n, d_2)$ and $D = \mathbb{Q}^n$.

\Goal To show: (1) $\mathbb{Q}^n$ is countable; (2) $\overline{\mathbb{Q}^n} = \mathbb{R}^n$.

\Strategy
\begin{itemize}
  \item Countability: $\mathbb{Q}^n$ is a finite Cartesian product of countable sets.
  \item Density: given $x = (x_1, \ldots, x_n) \in \mathbb{R}^n$ and $\varepsilon > 0$,
        approximate each coordinate $x_i$ by a rational $q_i$ within $\varepsilon/\sqrt{n}$,
        then bound $d_2(q, x)$ using the Euclidean norm formula.
\end{itemize}

\end{proof}

\begin{remark*}[Proof shape]
Component-wise rational approximation; Euclidean distance calculation
to show the resulting point is within $\varepsilon$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Proposition~\ref{prop:Q-dense-R}: $\mathbb{Q}$ is dense in $\mathbb{R}$.
  \item Fact: finite product of countable sets is countable.
  \item Definition~\ref{def:separable}: separable metric space.
  \item Euclidean metric: $d_2(q,x) = \sqrt{\sum_i (q_i - x_i)^2}$.
\end{itemize}
\end{remark*}
