\subsection*{Translations are Isometries}
\label{prf:translations-are-isometries}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:translations-are-isometries]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim For any $a \in \mathbb{R}^n$, the map $T_a(x) = x + a$ is an isometry.

\Given $a \in \mathbb{R}^n$ and points $x, y \in \mathbb{R}^n$.

\Goal To show $\|T_a(x) - T_a(y)\|_2 = \|x - y\|_2$.

\Strategy $T_a(x) - T_a(y) = (x+a) - (y+a) = x - y$. Done.
