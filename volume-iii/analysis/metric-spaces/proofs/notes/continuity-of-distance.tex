\subsection*{Continuity of Distance}
\label{prf:continuity-of-distance}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:continuity-of-distance]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and fix $z \in X$.
If $x_n \to x$ in $(X,d)$, then $d(x_n, z) \to d(x, z)$ in $\mathbb{R}$.

\Given A metric space $(X,d)$, a fixed point $z \in X$,
and a sequence $(x_n)$ with $x_n \to x$.

\Goal To show $|d(x_n, z) - d(x, z)| \to 0$.

\Strategy The reverse triangle inequality gives
$|d(x_n,z) - d(x,z)| \le d(x_n, x)$.
The right side tends to $0$ by the assumption $x_n \to x$.

\end{proof}

\begin{remark*}[Proof shape]
One-line squeeze: bound $|d(x_n,z) - d(x,z)|$ by $d(x_n,x)$
using the reverse triangle inequality, then take $n \to \infty$.
\end{remark*}

  \item Definition~\ref{def:convergence-metric}: $x_n \to x$ iff $d(x_n,x) \to 0$.
\end{itemize}
\end{remark*}
