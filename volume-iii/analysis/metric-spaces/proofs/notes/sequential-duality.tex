\subsection*{Sequential Duality of Open and Closed Sets}
\label{prf:sequential-duality}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:sequential-duality]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim $U \subseteq X$ is open if and only if:
whenever $x_n \to p \in U$, eventually $x_n \in U$.

\Given A metric space $(X,d)$ and $U \subseteq X$.

\Goal To prove: $U$ open $\iff$ $U$ is sequentially open.

\Strategy
$(\Rightarrow)$: If $U$ is open and $x_n \to p \in U$, pick a ball
$B_r(p) \subseteq U$; convergence gives $x_n \in B_r(p) \subseteq U$ eventually.

$(\Leftarrow)$: Contrapositive: if $U$ is not open, some $p \in U$ has
no ball inside $U$. Build $x_n \in X \setminus U$ with $x_n \to p$,
violating sequential openness.

\end{proof}

\begin{remark*}[Proof shape]
The $(\Rightarrow)$ direction is a direct read-off of the definitions.
The $(\Leftarrow)$ direction uses a $1/n$ construction to produce
a sequence in the complement.
\end{remark*}

\end{remark*}
