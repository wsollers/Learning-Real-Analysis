\subsection*{Generalised Triangle Inequality}
\label{prf:generalised-triangle}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:generalised-triangle]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and $x_1, \ldots, x_n \in X$ with $n \ge 2$.
Then:
\[
d(x_1, x_n) \le \sum_{k=1}^{n-1} d(x_k, x_{k+1}).
\]

\Given A metric space $(X,d)$ and points $x_1, \ldots, x_n \in X$.

\Goal To show the telescoping bound on $d(x_1, x_n)$.

\Strategy This is induction on $n$. The base case $n=2$ is the metric axiom M4.
For the inductive step, apply M4 to split off the last step, then apply
the inductive hypothesis to the remainder.

\end{proof}
