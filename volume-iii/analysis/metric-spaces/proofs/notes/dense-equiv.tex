\subsection*{Geometric Characterization of Density}
\label{prf:dense-equiv}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:dense-equiv]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space and $D \subseteq X$.
The following are equivalent:
\begin{enumerate}[(i)]
  \item $\overline{D} = X$ ($D$ is dense).
  \item Every open ball meets $D$: $\forall\, x \in X,\; \forall\, r>0,\; B_r(x) \cap D \neq \varnothing$.
  \item Every non-empty open set meets $D$.
\end{enumerate}

\Given A metric space $(X,d)$ and subset $D \subseteq X$.

\Goal To establish (i) $\Leftrightarrow$ (ii) $\Leftrightarrow$ (iii).

\Strategy Prove (i) $\Rightarrow$ (ii) $\Rightarrow$ (iii) $\Rightarrow$ (i)
using the geometric characterization of closure (every ball around a closure
point meets the set) and the fact that open sets contain open balls.

\end{proof}

\begin{remark*}[Proof shape]
Circular implication: each condition is a mild strengthening or weakening
of the closure definition. The circle closes because open sets always
contain open balls.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:dense}: $D$ dense iff $\overline{D} = X$.
  \item Definition~\ref{def:closure}: $x \in \overline{D}$ iff every ball around $x$
        meets $D$.
  \item Definition~\ref{def:open-set}: every open set contains an open ball around
        each of its points.
\end{itemize}
\end{remark*}
