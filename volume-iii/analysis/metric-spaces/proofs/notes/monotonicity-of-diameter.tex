\subsection*{Monotonicity of Diameter}
\label{prf:monotonicity-of-diameter}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:monotonicity-of-diameter]{Back to Theorem
(Monotonicity of Diameter) in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X, d)$ be a metric space and let $A \subseteq B \subseteq X$.
Then $\operatorname{diam}(A) \le \operatorname{diam}(B)$.

\Given A metric space $(X, d)$ and subsets $A \subseteq B \subseteq X$.

\Goal To show $\operatorname{diam}(A) \le \operatorname{diam}(B)$.

\Strategy Unfold the definition of diameter as a supremum of pairwise
distances. Since $A \subseteq B$, the set of pairwise distances in $A$
is a subset of the set of pairwise distances in $B$.
A supremum over a larger collection is at least as large.

\medskip
\noindent\textbf{Case 1: $A = \varnothing$.}

\ByDef $\operatorname{diam}(\varnothing) = 0$, and diameters are non-negative, so
$\operatorname{diam}(A) = 0 \le \operatorname{diam}(B)$.

\medskip
\noindent\textbf{Case 2: $A \neq \varnothing$.}

\ByDef
\[
  \operatorname{diam}(A) \;=\; \sup\bigl\{\, d(x, y) \mid x, y \in A \,\bigr\}.
\]

Let $x, y \in A$ be arbitrary.

\ByAss $A \subseteq B$, so $x \in B$ and $y \in B$.

\Therefore $d(x, y)$ belongs to the set $\{\, d(u, v) \mid u, v \in B \,\}$.

\Hence $d(x, y) \le \sup\{\, d(u, v) \mid u, v \in B \,\} = \operatorname{diam}(B)$.

Since every pairwise distance in $A$ is bounded above by $\operatorname{diam}(B)$,
and $\operatorname{diam}(A)$ is the \emph{least} such upper bound:
\[
  \operatorname{diam}(A) \le \operatorname{diam}(B). \qquad \AsReq
\]
\end{proof}

\begin{remark*}[Proof shape]
Direct supremum comparison: $A \subseteq B$ makes the distance-set of $A$
a subset of the distance-set of $B$, so $\sup$ of the smaller set
cannot exceed $\sup$ of the larger.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:diameter}: $\operatorname{diam}(A) = \sup\{d(x,y) : x,y \in A\}$,
        $\operatorname{diam}(\varnothing) = 0$.
  \item If $S \subseteq T \subseteq \mathbb{R}$ then $\sup S \le \sup T$.
\end{itemize}
\end{remark*}
