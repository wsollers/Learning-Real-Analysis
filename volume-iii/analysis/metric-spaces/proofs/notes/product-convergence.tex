\subsection*{Convergence in a Product is Component-Wise}
\label{prf:product-convergence}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:product-convergence]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(x_n, y_n)$ be a sequence in $X \times Y$ and $(x,y) \in X \times Y$.
Then $(x_n, y_n) \to (x,y)$ if and only if $x_n \to x$ and $y_n \to y$.

\Given Metric spaces $(X, d_X)$ and $(Y, d_Y)$ with product metric $d_\infty$
on $X \times Y$.

\Goal To prove the biconditional.

\Strategy Work with $d_\infty = \max(d_X, d_Y)$.
$(\Rightarrow)$: $d_\infty \to 0$ forces each component $\le d_\infty \to 0$.
$(\Leftarrow)$: both components $\to 0$ forces $\max \to 0$.

\end{proof}

\begin{remark*}[Proof shape]
The $\max$ metric makes the biconditional nearly trivial:
convergence of $\max(a_n, b_n) \to 0$ is equivalent to
$a_n \to 0$ and $b_n \to 0$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:product-metrics}: $d_\infty = \max(d_X, d_Y)$.
  \item Proposition~\ref{prop:product-metrics-equiv}: equivalence of $d_1, d_2, d_\infty$.
  \item Definition~\ref{def:convergence-metric}: convergence in a metric space.
  \item Real analysis: $\max(a_n, b_n) \to 0 \iff a_n \to 0 \land b_n \to 0$.
\end{itemize}
\end{remark*}
