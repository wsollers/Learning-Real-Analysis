\subsection*{Continuity of the Metric}
\label{prf:continuity-of-the-metric}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:continuity-of-the-metric]{Back to Theorem
(Continuity of the Metric) in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space. If $(x_n)$ and $(y_n)$ are sequences in $X$
such that $x_n \to x$ and $y_n \to y$, then
\[
d(x_n, y_n) \to d(x,y).
\]

\Given A metric space $(X,d)$ and sequences $(x_n)$, $(y_n)$ in $X$ with
$x_n \to x$ and $y_n \to y$.

\Goal To show that $d(x_n,y_n) \to d(x,y)$ as a real sequence.

\Strategy Use the definition of convergence in $\mathbb{R}$ for the sequence
$d(x_n,y_n)$, together with a bound obtained from the triangle inequality
(or the reverse triangle inequality) relating $d(x_n,y_n)$ and $d(x,y)$
to the quantities $d(x_n,x)$ and $d(y_n,y)$.

\end{proof}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition of convergence in a metric space.
  \item Triangle inequality (and/or reverse triangle inequality).
  \item Definition of convergence of a real sequence.
\end{itemize}
\end{remark*}
