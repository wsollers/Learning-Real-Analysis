\subsection*{Reverse Triangle Inequality}
\label{prf:reverse-triangle}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:reverse-triangle]{Back to Proposition in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space. For all $x, y, z \in X$:
\[
|d(x,z) - d(y,z)| \le d(x,y).
\]

\Given A metric space $(X,d)$ and points $x, y, z \in X$.

\Goal To show $|d(x,z) - d(y,z)| \le d(x,y)$.

\Strategy An absolute value $|a - b| \le c$ is equivalent to two inequalities:
$a - b \le c$ and $b - a \le c$. Prove each using the triangle inequality
applied in different orders. Symmetry of the metric handles the swap.

\end{proof}

\begin{remark*}[Proof shape]
Two applications of the triangle inequality (one for each direction of the
absolute value), plus symmetry to swap $x$ and $y$.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}

  \item $|a - b| \le c \iff (a - b \le c) \land (b - a \le c)$.
\end{itemize}
\end{remark*}
