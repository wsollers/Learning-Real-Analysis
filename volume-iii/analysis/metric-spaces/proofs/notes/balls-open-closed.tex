\subsection*{Balls are Open and Closed}
\label{prf:balls-open-closed}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[prop:balls-open-closed]{Back to Proposition (Balls are open and closed) in Notes}
\end{remark*}

\begin{proof}
\Claim Let $(X,d)$ be a metric space, $x \in X$, and $r > 0$.
(i) $B_r(x)$ is open. \quad (ii) $\overline{B}_r(x)$ is closed.

\Given A metric space $(X,d)$, a centre $x \in X$, and radius $r > 0$.

\Goal (i) Every point of $B_r(x)$ has an open ball around it contained in $B_r(x)$.
(ii) $X \setminus \overline{B}_r(x)$ is open.

\Strategy Both parts use the radial gap: the slack $\delta > 0$ left
between a point and the boundary of its ball.

\medskip
\noindent\textit{Part (i): $B_r(x)$ is open.}

\Let $y \in B_r(x)$.

\ByDef $d(x, y) < r$.

\Let $\delta := r - d(x, y) > 0$. (This is the radial gap of $y$ in $B_r(x)$.)

\Goal Show $B_\delta(y) \subseteq B_r(x)$.

\Let $z \in B_\delta(y)$, so $d(y, z) < \delta$.

\ByThm{Triangle inequality (M4)}
\[
d(x, z) \le d(x, y) + d(y, z) < d(x, y) + \delta = d(x,y) + r - d(x,y) = r.
\]

\Therefore $z \in B_r(x)$.

Since $z \in B_\delta(y)$ was arbitrary, $B_\delta(y) \subseteq B_r(x)$.
Since $y \in B_r(x)$ was arbitrary, $B_r(x)$ is open. \AsReq

\medskip
\noindent\textit{Part (ii): $\overline{B}_r(x)$ is closed, i.e.\ $U := X \setminus \overline{B}_r(x)$ is open.}

\Let $y \in U$, so $d(x, y) > r$.

\Let $\delta := d(x, y) - r > 0$.

\Goal Show $B_\delta(y) \subseteq U$, i.e.\ $B_\delta(y) \cap \overline{B}_r(x) = \varnothing$.

\Let $z \in B_\delta(y)$, so $d(y, z) < \delta$.

By the triangle inequality applied to $x, z, y$:
\[
d(x, y) \le d(x, z) + d(z, y),
\]
so
\[
d(x, z) \ge d(x, y) - d(z, y) > d(x, y) - \delta = r.
\]

\Therefore $d(x, z) > r$, so $z \notin \overline{B}_r(x)$, so $z \in U$.

Since $z \in B_\delta(y)$ was arbitrary, $B_\delta(y) \subseteq U$.
Since $y \in U$ was arbitrary, $U$ is open, so $\overline{B}_r(x)$ is closed. \AsReq
\end{proof}

\begin{remark*}[Proof shape]
Both parts are the same strategy: for a point inside/outside a ball,
compute the radial gap $\delta$ (the slack in the relevant inequality)
and use the triangle inequality to show a $\delta$-ball fits entirely
inside/outside the original ball.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item Definition~\ref{def:open-ball} and Definition~\ref{def:closed-ball}.
  \item Definition~\ref{def:open-closed-set}: $U$ is open iff every point has a ball inside $U$.
  \item Triangle inequality (M4) for the metric $d$.
\end{itemize}
\end{remark*}
