\begin{theorem}[Linearity]
For functions \(f,g\) and constants \(a,b\),

\[
\Delta(af + bg) = a\,\Delta f + b\,\Delta g.
\]
\end{theorem}

\begin{proof}
\[
\Delta(af+bg)(x)
=
(af+bg)(x+1)-(af+bg)(x)
\]

\[
=
a(f(x+1)-f(x)) + b(g(x+1)-g(x))
\]

\[
=
a\Delta f(x)+b\Delta g(x).
\]
\end{proof}


\begin{proposition}
If \(c\) is constant, then

\[
\Delta c = 0.
\]
\end{proposition}


\begin{theorem}[Product Rule]
\[
\Delta(fg)(x)
=
f(x)\Delta g(x) + g(x+1)\Delta f(x).
\]
\end{theorem}
