\subsubsection{Basic Definitions}

\begin{definition}[Vector Space — informal description]
A vector space is a set equipped with addition and scalar multiplication
that behave like familiar vector arithmetic.
\end{definition}

\begin{remark}
The informal description suggests structure, but precision requires axioms.
\end{remark}

\begin{definition}[Vector Space — formal definition]
Let $\mathbb{F}$ be a field.
A vector space over $\mathbb{F}$ is a set $V$ together with
\[
+: V \times V \to V,
\quad
\cdot: \mathbb{F} \times V \to V,
\]
satisfying the standard axioms (associativity, commutativity,
distributivity, identity elements, and inverses).
\end{definition}

\begin{remark}[Logical structure]
A vector space is an ordered triple
\[
(V,+,\cdot)
\]
satisfying a finite list of universal algebraic axioms.
\end{remark}
