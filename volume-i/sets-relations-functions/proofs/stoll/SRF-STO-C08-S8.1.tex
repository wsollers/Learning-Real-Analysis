\subsubsection[Finite Axiom of Choice]{Proof --- SRF-STO-C08-S8.1}

\bigskip

\noindent
\textbf{Source.}
\srccite{StollSetTheoryLogic}{Chapter~2, \S8, Exercise~8.1}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Establish the axiom of choice in the form $(\mathrm{AC}_2)$ for finite
collections: prove by induction that if
$A_1 \times A_2 \times \cdots \times A_n = \varnothing$, then at least
one $A_i = \varnothing$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
$\forall n \geq 1$: $\prod_{i=1}^n A_i = \varnothing
\Rightarrow \exists i \in \{1,\ldots,n\}$, $A_i = \varnothing$.

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Induction on $n$. Key lemma: $X \times Y = \varnothing \iff X = \varnothing$
or $Y = \varnothing$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
We proceed by induction on $n \geq 1$.

\medskip
\textit{Base case} ($n = 1$): $A_1 = \varnothing$ implies the product is
empty and $A_1$ is empty. \checkmark

\medskip
\textit{Inductive step:} Assume the result holds for products of $n$ sets.
Consider $A_1 \times \cdots \times A_{n+1}
= (A_1 \times \cdots \times A_n) \times A_{n+1}$.

A Cartesian product $X \times Y = \varnothing$ iff $X = \varnothing$ or
$Y = \varnothing$.

% Apply to X = A_1 x ... x A_n and Y = A_{n+1}.
% Case split and apply the induction hypothesis.
\end{proof}

\vspace{0.75em}

\noindent
\textbf{Remarks.}
\begin{itemize}
  \item \textbf{Significance:} The full Axiom of Choice is needed for
    \emph{infinite} families. This exercise shows the finite case requires
    no extra axioms beyond ordinary induction --- it illustrates where the
    foundational content of AC actually lies.
  \item \textbf{Mistakes to avoid:} Be precise about how the $n$-fold
    product is defined recursively. The associativity of Cartesian product
    is doing real work here.
\end{itemize}
