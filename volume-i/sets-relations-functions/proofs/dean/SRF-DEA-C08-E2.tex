\subsubsection[Power Set Ordered by Inclusion is a Chain Iff $|X| \leq 1$]{Proof --- SRF-DEA-C08-E2}

\bigskip

\noindent
\textbf{Source.}
\srccite{Dean2015OrderTheory}{Chapter~VIII, Exercise~2}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
State and prove a theorem of the form:
\[
  \mathcal{P}(X), \text{ ordered by inclusion, is a chain if and only if } \ldots
\]

\vspace{0.75em}

\noindent
\textbf{Claim.}
$\mathcal{P}(X)$ ordered by $\subseteq$ is a chain if and only if $|X| \leq 1$.

\vspace{0.75em}

\noindent
\textbf{Logical form.}
$(\forall A, B \subseteq X: A \subseteq B \lor B \subseteq A)
\iff |X| \leq 1$.

\vspace{0.75em}

\noindent
\textbf{Proof strategy.}
Biconditional. $(\Leftarrow)$: direct check for $|X|=0$ and $|X|=1$.
$(\Rightarrow)$: if $|X| \geq 2$, exhibit two incomparable subsets (singletons).

\vspace{0.75em}

\noindent
\textbf{Proof.}
\begin{proof}
$(\Leftarrow)$: If $X = \varnothing$, then $\mathcal{P}(X) = \{\varnothing\}$,
a single-element chain. If $|X|=1$, say $X=\{a\}$, then
$\mathcal{P}(X) = \{\varnothing, \{a\}\}$ with $\varnothing \subseteq \{a\}$,
a chain.

$(\Rightarrow)$: Suppose $|X| \geq 2$. Let $a, b \in X$ with $a \neq b$.
Consider $A = \{a\}$ and $B = \{b\}$. Then $A \not\subseteq B$ (since $a
\notin B$) and $B \not\subseteq A$ (since $b \notin A$). Hence $A$ and $B$
are incomparable in $(\mathcal{P}(X), \subseteq)$, so it is not a chain.
\end{proof}

\vspace{0.75em}

\noindent
\textbf{Remarks.}
\begin{itemize}
  \item \textbf{Canonical non-example:} $\mathcal{P}(\{a,b\}) =
    \{\varnothing, \{a\}, \{b\}, \{a,b\}\}$ under $\subseteq$ is the
    simplest poset that is not a chain. This structure (called the
    ``diamond'' or $M_2$) recurs throughout lattice theory.
\end{itemize}
