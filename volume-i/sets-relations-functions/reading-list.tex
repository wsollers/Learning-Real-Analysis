\noindent
\textbf{Goal.} Master sets, relations, and functions: set operations, products,
relations, equivalence relations, orders, functions, images/preimages, and composition.

\vspace{1em}

\begin{longtable}{|p{4.2cm}|p{10.8cm}|}
\hline
\textbf{Source} & \textbf{Required Reading for Mastery} \\
\hline

Gerstein — \textit{Introduction to Mathematical Structures and Proofs} &
\begin{itemize}
  \item Chapter 2 — Sets (incl.\ relations/partitions)
  \item Chapter 3 — Functions (injective/surjective/bijective; composition)
\end{itemize}
\\ \hline

Halmos — \textit{Naive Set Theory} &
\begin{itemize}
  \item Sets and operations; relations; functions; families; inverses/composites
\end{itemize}
\\ \hline

Chiossi — \textit{Essential Mathematics for Undergraduates} &
\begin{itemize}
  \item Part I Chapter 2 — Naive Set Theory
  \item Part I Chapter 3 — Functions
\end{itemize}
\\ \hline

Stoll — \textit{Set Theory and Logic} &
\begin{itemize}
  \item Sets/relations/functions; equivalence relations; order relations (foundations)
\end{itemize}
\\ \hline

Kolmogorov \& Fomin — \textit{Introductory Real Analysis} &
\begin{itemize}
  \item Chapter 1: Sets and Functions (images/preimages, equivalence relations)
\end{itemize}
\\ \hline

Suppes — \textit{Axiomatic Set Theory} &
\begin{itemize}
  \item Basic set operations and relations (reference-level preliminaries)
\end{itemize}
\\ \hline

\end{longtable}
