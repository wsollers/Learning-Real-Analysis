\noindent
\textbf{Goal.} Prove $(\forall x)(\forall y)(\forall z)(\forall w)((x-y)+(z-w)=(x+z)-(y+w))$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\[
\begin{array}{r l l}
(1) & \text{Let x, y, z, w be arbitrary} &  \\
(2) & \text{(x-y)+(z-w)}  & \text{LHS} \\
(3) & \text{x + (-y) + z + (-w)} & \text{Definition of subtraction} \\
(4) & \text{x + z + (-y) + (-w)} & \text{Associativity, Commutativity} \\
(5) & \text{(y+w)+(-(y+w))=0} & \text{Definition of additive inverse} \\
(6) & \text{-(y+w)=(-y)+(-w)} & \text{Uniqueness of Inverse} \\
(7) & \text{x + z + -(y + w)} & \text{Substitution} \\
(8) & \text{x + z - (y + w)} & \text{Definition of subtraction} \\
(9) & \text{(x + z) - (y + w)} & \text{Associativity} \\
(10) & (\forall x)(\forall y)(\forall z)(\forall w)((x-y)+(z-w)=(x+z)-(y+w)) & \text{Universal Generalization} \\
\end{array}
\]
