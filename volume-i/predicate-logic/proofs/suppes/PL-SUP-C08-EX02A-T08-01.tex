\noindent
\textbf{Goal.} Prove $(\forall x)(\forall y)(\forall z)\bigl((x-y)+(y-z)=x-z\bigr)$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\[
\begin{array}{r l l}
(1) & \text{Let $x,y,z$ be arbitrary.} & \text{Universal Generalization setup} \\
(2) & (x-y)+(y-z) & \text{Start (LHS)} \\
(3) & = (x+(-y))+(y-z) & \text{Definition of subtraction} \\
(4) & = (x+(-y))+(y+(-z)) & \text{Definition of subtraction} \\
(5) & = ((x+(-y))+y)+(-z) & \text{Additive Associativity} \\
(6) & = (x+((-y)+y))+(-z) & \text{Additive Associativity} \\
(7) & = (x+0)+(-z) & \text{Additive Inverse: }(-y)+y=0 \\
(8) & = x+(-z) & \text{Additive Identity: }x+0=x \\
(9) & = x-z & \text{Definition of subtraction} \\
(10) & (x-y)+(y-z)=x-z & \text{From (2)--(9)} \\
(11) & (\forall x)(\forall y)(\forall z)\bigl((x-y)+(y-z)=x-z\bigr) & \text{Universal Generalization} \\
\end{array}
\]
