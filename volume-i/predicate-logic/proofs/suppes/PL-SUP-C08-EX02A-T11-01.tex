\noindent
\textbf{Goal.} Prove $(\forall x)(\forall y)((-y+-x)+(x+y)=0)$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\[
\begin{array}{r l l}
(1) & \text{Let $x,y$ be arbitrary.} &  \\
(2) & (-y+-x)+(x+y) & \text{LHS} \\
(3) & (-y+y)+(x+-x) & \text{Associativity and commutativity} \\
(4) & 0+0 & \text{Additive inverse} \\
(5) & 0 & \text{Identity} \\
(6) & (\forall x)(\forall y)((-y+-x)+(x+y)=0) & \forall\text{-Gen}
\end{array}
\]
