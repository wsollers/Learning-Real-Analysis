\section*{Proof — RA-ABB-C01-S1.3-E06A}

\noindent
\hyperlink{ws-RA-ABB-C01-S1.3-E06A}{\textbf{← Back to worksheet}}

\noindent
\textbf{Source.}
\srccite{AbbottUnderstandingAnalysis}{Chapter~1, \S1.3, Exercise~1.3.6(a)}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Given sets $A$ and $B$, define
\[
A+B=\{a+b : a\in A \text{ and } b\in B\}.
\]
Let $s=\sup A$ and $t=\sup B$. Show that $s+t$ is an upper bound for $A+B$.

\vspace{0.75em}

\noindent
\textbf{Proof (stub).}
\[
\begin{array}{r l l}
(1) & \text{Let }a\in A\text{ and }b\in B\text{ be arbitrary.} & \\
(2) & a\le s \text{ and } b\le t & \text{Definition of supremum as upper bound} \\
(3) & a+b \le s+t & \text{Order properties of addition} \\
(4) & \text{Conclude }s+t\text{ is an upper bound for }A+B. &
\end{array}
\]
