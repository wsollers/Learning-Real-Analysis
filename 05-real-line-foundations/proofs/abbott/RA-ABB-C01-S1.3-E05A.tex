\section*{Proof — RA-ABB-C01-S1.3-E05A}

\noindent
\hyperlink{ws-RA-ABB-C01-S1.3-E05A}{\textbf{← Back to worksheet}}

\noindent
\textbf{Source.}
\srccite{AbbottUnderstandingAnalysis}{Chapter~1, \S1.3, Exercise~1.3.5(a)}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Let $A\subseteq\mathbb{R}$ be nonempty and bounded above, and let $c\in\mathbb{R}$. Define
\[
cA=\{\, ca : a\in A \,\}.
\]
If $c\ge 0$, show that $\sup(cA)=c\,\sup(A)$.

\vspace{0.75em}

\noindent
\textbf{Proof (stub).}
\[
\begin{array}{r l l}
(1) & \text{Let }s=\sup(A). & \\
(2) & \text{Show }cs\text{ is an upper bound of }cA\text{ (use }a\le s\Rightarrow ca\le cs\text{ for }c\ge0). & \\
(3) & \text{Let }u\text{ be any upper bound of }cA. & \\
(4) & \text{Translate }u\text{ being an upper bound of }cA\text{ into a bound on }A. & \\
(5) & \text{Use minimality of }s\text{ to conclude }cs\le u. & \\
(6) & \text{Conclude }\sup(cA)=cs. &
\end{array}
\]
