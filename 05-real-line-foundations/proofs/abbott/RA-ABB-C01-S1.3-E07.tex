\section*{Proof — RA-ABB-C01-S1.3-E07}

\noindent
\hyperlink{ws-RA-ABB-C01-S1.3-E07}{\textbf{← Back to worksheet}}

\noindent
\textbf{Source.}
\srccite{AbbottUnderstandingAnalysis}{Chapter~1, \S1.3, Exercise~1.3.7}.

\vspace{0.75em}

\noindent
\textbf{Goal.} Prove that if $a$ is an upper bound for $A$, and if $a\in A$, then $a=\sup A$.

\vspace{0.75em}

\noindent
\textbf{Proof (stub).}
\[
\begin{array}{r l l}
(1) & \text{Assume }a\text{ is an upper bound of }A\text{ and }a\in A. & \\
(2) & \text{Let }s=\sup A. & \\
(3) & \text{Use: }s\le a\text{ because }a\text{ is an upper bound and }s\text{ is least.} & \\
(4) & \text{Use: }a\le s\text{ because }a\in A\text{ and }s\text{ is an upper bound.} & \\
(5) & \text{Conclude }a=s. &
\end{array}
\]
