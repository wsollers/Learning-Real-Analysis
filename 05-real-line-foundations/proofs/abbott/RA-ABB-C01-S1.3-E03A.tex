\section*{Proof — RA-ABB-C01-S1.3-E03A}

\noindent
\hyperlink{ws-RA-ABB-C01-S1.3-E03A}{\textbf{← Back to worksheet}}

\noindent
\textbf{Source.}
\srccite{AbbottUnderstandingAnalysis}{Chapter~1, \S1.3, Exercise~1.3.3(a)}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Let $A$ be nonempty and bounded below, and define
\[
B=\{\, b\in\mathbb{R} : b \text{ is a lower bound for } A \,\}.
\]
Show that $\sup B = \inf A$.

\vspace{0.75em}

\noindent
\textbf{Proof (stub).}
\[
\begin{array}{r l l}
(1) & \text{Assume }A\neq \varnothing\text{ and is bounded below. Define }B\text{ as above.} & \\
(2) & \text{Show }B\neq\varnothing\text{ and is bounded above.} & \\
(3) & \text{Let }\beta=\sup B\text{ (exists by completeness).} & \\
(4) & \text{Prove }\beta\text{ is a lower bound for }A. & \\
(5) & \text{Prove }\beta\text{ is the greatest lower bound for }A. & \\
(6) & \text{Conclude }\beta=\inf A. & \\
\end{array}
\]
