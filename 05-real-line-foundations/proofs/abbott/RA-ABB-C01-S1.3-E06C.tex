\noindent
\textbf{Goal.} Show that $\sup(A+B)=s+t$.

\vspace{0.75em}

\noindent
\textbf{Proof (stub).}
\[
\begin{array}{r l l}
(1) & \text{From (a), }s+t\text{ is an upper bound of }A+B. & \\
(2) & \text{Let }u\text{ be any upper bound of }A+B. & \\
(3) & \text{Use (b) to show }t\le u-a\text{ for each }a\in A. & \\
(4) & \text{Rearrange to }a\le u-t\text{ for each }a\in A. & \\
(5) & \text{Conclude }u-t\text{ is an upper bound of }A,\text{ hence }s\le u-t. & \\
(6) & \text{Therefore }s+t\le u. & \\
(7) & \text{Conclude }\sup(A+B)=s+t. &
\end{array}
\]
