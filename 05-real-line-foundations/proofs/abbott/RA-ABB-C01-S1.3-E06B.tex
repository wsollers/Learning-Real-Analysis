\section*{Proof — RA-ABB-C01-S1.3-E06B}

\noindent
\hyperlink{ws-RA-ABB-C01-S1.3-E06B}{\textbf{← Back to worksheet}}

\noindent
\textbf{Source.}
\srccite{AbbottUnderstandingAnalysis}{Chapter~1, \S1.3, Exercise~1.3.6(b)}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Let $u$ be an arbitrary upper bound for $A+B$, and temporarily fix $a\in A$.
Show that $t \le u-a$.

\vspace{0.75em}

\noindent
\textbf{Proof (stub).}
\begin{itemize}
\item Start from: $u$ upper bounds $A+B$ means $\forall a\in A\,\forall b\in B,\ a+b\le u$.
\item Fix $a\in A$ (but keep $b\in B$ arbitrary).
\item Derive $b\le u-a$ for all $b\in B$.
\item Conclude $u-a$ is an upper bound of $B$, hence $t=\sup B\le u-a$.
\end{itemize}
