\subsubsection*{TB-P1-C01-WS-002 (10 problems)}
\hypertarget{TB-P1-C01-WS-002}{}

\noindent\textbf{Navigation: }
\hyperlink{P1-TB-C01}{Back to chapter}

\begin{enumerate}

\item \textbf{Negation flips sup/inf.}
Let $A \subseteq \mathbb{R}$ and define $B = \{-x : x \in A\}$.
Find and prove relations between $\sup A$ and $\inf B$, and between $\inf A$ and $\sup B$.
(\hyperlink{TB-P1-C01-015}{proof})

\item \textbf{Translation invariance of sup/inf.}
Let $A \subseteq \mathbb{R}$ and let $r \in \mathbb{R}$. Define $B = \{x+r : x \in A\}$.
Prove that
\[
\sup B = \sup A + r
\quad\text{and}\quad
\inf B = \inf A + r.
\]
(\hyperlink{TB-P1-C01-016}{proof})

\item \textbf{Positive scaling of sup/inf.}
Let $A \subseteq \mathbb{R}$ and let $r>0$. Define $B = \{rx : x \in A\}$.
Prove that
\[
\sup B = r\,\sup A
\quad\text{and}\quad
\inf B = r\,\inf A.
\]
(\hyperlink{TB-P1-C01-017}{proof})

\item \textbf{Negative scaling of sup/inf.}
Let $A \subseteq \mathbb{R}$ and let $r<0$. Define $B = \{rx : x \in A\}$.
Express $\sup B$ and $\inf B$ in terms of $\sup A$ and $\inf A$, and prove your formulas.
(\hyperlink{TB-P1-C01-018}{proof})

\item \textbf{Inclusion bounds sups/infs.}
Let $A,B \subseteq \mathbb{R}$ with $A \subseteq B$.
Assuming the relevant bounds exist, prove:
\[
\sup A \le \sup B
\quad\text{and}\quad
\inf B \le \inf A.
\]
(\hyperlink{TB-P1-C01-019}{proof})

\item \textbf{Supremum of a union.}
Let $A,B \subseteq \mathbb{R}$ and $C = A \cup B$. Assume $A$ and $B$ are bounded above.
Prove
\[
\sup C = \max\{\sup A,\sup B\}.
\]
(\hyperlink{TB-P1-C01-020}{proof})

\item \textbf{Intersection: sup inequality + strict example.}
Let $A,B \subseteq \mathbb{R}$ and $C = A \cap B$, with $C \neq \emptyset$.
Prove
\[
\sup C \le \min\{\sup A,\sup B\},
\]
and give an example where the inequality is strict.
(\hyperlink{TB-P1-C01-021}{proof})

\item \textbf{Minkowski sum: supremum.}
Let $A,B \subseteq \mathbb{R}$ and define
\[
C = A+B := \{x+y : x\in A,\ y\in B\}.
\]
Assume $A$ and $B$ are bounded above. Prove
\[
\sup C = \sup A + \sup B.
\]
(\hyperlink{TB-P1-C01-022}{proof})

\item \textbf{Minkowski sum: infimum.}
With $C=A+B$ as above, assume $A$ and $B$ are bounded below. Prove
\[
\inf C = \inf A + \inf B.
\]
(\hyperlink{TB-P1-C01-023}{proof})

\item \textbf{“Not an upper bound” characterization.}
Let $E \subseteq \mathbb{R}$. Prove that $x$ is \emph{not} an upper bound of $E$ if and only if
there exists $e \in E$ such that $e > x$.
(\hyperlink{TB-P1-C01-024}{proof})

\end{enumerate}
