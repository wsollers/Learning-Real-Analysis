\subsubsection*{TB-P1-C01-WS-001 (10 problems)}
\hypertarget{TB-P1-C01-WS-001}{}

\noindent\textbf{Navigation: }
\hyperlink{P1-TB-C01}{Back to chapter}

\begin{enumerate}

\item \textbf{De Morgan's law.}
Prove $(A\cup B)^c = A^c \cap B^c$ (complements relative to a fixed universe).
(\hyperlink{TB-P1-C01-001}{proof})

\item \textbf{Set difference distributes over union.}
Prove $A\setminus (B\cup C) = (A\setminus B)\cap (A\setminus C)$.
(\hyperlink{TB-P1-C01-002}{proof})

\item \textbf{Uniqueness of least upper bounds.}
Prove that if a set $S \subseteq \mathbb{R}$ has a supremum, then it is unique.
(\hyperlink{TB-P1-C01-003}{proof})

\item \textbf{Characterization of supremum.}
Let $S \subseteq \mathbb{R}$ be nonempty and bounded above.
Prove that $\alpha = \sup S$ if and only if:
\begin{enumerate}[label=(\roman*)]
\item $\alpha$ is an upper bound of $S$, and
\item for every $\varepsilon > 0$ there exists $s \in S$ such that $\alpha - \varepsilon < s$.
\end{enumerate}
(\hyperlink{TB-P1-C01-004}{proof})

\item \textbf{Existence of infimum via supremum.}
Let $S \subseteq \mathbb{R}$ be nonempty and bounded below.
Prove that
\[
\inf S = -\sup(-S).
\]
(\hyperlink{TB-P1-C01-005}{proof})

\item \textbf{Order translation invariance.}
Prove that if $a \le b$, then $a + c \le b + c$ for all $c \in \mathbb{R}$.
(\hyperlink{TB-P1-C01-006}{proof})

\item \textbf{Order scaling by positive numbers.}
Prove that if $a \le b$ and $c > 0$, then $ac \le bc$.
(\hyperlink{TB-P1-C01-007}{proof})

\item \textbf{Archimedean property.}
Prove that for every $x \in \mathbb{R}$ there exists $n \in \mathbb{N}$ such that $n > x$.
(\hyperlink{TB-P1-C01-008}{proof})

\item \textbf{No maximum for open bounded intervals.}
Prove that the interval $(a,b)$ has no maximum element.
(\hyperlink{TB-P1-C01-009}{proof})

\item \textbf{Supremum of translated sets.}
Let $S \subseteq \mathbb{R}$ be bounded above and let $c \in \mathbb{R}$.
Prove that
\[
\sup(S + c) = \sup S + c,
\]
where $S + c = \{s + c : s \in S\}$.
(\hyperlink{TB-P1-C01-010}{proof})

\end{enumerate}
