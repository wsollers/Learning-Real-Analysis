\subsubsection*{GER-P1-C02-WS-001 (12 problems)}
\hypertarget{GER-P1-C02-WS-001}{}

\noindent\textbf{Navigation: }
\hyperlink{P1-GER-C02}{Back to chapter}

\begin{enumerate}
\item \textbf{Basic inclusions.} Prove $A\cap B\subseteq A$ and $A\cap B\subseteq B$. (\hyperlink{GER-P1-C02-001}{proof})
\item \textbf{De Morgan.} Prove $(A\cup B)^c=A^c\cap B^c$. (\hyperlink{GER-P1-C02-002}{proof})
\item \textbf{Set difference.} Prove $A\setminus(B\cup C)=(A\setminus B)\cap(A\setminus C)$. (\hyperlink{GER-P1-C02-003}{proof})
\item \textbf{Symmetric difference.} Prove $A\triangle B=B\triangle A$. (\hyperlink{GER-P1-C02-004}{proof})
\item \textbf{Power set monotonicity.} If $A\subseteq B$, prove $\mathcal P(A)\subseteq\mathcal P(B)$. (\hyperlink{GER-P1-C02-005}{proof})
\item \textbf{Cartesian product distributes over union.} Prove $A\times(B\cup C)=(A\times B)\cup(A\times C)$. (\hyperlink{GER-P1-C02-006}{proof})
\item \textbf{Empty intersection.} Show $A\times\varnothing=\varnothing$. (\hyperlink{GER-P1-C02-007}{proof})
\item \textbf{Indexed unions.} Prove $\bigcup_{i\in I}(A_i\cap B)\subseteq(\bigcup_{i\in I}A_i)\cap B$. (\hyperlink{GER-P1-C02-008}{proof})
\item \textbf{Partitions.} Prove equivalence relations induce partitions. (\hyperlink{GER-P1-C02-009}{proof})
\item \textbf{Converses.} Prove partitions induce equivalence relations. (\hyperlink{GER-P1-C02-010}{proof})
\item \textbf{Functions as sets.} Prove a function is a special relation. (\hyperlink{GER-P1-C02-011}{proof})
\item \textbf{Induction principle.} State and prove the principle of mathematical induction. (\hyperlink{GER-P1-C02-012}{proof})
\end{enumerate}
