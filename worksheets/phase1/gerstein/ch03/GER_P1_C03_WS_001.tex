\subsubsection*{GER-P1-C03-WS-001 (10 problems)}
\hypertarget{GER-P1-C03-WS-001}{}

\noindent\textbf{Navigation: }
\hyperlink{P1-GER-C03}{Back to chapter}

\begin{enumerate}
\item \textbf{Injective characterization.} Prove $f$ is injective iff $f(x)=f(y)\Rightarrow x=y$. (\hyperlink{GER-P1-C03-001}{proof})
\item \textbf{Surjective characterization.} Prove $f$ is surjective iff $\forall y\exists x\,f(x)=y$. (\hyperlink{GER-P1-C03-002}{proof})
\item \textbf{Bijective inverse.} Prove a bijection has an inverse function. (\hyperlink{GER-P1-C03-003}{proof})
\item \textbf{Inverse properties.} Prove $(f^{-1})^{-1}=f$. (\hyperlink{GER-P1-C03-004}{proof})
\item \textbf{Composition associativity.} Prove $(h\circ g)\circ f=h\circ(g\circ f)$. (\hyperlink{GER-P1-C03-005}{proof})
\item \textbf{Composition injectivity.} If $g\circ f$ is injective, prove $f$ is injective. (\hyperlink{GER-P1-C03-006}{proof})
\item \textbf{Composition surjectivity.} If $g\circ f$ is surjective, prove $g$ is surjective. (\hyperlink{GER-P1-C03-007}{proof})
\item \textbf{Images.} Prove $f(A\cup B)=f(A)\cup f(B)$. (\hyperlink{GER-P1-C03-008}{proof})
\item \textbf{Preimages.} Prove $f^{-1}(C\cap D)=f^{-1}(C)\cap f^{-1}(D)$. (\hyperlink{GER-P1-C03-009}{proof})
\item \textbf{Constant functions.} Characterize injectivity/surjectivity of constant functions. (\hyperlink{GER-P1-C03-010}{proof})
\end{enumerate}
