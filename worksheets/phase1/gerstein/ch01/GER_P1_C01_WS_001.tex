\subsubsection*{GER-P1-C01-WS-001 (12 problems)}
\hypertarget{GER-P1-C01-WS-001}{}

\noindent\textbf{Navigation: }
\hyperlink{P1-GER-C01}{Back to chapter}

\begin{enumerate}
\item \textbf{Translation to symbols.}
Translate each statement into predicate logic, then negate it and simplify.
(\hyperlink{GER-P1-C01-001}{proof})

\item \textbf{Truth tables.}
Determine whether each formula is a tautology, contradiction, or contingent.
(\hyperlink{GER-P1-C01-002}{proof})

\item \textbf{Logical equivalence without tables.}
Prove $(p\to q)\wedge(p\to r)\equiv p\to(q\wedge r)$ using equivalence laws.
(\hyperlink{GER-P1-C01-003}{proof})

\item \textbf{Contrapositive.}
Prove $p\to q$ is logically equivalent to $\neg q\to\neg p$.
(\hyperlink{GER-P1-C01-004}{proof})

\item \textbf{Converse failure.}
Give a counterexample showing the converse of a true implication need not be true.
(\hyperlink{GER-P1-C01-005}{proof})

\item \textbf{Direct proof.}
Prove: If $n$ is odd, then $n^2$ is odd.
(\hyperlink{GER-P1-C01-006}{proof})

\item \textbf{Contradiction.}
Prove: There is no smallest positive real number.
(\hyperlink{GER-P1-C01-007}{proof})

\item \textbf{Cases.}
Prove: For every integer $n$, $n^2$ is even or odd.
(\hyperlink{GER-P1-C01-008}{proof})

\item \textbf{Quantifier order.}
Determine the truth of $\forall x\exists y(y>x)$ and $\exists y\forall x(y>x)$ over $\mathbb{R}$.
(\hyperlink{GER-P1-C01-009}{proof})

\item \textbf{Uniqueness pattern.}
State and prove the uniqueness pattern using quantifiers.
(\hyperlink{GER-P1-C01-010}{proof})

\item \textbf{Set-builder to predicate.}
Rewrite $x\in\{t\in\mathbb{R}:t^2<2\}$ as a predicate and negate it.
(\hyperlink{GER-P1-C01-011}{proof})

\item \textbf{Debug a proof.}
Identify and correct the logical error: “If $p\vee q$ is true, then $p$ is true.”
(\hyperlink{GER-P1-C01-012}{proof})
\end{enumerate}
