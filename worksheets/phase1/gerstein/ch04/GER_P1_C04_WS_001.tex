\subsubsection*{GER-P1-C04-WS-001 (10 problems)}
\hypertarget{GER-P1-C04-WS-001}{}

\noindent\textbf{Navigation: }
\hyperlink{P1-GER-C04}{Back to chapter}

\begin{enumerate}
\item \textbf{Finite sets.} Prove a subset of a finite set is finite. (\hyperlink{GER-P1-C04-001}{proof})
\item \textbf{Injections and finiteness.} Prove an injective map from a finite set is bijective onto its image. (\hyperlink{GER-P1-C04-002}{proof})
\item \textbf{Pigeonhole principle.} State and prove it. (\hyperlink{GER-P1-C04-003}{proof})
\item \textbf{Countability of $\mathbb{Z}$.} Prove $\mathbb{Z}$ is countable. (\hyperlink{GER-P1-C04-004}{proof})
\item \textbf{Countability of $\mathbb{Q}$.} Prove $\mathbb{Q}$ is countable. (\hyperlink{GER-P1-C04-005}{proof})
\item \textbf{Uncountability.} Prove $(0,1)$ is uncountable. (\hyperlink{GER-P1-C04-006}{proof})
\item \textbf{Subsets of countable sets.} Prove any subset of a countable set is countable or finite. (\hyperlink{GER-P1-C04-007}{proof})
\item \textbf{Finite unions.} Prove a finite union of finite sets is finite. (\hyperlink{GER-P1-C04-008}{proof})
\item \textbf{Infinite unions.} Give an example where a countable union of finite sets is infinite. (\hyperlink{GER-P1-C04-009}{proof})
\item \textbf{Bijections preserve cardinality.} Prove sets in bijection have equal cardinality. (\hyperlink{GER-P1-C04-010}{proof})
\end{enumerate}
