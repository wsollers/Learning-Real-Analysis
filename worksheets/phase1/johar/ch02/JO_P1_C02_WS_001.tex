\subsubsection*{JO-P1-C02-WS-001 (8 problems)}
\hypertarget{JO-P1-C02-WS-001}{}

\noindent\textbf{Navigation: }
\hyperlink{P1-JOHAR}{Back to Johar section}

\begin{enumerate}
\item \textbf{Inverse relation.} Let $R$ be a relation on a set $A$ and $R^{-1}$ its inverse. Prove $(R^{-1})^{-1}=R$. (\hyperlink{JO-P1-C02-001}{proof})
\item \textbf{Congruence mod $n$ is an equivalence relation.} Fix $n\in\mathbb{N}$ with $n\ge 2$. Prove that $a\equiv b\pmod n$ is an equivalence relation on $\mathbb{Z}$. (\hyperlink{JO-P1-C02-002}{proof})
\item \textbf{Order on $\mathbb{N}$ is a total order.} Prove that $\le$ on $\mathbb{N}$ is reflexive, antisymmetric, transitive, and total. (\hyperlink{JO-P1-C02-003}{proof})
\item \textbf{Translation invariance of order on $\mathbb{Z}$.} Prove that if $a\le b$ in $\mathbb{Z}$ then $a+c\le b+c$ for all $c\in\mathbb{Z}$. (\hyperlink{JO-P1-C02-004}{proof})
\item \textbf{Multiplication by positive preserves order.} Prove that if $a\le b$ in $\mathbb{Z}$ and $c\in\mathbb{Z}$ with $c>0$, then $ac\le bc$. (\hyperlink{JO-P1-C02-005}{proof})
\item \textbf{Trichotomy on $\mathbb{Z}$.} Prove that for any $z\in\mathbb{Z}$ exactly one of $z<0$, $z=0$, $z>0$ holds. (\hyperlink{JO-P1-C02-006}{proof})
\item \textbf{Cancellation law in $\mathbb{Z}$.} Prove that for integers $a,b,c$, if $a+c=b+c$ then $a=b$. (\hyperlink{JO-P1-C02-007}{proof})
\item \textbf{No largest natural number.} Prove that $\mathbb{N}$ has no largest element. (\hyperlink{JO-P1-C02-008}{proof})
\end{enumerate}
