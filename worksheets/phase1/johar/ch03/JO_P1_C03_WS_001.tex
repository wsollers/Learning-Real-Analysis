\subsubsection*{JO-P1-C03-WS-001 (8 problems)}
\hypertarget{JO-P1-C03-WS-001}{}

\noindent\textbf{Navigation: }
\hyperlink{P1-JOHAR}{Back to Johar section}

\begin{enumerate}
\item \textbf{Density of rationals.} Prove that if $a<b$ are rational numbers, then there exists $q\in\mathbb{Q}$ such that $a<q<b$. (\hyperlink{JO-P1-C03-001}{proof})
\item \textbf{Countability of $\mathbb{Q}$.} Prove that $\mathbb{Q}$ is countable. (\hyperlink{JO-P1-C03-002}{proof})
\item \textbf{Uncountability of $(0,1)$ (diagonal).} Prove that $(0,1)$ is uncountable. (\hyperlink{JO-P1-C03-003}{proof})
\item \textbf{No rational square root of 2.} Prove that $\sqrt{2}\notin\mathbb{Q}$. (\hyperlink{JO-P1-C03-004}{proof})
\item \textbf{Supremum characterization.} Let $S\subseteq\mathbb{R}$ be nonempty and bounded above. Prove the $\varepsilon$-characterization of $\sup S$: for every $\varepsilon>0$ there exists $s\in S$ with $\sup S-\varepsilon < s \le \sup S$. (\hyperlink{JO-P1-C03-005}{proof})
\item \textbf{Dedekind cut downward closed.} Let $A\subseteq\mathbb{Q}$ be a Dedekind cut. Prove that if $r\in A$ and $q\in\mathbb{Q}$ with $q<r$, then $q\in A$. (\hyperlink{JO-P1-C03-006}{proof})
\item \textbf{No greatest element in a cut.} Let $A$ be a Dedekind cut. Prove that $A$ has no greatest element. (\hyperlink{JO-P1-C03-007}{proof})
\item \textbf{Cut corresponding to a rational.} For $r\in\mathbb{Q}$ define $A_r=\{q\in\mathbb{Q}: q<r\}$. Prove that $A_r$ is a Dedekind cut. (\hyperlink{JO-P1-C03-008}{proof})
\end{enumerate}
