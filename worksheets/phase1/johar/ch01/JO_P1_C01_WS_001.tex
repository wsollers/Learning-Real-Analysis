\subsubsection*{JO-P1-C01-WS-001 (8 problems)}
\hypertarget{JO-P1-C01-WS-001}{}

\noindent\textbf{Navigation: }
\hyperlink{P1-JOHAR}{Back to Johar section}

\begin{enumerate}
\item \textbf{Contrapositive equivalence.} Prove that $P \Rightarrow Q$ is logically equivalent to $\neg Q \Rightarrow \neg P$. (\hyperlink{JO-P1-C01-001}{proof})
\item \textbf{Negation of quantifiers.} Negate and simplify: $\forall x\in A\ \exists y\in B\, P(x,y)$. (\hyperlink{JO-P1-C01-002}{proof})
\item \textbf{De Morgan's laws.} Prove $(A\cup B)^c = A^c\cap B^c$ (relative to a fixed universe). (\hyperlink{JO-P1-C01-003}{proof})
\item \textbf{Set equality via double inclusion.} Prove $A=B$ if and only if $A\subseteq B$ and $B\subseteq A$. (\hyperlink{JO-P1-C01-004}{proof})
\item \textbf{Equivalence classes partition.} Let $\sim$ be an equivalence relation on a set $X$. Prove that the equivalence classes form a partition of $X$. (\hyperlink{JO-P1-C01-005}{proof})
\item \textbf{Intersecting classes are equal.} Let $\sim$ be an equivalence relation on $X$. Prove that if $[x]\cap [y]\neq\varnothing$ then $[x]=[y]$. (\hyperlink{JO-P1-C01-006}{proof})
\item \textbf{Preimage preserves unions.} Let $f:X\to Y$ and $E,F\subseteq Y$. Prove $f^{-1}(E\cup F)=f^{-1}(E)\cup f^{-1}(F)$. (\hyperlink{JO-P1-C01-007}{proof})
\item \textbf{Composition of injective functions.} Prove that if $f:X\to Y$ and $g:Y\to Z$ are injective, then $g\circ f$ is injective. (\hyperlink{JO-P1-C01-008}{proof})
\end{enumerate}
