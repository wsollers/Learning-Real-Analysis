\section*{Proof — RA-ROS-C04-E049}

\noindent
\hyperlink{ws-RA-ROS-C04-E049}{\textbf{← Back to worksheet}}

\noindent
\textbf{Source.}
\srccite{RossElementaryAnalysis}{Chapter~4, Exercise~4.9}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Complete the proof that $\inf S = -\sup(-S)$ (as in Corollary~4.5) by proving (1) and (2).

\vspace{0.75em}


\noindent
\textbf{Proof.}

Let $S \subseteq \mathbb{R}$ be nonempty and bounded below, and set
\[
s_0 := \inf S.
\]
Define
\[
- S := \{ -s \mid s \in S \}.
\]

\noindent\emph{Claim.} $\sup(-S) = -s_0$.

\smallskip
First we show that $-s_0$ is an upper bound for $-S$.
Since $s_0 = \inf S$, we have
\[
s_0 \le s \quad \text{for all } s \in S.
\]
Multiplying by $-1$ reverses the inequality, so
\[
-s \le -s_0 \quad \text{for all } s \in S,
\]
which means $-s_0$ is an upper bound of $-S$.

\smallskip
Now let $u$ be any upper bound of $-S$. Then
\[
-s \le u \quad \text{for all } s \in S.
\]
Multiplying by $-1$ gives
\[
s \ge -u \quad \text{for all } s \in S,
\]
so $-u$ is a lower bound of $S$. Because $s_0 = \inf S$ is the greatest
lower bound, we have $-u \le s_0$, and hence (multiplying by $-1$)
\[
u \ge -s_0.
\]
Therefore $-s_0$ is less than or equal to every upper bound of $-S$, so
\[
\sup(-S) = -s_0.
\]

Finally, multiplying both sides by $-1$ yields
\[
\inf S = s_0 = -\sup(-S).
\]
\qed
