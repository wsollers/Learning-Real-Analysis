\section*{Proof — RA-ROS-C04-E048B}

\noindent
\hyperlink{ws-RA-ROS-C04-E048B}{\textbf{← Back to worksheet}}

\noindent
\textbf{Source.}
\srccite{RossElementaryAnalysis}{Chapter~4, Exercise~4.8(b)}.

\vspace{0.75em}

\noindent
\textbf{Goal.}
Let $S$ and $T$ be nonempty subsets of $\mathbb{R}$ with the property that
$s\le t$ for all $s\in S$ and $t\in T$.
Observe that $S$ is bounded above and $T$ is bounded below. Prove that
\[
\sup S \le \inf T.
\]

\vspace{0.75em}

\noindent\textbf{Proof.}
Let $S$ and $T$ be nonempty subsets of $\mathbb{R}$ such that
$s \le t$ for all $s \in S$ and $t \in T$.

Let $t \in T$ be arbitrary. Then for every $s \in S$ we have $s \le t$.
Hence $t$ is an upper bound of $S$, and therefore $\sup S \le t$.

Since $t \in T$ was arbitrary, we have $\sup S \le t$ for all $t \in T$.
Thus $\sup S$ is a lower bound of $T$. By definition of infimum, it follows that
\[
\sup S \le \inf T.
\]
\qed
