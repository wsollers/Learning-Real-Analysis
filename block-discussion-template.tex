% block-discussion-template.tex
% Reusable Abbott-style Discussion/Motivation boxes (drop-in preamble snippet)
% Requires: \usepackage[most]{tcolorbox}

% ---- Discussion block (Abbott-style) ----
\newtcolorbox{discussion}[1][]{%
  enhanced,
  breakable,
  colback=gray!6,
  colframe=black!35,
  boxrule=0.6pt,
  arc=2mm,
  left=6pt,right=6pt,top=6pt,bottom=6pt,
  fonttitle=\bfseries,
  title={Discussion},
  #1
}

% ---- Motivation block (same style, different label) ----
\newtcolorbox{motivation}[1][]{%
  enhanced,
  breakable,
  colback=gray!6,
  colframe=black!35,
  boxrule=0.6pt,
  arc=2mm,
  left=6pt,right=6pt,top=6pt,bottom=6pt,
  fonttitle=\bfseries,
  title={Motivation},
  #1
}

% ---- Historical Note block (optional) ----
\newtcolorbox{historicalnote}[1][]{%
  enhanced,
  breakable,
  colback=gray!6,
  colframe=black!35,
  boxrule=0.6pt,
  arc=2mm,
  left=6pt,right=6pt,top=6pt,bottom=6pt,
  fonttitle=\bfseries,
  title={Historical Note},
  #1
}

% -------------------------
% Usage examples:
%
% \begin{discussion}
% Text...
% \end{discussion}
%
% \begin{discussion}[title={Discussion: The Least Upper Bound Property}]
% Text...
% \end{discussion}
%
% \begin{motivation}
% Text...
% \end{motivation}
%
% Important:
% In your main preamble, include:
% \usepackage[most]{tcolorbox}
% -------------------------
