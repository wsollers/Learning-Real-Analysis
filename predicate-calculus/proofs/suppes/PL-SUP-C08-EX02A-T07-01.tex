\noindent
\textbf{Goal.} Prove $(\forall x)(0 - x = -x)$.

\vspace{0.75em}

\vspace{0.75em}

\noindent
\textbf{Proof (stub).}
\[
\begin{array}{r l l}
(1) & \text{Let a be arbitrary} &  \\

(2) & \text{0 - a = 0 + (-a)} & \text{Definition of Subtraction} \\
(3) & \text{0 + (-a) = -a} & \text{Identity} \\
(4) & \text{0 - a = -a} & \text{Transitivity} \\
(5) & \forall x (0-x = -x) & \text{Generalization} \\
\end{array}
\]
