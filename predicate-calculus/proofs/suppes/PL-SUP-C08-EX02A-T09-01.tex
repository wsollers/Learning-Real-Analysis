\noindent
\textbf{Goal.} Prove $(\forall x)(\forall y)((-x=y)\rightarrow(x=-y))$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\[
\begin{array}{r l l}
(1) & \text{Let $x,y$ be arbitrary.} &  \\
(2) & \text{Assume $-x = y$.} & \text{Assumption} \\
(3) & -x + (-y) = y + (-y) & \text{Substitution of equals} \\
(4) & (-x + (-y)) + x = (y + (-y)) + x & \text{Substitution of equals} \\
(5) & (-x + x) + (-y) = (y + (-y)) + x & \text{Associativity} \\
(6) & 0 + (-y) = 0 + x & \text{Additive inverse} \\
(7) & -y = x & \text{Identity} \\
(8) & x = -y & \text{Symmetry of $=$} \\
(9) & (-x = y) \rightarrow (x = -y) & \text{Conditional proof} \\
(10) & (\forall x)(\forall y)((-x=y)\rightarrow(x=-y)) & \text{Universal Generalization}
\end{array}
\]
