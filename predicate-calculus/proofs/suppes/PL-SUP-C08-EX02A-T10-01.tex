\noindent
\textbf{Goal.} Prove $(\forall x)(\forall y)(\forall z)((x+y=z+y)\rightarrow(x=z))$.

\vspace{0.75em}

\noindent
\textbf{Proof.}
\[
\begin{array}{r l l}
(1) & \text{Let $x,y,z$ be arbitrary.} &  \\
(2) & \text{Assume $x+y=z+y$.} & \text{Assumption} \\
(3) & (x+y)+(-y)=(z+y)+(-y) & \text{Substitution of equals} \\
(4) & x+(y+(-y))=z+(y+(-y)) & \text{Associativity} \\
(5) & x+0=z+0 & \text{Additive inverse} \\
(6) & x=z & \text{Identity} \\
(7) & (x+y=z+y)\rightarrow(x=z) & \text{Conditional proof} \\
(8) & (\forall x)(\forall y)(\forall z)\bigl((x+y=z+y)\rightarrow(x=z)\bigr) & \forall\text{-Gen}
\end{array}
\]
