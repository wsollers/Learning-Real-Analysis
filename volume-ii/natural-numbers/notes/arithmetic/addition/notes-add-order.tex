% Def 2.2.11
% ---------------------------------------------------------
\begin{definition}[Tao 2.2.11 — Order on $\mathbb{N}$]
\label{def:order}
Let $n, m \in \mathbb{N}$.
\[
n \geq m \quad (\text{equivalently } m \leq n)
\quad \iff \quad
\exists\, a \in \mathbb{N},\; n = m + a.
\]
\[
n > m \quad (\text{equivalently } m < n)
\quad \iff \quad
n \geq m \text{ and } n \neq m.
\]
\end{definition}

\begin{remark}[Unpacking $\geq$]
$n \geq m$ is an \emph{existence} claim: you must exhibit a witness $a$.
$n > m$ adds a \emph{negation}: the witness $a$ must be nonzero,
i.e., positive.
Always unpack the definition before writing any proof involving order.
\end{remark}

% P2.2.12
% ---------------------------------------------------------
\begin{proposition}[Tao 2.2.12 — Basic properties of order]
\label{prop:order-basic}
Let $a, b, c \in \mathbb{N}$.
\begin{enumerate}
  \item[(a)] \textbf{Reflexive.} $a \geq a$.
  \item[(b)] \textbf{Transitive.} If $a \geq b$ and $b \geq c$, then $a \geq c$.
  \item[(c)] \textbf{Anti-symmetric.} If $a \geq b$ and $b \geq a$, then $a = b$.
  \item[(d)] \textbf{Addition preserves order.}
    $a \geq b \iff a + c \geq b + c$.
  \item[(e)] $a < b \iff a\pp \leq b$.
  \item[(f)] $a < b \iff b = a + d$ for some positive $d$.
\end{enumerate}
\end{proposition}

\begin{remark}[Strategy for each part]
\begin{itemize}
  \item (a): existence proof. Witness is immediate from L2.2.2.
  \item (b): combine two witnesses using P2.2.5 to get a new witness.
  \item (c): combine witnesses to get $a = a + (n+m)$,
    cancel via P2.2.6, then apply C2.2.9 to conclude $n = m = 0$.
  \item (d): biconditional --- two directions, each an existence argument
    plus P2.2.5/P2.2.4 for regrouping, P2.2.6 for the backward direction.
  \item (e): forward direction uses L2.2.10 to extract a predecessor,
    then L2.2.3 and A2 to rewrite. Backward direction: exhibit witness
    and invoke C2.2.9 and Def 2.2.7.
  \item (f): unpacks $<$ directly using Def 2.2.7.
\end{itemize}
\end{remark}

\begin{proposition}[Tao 2.2.13 — Trichotomy]
\label{prop:trichotomy}
For all $a, b \in \mathbb{N}$, exactly one of the following holds:
\[
a < b, \qquad a = b, \qquad a > b.
\]
\end{proposition}

\begin{remark}[Two obligations]
Must show: (i) at most one holds (mutual exclusion),
and (ii) at least one holds (exhaustion by induction on $a$).
\end{remark}
