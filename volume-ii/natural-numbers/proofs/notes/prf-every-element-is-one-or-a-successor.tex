\begin{theorem}[Every Element Is Either $1$ or a Successor]
\label{thm:every-element-is-one-or-a-successor}
Let $(P,S,1)$ be a Peano system. For every $x \in P$, either $x=1$ or there
exists $u \in P$ such that $S(u)=x$.
\end{theorem}
