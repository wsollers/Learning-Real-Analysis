\begin{theorem}[Iterator Successor Step]
\label{thm:iterator-successor-step}
Let $(P,S,1)$ be a Peano system, let $X$ be a set, and let $(a,T)$ be iterator
data on $X$. If $f$ is the iterator-generated function determined by $(a,T)$,
then
\[
\forall n \in P\;(f(S(n))=T(f(n))).
\]
\end{theorem}
