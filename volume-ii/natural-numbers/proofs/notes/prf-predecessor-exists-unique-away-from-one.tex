\begin{theorem}[Predecessor Existence and Uniqueness Away from $1$]
\label{thm:predecessor-exists-unique-away-from-one}
Let $(P,S,1)$ be a Peano system, and let $x \in P$ with $x \neq 1$. Then
there exists a unique $u \in P$ such that $S(u)=x$.
\end{theorem}
