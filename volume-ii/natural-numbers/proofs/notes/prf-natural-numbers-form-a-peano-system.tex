\begin{theorem*}[The Natural Numbers Form a Peano System]
There exists a function $S_{\mathbb{N}}:\mathbb{N}\to\mathbb{N}$ such that
$(\mathbb{N},S_{\mathbb{N}},1)$ is a Peano system.
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
By the Existence of a Peano System axiom, there exists a Peano system
$(P,S,1)$. Define
\[
\mathbb{N}:=P
\qquad\text{and}\qquad
S_{\mathbb{N}}:=S.
\]
Since $(P,S,1)$ is a Peano system, this change of notation gives that
$(\mathbb{N},S_{\mathbb{N}},1)$ is a Peano system. In particular,
$S_{\mathbb{N}}:\mathbb{N}\to\mathbb{N}$.
\end{proof}

\begin{proof}
\textbf{Detailed Learning Proof.}~\\
The Existence of a Peano System axiom says that at least one triple
$(P,S,1)$ satisfies the Peano system axioms. Thus we may choose such a
triple.

\textbf{Step 1. Name the chosen underlying set.}
We define
\[
\mathbb{N}:=P.
\]
This is a choice of notation: the symbol $\mathbb{N}$ now denotes the
underlying set of the chosen Peano system.

\textbf{Step 2. Name the chosen successor map.}
We also define
\[
S_{\mathbb{N}}:=S.
\]
Since $S:P\to P$ and $\mathbb{N}=P$, it follows that
\[
S_{\mathbb{N}}:\mathbb{N}\to\mathbb{N}.
\]

\textbf{Step 3. Transfer the Peano-system property through the notation.}
Because the chosen triple $(P,S,1)$ is a Peano system, replacing $P$ by its
new name $\mathbb{N}$ and replacing $S$ by its new name $S_{\mathbb{N}}$
gives
\[
(\mathbb{N},S_{\mathbb{N}},1)\text{ is a Peano system}.
\]
Therefore the natural numbers, equipped with their successor function, form a
Peano system.
\end{proof}

\begin{remark*}[Proof structure]
The proof is a naming argument. The existence axiom supplies an abstract
Peano system, and the theorem fixes the conventional notation
$\mathbb{N}$ for its underlying set.
\end{remark*}
