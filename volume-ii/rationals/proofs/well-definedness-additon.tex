\subsection*{Addition is Well-Defined}
\label{prf:q-addition-well-defined}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:q-addition-well-defined]{Back to Theorem in Notes}
\end{remark*}

\begin{proof}

\Claim If
\[
(a,b) \sim (a',b')
\quad \text{and} \quad
(c,d) \sim (c',d'),
\]
then
\[
(ad+bc,\,bd) \sim (a'd'+b'c',\,b'd').
\]

\Given $(a,b),(a',b'),(c,d),(c',d') \in W$ such that
\[
ab' = a'b
\quad\text{and}\quad
cd' = c'd.
\]

\Goal To show
\[
(ad+bc)(b'd') = (a'd'+b'c')(bd).
\]

\Strategy
Use the relations $ab' = a'b$ and $cd' = c'd$ to replace cross-products.
Expand both sides and regroup terms until they match.

\end{proof}

\begin{remark*}[Proof shape]
Expand both products.  
Use $ab' = a'b$ and $cd' = c'd$ to substitute equivalent expressions.  
After regrouping terms the two sides coincide.
\end{remark*}
