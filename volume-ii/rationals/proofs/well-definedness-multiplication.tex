\subsection*{Multiplication is Well-Defined}
\label{prf:q-multiplication-well-defined}

\begin{remark*}[Return]
$\leftarrow$ \hyperref[thm:q-multiplication-well-defined]{Back to Theorem in Notes}
\end{remark*}

\begin{proof}

\Claim If
\[
(a,b) \sim (a',b')
\quad \text{and} \quad
(c,d) \sim (c',d'),
\]
then
\[
(ac,\,bd) \sim (a'c',\,b'd').
\]

\Given $(a,b),(a',b'),(c,d),(c',d') \in W$ such that
\[
ab' = a'b
\quad\text{and}\quad
cd' = c'd.
\]

\Goal To show
\[
(ac)(b'd') = (a'c')(bd).
\]

\Strategy
Multiply the equalities $ab' = a'b$ and $cd' = c'd$ together
and rearrange the resulting products.
