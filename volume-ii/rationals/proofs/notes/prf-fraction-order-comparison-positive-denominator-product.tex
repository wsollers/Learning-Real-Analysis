\begin{theorem*}[Fraction Comparison When the Denominator Product Is Positive]
For any $a,b,c,d\in F$, if $b\neq 0$, $d\neq 0$, and $0<b\cdot d$, then
\[
\frac{a}{b}<\frac{c}{d}
\quad\Longleftrightarrow\quad
a\cdot d<b\cdot c.
\]
\end{theorem*}
