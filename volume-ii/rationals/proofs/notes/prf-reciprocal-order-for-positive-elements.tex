\begin{theorem*}[Reciprocal Order for Positive Elements]
If $0<d<b$, then
\[
0<\frac{1}{b}<\frac{1}{d}.
\]
Conversely, if
\[
0<\frac{1}{b}<\frac{1}{d},
\]
then $0<d<b$.
\end{theorem*}
