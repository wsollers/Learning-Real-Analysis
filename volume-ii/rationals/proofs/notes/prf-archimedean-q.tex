\begin{theorem*}[Archimedean Property of $\mathbb{Q}$]
For every $x\in\mathbb{Q}$ there exists $n\in\mathbb{N}$ such that
\[
n>x.
\]
Equivalently, for every $x>0$ there exists $n\in\mathbb{N}$ such that
\[
\frac{1}{n}<x.
\]
\end{theorem*}
