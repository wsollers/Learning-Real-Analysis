\begin{theorem*}[General Fraction Comparison in an Ordered Field]
For any $a,b,c,d\in F$ with $b\neq 0$ and $d\neq 0$,
\[
\frac{a}{b}<\frac{c}{d}
\quad\Longleftrightarrow\quad
(a\cdot d)\cdot(b\cdot d)<(b\cdot c)\cdot(b\cdot d).
\]
\end{theorem*}
