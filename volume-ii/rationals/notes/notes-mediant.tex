% =========================================================
% The Mediant Property
% =========================================================

\subsection{The Mediant Property}

% ---------------------------------------------------------
% Definition
% ---------------------------------------------------------

\begin{definition}[Mediant of Two Fractions]
\label{def:mediant}

Let

\[
\frac{a}{b},\frac{c}{d}\in\mathbb{Q}
\quad
(b,d>0).
\]

The \emph{mediant} of these fractions is the rational number

\[
\frac{a+c}{b+d}.
\]

\end{definition}

% ---------------------------------------------------------
% Teacher exposition
% ---------------------------------------------------------

\begin{remark}[Teacher exposition]
\label{rem:mediant-teacher}

The mediant is obtained by adding the numerators and denominators
separately.

\[
\frac{a}{b},\frac{c}{d}
\quad\longrightarrow\quad
\frac{a+c}{b+d}.
\]

This operation is surprisingly powerful.  It produces a rational number
that lies between the original two fractions whenever

\[
\frac{a}{b} < \frac{c}{d}.
\]

Thus the mediant provides a simple mechanism for generating new rational
numbers between existing ones.

\end{remark}

% ---------------------------------------------------------
% Theorem
% ---------------------------------------------------------

\begin{theorem}[Mediant Inequality]
\label{thm:mediant-inequality}

Let

\[
\frac{a}{b},\frac{c}{d}\in\mathbb{Q}
\quad
(b,d>0)
\]

with

\[
\frac{a}{b} < \frac{c}{d}.
\]

Then the mediant satisfies

\[
\frac{a}{b}
<
\frac{a+c}{b+d}
<
\frac{c}{d}.
\]

\end{theorem}

% ---------------------------------------------------------

\begin{remark}[Proof idea]
\label{rem:mediant-proof-idea}

The inequalities follow from cross multiplication.

From

\[
\frac{a}{b} < \frac{c}{d}
\]

we obtain

\[
ad < bc.
\]

Simple algebraic manipulations then show that

\[
a(b+d) < b(a+c)
\]

and

\[
(a+c)d < (b+d)c.
\]

These inequalities yield the desired result.

\end{remark}

% ---------------------------------------------------------
% Corollary
% ---------------------------------------------------------

\begin{corollary}[Density of the Rational Numbers]
\label{cor:q-density}

For any rational numbers

\[
r,s\in\mathbb{Q}
\quad
(r<s)
\]

there exists another rational number $t$ such that

\[
r < t < s.
\]

\end{corollary}

\begin{remark}

One such number is the mediant.

Thus the rational numbers contain infinitely many numbers between any
two distinct rational numbers.

\end{remark}

% ---------------------------------------------------------
% Iteration
% ---------------------------------------------------------

\begin{remark}[Iterated Mediant Construction]
\label{rem:mediant-iteration}

The mediant can be applied repeatedly.

Starting with

\[
\frac{a}{b},\frac{c}{d}
\]

we obtain

\[
\frac{a+c}{b+d}.
\]

Applying the mediant again produces further fractions between them.
This process generates infinitely many rational numbers within the
interval.

\end{remark}

% ---------------------------------------------------------
% Structural significance
% ---------------------------------------------------------

\begin{remark}[Structural importance]
\label{rem:mediant-structure}

The mediant operation reveals an important structural feature of the
rational numbers: they are \emph{dense} and can be refined indefinitely.

This property plays a central role in several important constructions,
including

\begin{itemize}
\item Farey sequences,
\item the Stern--Brocot tree,
\item continued fractions,
\item rational approximations to real numbers.
\end{itemize}

\end{remark}

% ---------------------------------------------------------
% Big picture
% ---------------------------------------------------------

\begin{remark}[Conceptual interpretation]
\label{rem:mediant-concept}

The mediant operation provides a way to bisect rational intervals.

Thus it functions as a constructive tool for generating rational
numbers and for refining approximations.

In this sense the mediant property plays a role in $\mathbb{Q}$ similar
to that of the Archimedean property: it guarantees that rational
intervals can always be refined by inserting additional rational
numbers.

\end{remark}
