% =========================================================
% Dirichlet Approximation – Pigeonhole Diagram
% =========================================================

\begin{figure}[h]
\centering
\begin{tikzpicture}[x=1.0cm, y=1.0cm]

% ------------------------------------------------------------------
% Unit interval  [0, 1)
% ------------------------------------------------------------------
\draw[thick] (0,0) -- (10,0);
\node[below=4pt] at (0,0)  {$0$};
\node[below=4pt] at (10,0) {$1$};

% N = 9 dividing ticks (10 equal subintervals)
\foreach \x in {1,2,3,4,5,6,7,8,9} {
  \draw (\x, 0.15) -- (\x, -0.15);
}

% ------------------------------------------------------------------
% Caption below line
% ------------------------------------------------------------------
\node[font=\small] at (5, -0.9)
  {$[0,1)$ divided into $N$ equal subintervals};

% ------------------------------------------------------------------
% Fractional parts  (dots at y=0.5, labels at y=1.05)
%
% Spacing rules:
%   {x}   at 1.3   — alone in its interval [1,2)
%   {2x}  at 3.35  — in interval [3,4),  left side
%   {3x}  at 3.75  — in interval [3,4),  right side   <- PIGEONHOLE PAIR
%   {4x}  at 6.2   — alone in [6,7)
%   {5x}  at 7.7   — alone in [7,8)
%   {6x}  at 9.1   — alone in [9,10)
% ------------------------------------------------------------------

% Lone points
\fill (1.3, 0.5) circle (2.2pt);
\fill (6.2, 0.5) circle (2.2pt);
\fill (7.7, 0.5) circle (2.2pt);
\fill (9.1, 0.5) circle (2.2pt);

% Labels for lone points
\node[above=2pt, font=\small] at (1.3,  0.5) {$\{x\}$};
\node[above=2pt, font=\small] at (6.2,  0.5) {$\{4x\}$};
\node[above=2pt, font=\small] at (7.7,  0.5) {$\{5x\}$};
\node[above=2pt, font=\small] at (9.1,  0.5) {$\{6x\}$};

% ------------------------------------------------------------------
% Pigeonhole pair  — well separated inside [3,4)
% ------------------------------------------------------------------
\fill[red!80!black] (3.35, 0.5) circle (2.2pt);
\fill[red!80!black] (3.78, 0.5) circle (2.2pt);

% Labels sit above, offset left/right so they don't overlap
\node[above=2pt, font=\small, red!80!black] at (3.15, 0.5) {$\{2x\}$};
\node[above=2pt, font=\small, red!80!black] at (3.98, 0.5) {$\{3x\}$};

% Highlight box around the crowded interval
\draw[red, thick, rounded corners=1pt]
  (3.0, 0.18) rectangle (4.0, 0.82);

% Annotation above the box — on its own line well clear of labels
\draw[red, ->, thin] (3.5, 1.45) -- (3.5, 0.85);
\node[red, font=\small, align=center] at (3.5, 1.65)
  {two points in the\\ same subinterval};

% ------------------------------------------------------------------
% Gap arrow between the two pigeonholed points
% ------------------------------------------------------------------
\draw[<->, red!60, thin] (3.35, 0.22) -- (3.78, 0.22);
\node[below=1pt, red!60, font=\scriptsize] at (3.565, 0.22)
  {$< \tfrac{1}{N}$};

\end{tikzpicture}
\caption{Dirichlet's theorem rests on a simple geometric idea: too many fractional parts are forced
into too few subintervals.  The resulting close pair yields a strong rational approximation.}
\label{fig:dirichlet-pigeonhole}
\end{figure}
