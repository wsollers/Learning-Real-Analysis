\subsection{Rational Approximation}

% ---------------------------------------------------------
% Teacher exposition
% ---------------------------------------------------------

\begin{remark}[Teacher exposition]
\label{rem:rational-approx-teacher}

One of the most important properties of the rational numbers is that
they can approximate other numbers arbitrarily well.

Although many numbers (such as $\sqrt{2}$ or $\pi$) are not rational,
we can find rational numbers that lie extremely close to them.

For example,

\[
1.4,\quad 1.41,\quad 1.414,\quad 1.4142
\]

are rational numbers that approximate $\sqrt{2}$ with increasing
accuracy.

This phenomenon reflects a fundamental structural property of the
rational numbers: they are \emph{dense} in the real line.

\end{remark}

% ---------------------------------------------------------
% Density theorem
% ---------------------------------------------------------

\begin{theorem}[Density of the Rational Numbers]
\label{thm:q-dense}

For any real numbers $x$ and $y$ with

\[
x < y
\]

there exists a rational number $r$ such that

\[
x < r < y.
\]

\end{theorem}

% ---------------------------------------------------------

\begin{remark}

Thus between any two numbers on the real line there lies a rational
number.

In fact, there are infinitely many rational numbers in every interval.

\end{remark}

% ---------------------------------------------------------
% Stronger approximation theorem
% ---------------------------------------------------------

\begin{theorem}[Rational Approximation]
\label{thm:rational-approx}

For every real number $x$ and every $\varepsilon > 0$ there exists a
rational number $r$ such that

\[
|x-r| < \varepsilon.
\]

\end{theorem}

% ---------------------------------------------------------

\begin{remark}[Interpretation]

This theorem says that rational numbers can approximate any real number
with arbitrary precision.

No matter how small the tolerance $\varepsilon$ is chosen, a rational
number can always be found within that distance.

\end{remark}

% ---------------------------------------------------------
% Constructive methods
% ---------------------------------------------------------

\begin{remark}[Constructive approximations]

There are several systematic ways to produce rational approximations.

\textbf{Decimal truncation}

If

\[
x = 1.4142135\dots
\]

then

\[
1.4,\quad 1.41,\quad 1.414,\quad 1.4142
\]

are rational approximations.

\medskip

\textbf{Mediants}

Given two rational bounds

\[
\frac{a}{b} < x < \frac{c}{d}
\]

the mediant

\[
\frac{a+c}{b+d}
\]

provides a new rational approximation between them.

\medskip

\textbf{Continued fractions}

Continued fractions produce extremely efficient rational approximations
to real numbers.

\end{remark}

% ---------------------------------------------------------
% Sequences
% ---------------------------------------------------------

\begin{remark}[Approximation by sequences]

A powerful way to approximate numbers is by sequences of rational
numbers.

For example,

\[
1,\;
1.4,\;
1.41,\;
1.414,\;
1.4142,\dots
\]

is a sequence of rational numbers converging to $\sqrt{2}$.

Such sequences motivate the construction of the real numbers as limits
of rational sequences.

\end{remark}

% ---------------------------------------------------------
% Big picture
% ---------------------------------------------------------

\begin{remark}[Conceptual picture]

The rational numbers form a dense but incomplete number system.

They are dense because rational numbers appear everywhere along the
real line.

They are incomplete because certain limits of rational sequences are
not rational numbers.

This dual nature of $\mathbb{Q}$ leads naturally to the construction of
the real numbers.

\end{remark}

\subsection{Dirichlet's Approximation Theorem}

% ---------------------------------------------------------
% Teacher exposition
% ---------------------------------------------------------

\begin{remark}[Teacher exposition]
\label{rem:dirichlet-teacher}

The density of the rational numbers tells us that rational numbers
exist arbitrarily close to any real number.  However, density alone
does not tell us \emph{how close} these approximations can be.

Dirichlet's Approximation Theorem gives a powerful quantitative
statement: every real number can be approximated by rational numbers
with an error that decreases roughly like the reciprocal of the square
of the denominator.

This result shows that rational numbers approximate real numbers far
more efficiently than the density property alone would suggest.

\end{remark}

% ---------------------------------------------------------
% Theorem
% ---------------------------------------------------------

\begin{theorem}[Dirichlet's Approximation Theorem]
\label{thm:dirichlet-approx}

Let $x \in \mathbb{R}$.  Then there exist infinitely many rational
numbers $\frac{p}{q}$ with $p \in \mathbb{Z}$ and $q \in \mathbb{N}$
such that

\[
\left|x - \frac{p}{q}\right| < \frac{1}{q^2}.
\]

\end{theorem}

% ---------------------------------------------------------
% Interpretation
% ---------------------------------------------------------

\begin{remark}[Interpretation]
\label{rem:dirichlet-interpret}

The inequality

\[
\left|x - \frac{p}{q}\right| < \frac{1}{q^2}
\]

means that the approximation error decreases very rapidly as the
denominator $q$ increases.

For example, if $q=100$, the error is guaranteed to be less than

\[
\frac{1}{100^2} = 10^{-4}.
\]

Thus rational numbers with moderately large denominators provide
extremely accurate approximations to real numbers.

\end{remark}

% ---------------------------------------------------------
% Proof idea
% ---------------------------------------------------------

\begin{remark}[Idea of the proof]
\label{rem:dirichlet-proof-idea}

The proof uses the pigeonhole principle.

Consider the fractional parts of the numbers

\[
x,\;2x,\;3x,\;\dots,\;(N+1)x.
\]

These $N+1$ numbers lie in the interval $[0,1)$.

Divide this interval into $N$ equal subintervals.  Since there are
$N+1$ numbers but only $N$ intervals, two of these fractional parts
must lie in the same interval.

Taking the difference of these two numbers produces integers $p$ and
$q$ such that

\[
\left|x - \frac{p}{q}\right| < \frac{1}{q^2}.
\]

\end{remark}

% ---------------------------------------------------------
% Example
% ---------------------------------------------------------

\begin{remark}[Example]

The number $\pi$ has many excellent rational approximations.

For example,

\[
\frac{22}{7} \approx 3.142857
\]

and

\[
\frac{355}{113} \approx 3.14159292.
\]

The approximation

\[
\frac{355}{113}
\]

is remarkably accurate and satisfies the type of bound predicted by
Dirichlet's theorem.

\end{remark}

% ---------------------------------------------------------
% Conceptual significance
% ---------------------------------------------------------

\begin{remark}[Conceptual significance]
\label{rem:dirichlet-significance}

Dirichlet's theorem reveals that the rational numbers form an
extremely rich approximation system for the real line.

While the rationals are countable, they nevertheless approximate every
real number with arbitrarily high precision.

This result marks the beginning of the field known as
\emph{Diophantine approximation}, which studies how well real numbers
can be approximated by rational numbers.

\end{remark}
