\subsubsection[Remaining Ring Laws (Ex 4.1.4)]{Proof --- RA-TAO-C04-S4.1-E04}

\bigskip

\noindent\textbf{Source.}
\srccite{TaoAnalysis1}{Chapter~4, \S4.1, Exercise~4.1.4}.

\vspace{0.75em}

\noindent\textbf{Goal.}
Prove the remaining identities in Proposition~4.1.6 not established
in the text.
Tao proves associativity of multiplication $(xy)z = x(yz)$ as a model.
The remaining identities to prove are:

\medskip
\begin{enumerate}
  \item[(i)] $x + y = y + x$ \quad (commutativity of addition)
  \item[(ii)] $(x+y)+z = x+(y+z)$ \quad (associativity of addition)
  \item[(iii)] $x + 0 = 0 + x = x$ \quad (additive identity)
  \item[(iv)] $x + (-x) = (-x) + x = 0$ \quad (additive inverse)
  \item[(v)] $xy = yx$ \quad (commutativity of multiplication)
  \item[(vi)] $x \cdot 1 = 1 \cdot x = x$ \quad (multiplicative identity --- follows from (v) once proved)
  \item[(vii)] $x(y+z) = xy + xz$ \quad (left distributive law)
  \item[(viii)] $(y+z)x = yx + zx$ \quad (right distributive law --- follows from (v) and (vii))
\end{enumerate}

\vspace{0.75em}

\noindent\textbf{Logical form.}
$\forall x, y, z \in \mathbb{Z}$: each identity above holds.

\vspace{0.75em}

\noindent\textbf{Available toolkit.}
Definition~4.1.2 (addition and multiplication formulas),
Definition~4.1.1 (equality),
ring laws for $\mathbb{N}$ (P2.2.4, P2.2.5, P2.3.4, P2.3.5, etc.),
write $x = (a{-\!\!-}b)$, $y = (c{-\!\!-}d)$, $z = (e{-\!\!-}f)$.
