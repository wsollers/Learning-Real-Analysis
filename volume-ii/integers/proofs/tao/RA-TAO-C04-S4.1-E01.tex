\subsubsection[Equality on Z: Reflexivity and Symmetry (Ex 4.1.1)]{Proof --- RA-TAO-C04-S4.1-E01}

\bigskip

\noindent\textbf{Source.}
\srccite{TaoAnalysis1}{Chapter~4, \S4.1, Exercise~4.1.1}.

\vspace{0.75em}

\noindent\textbf{Goal.}
Verify that the definition of equality on $\mathbb{Z}$ is both
reflexive and symmetric.

\medskip
\noindent\textbf{Recall.} Two integers $a {-\!\!-} b$ and $c {-\!\!-} d$
are equal iff $a + d = c + b$.

\medskip
\begin{enumerate}
  \item[\textbf{(i)}] \textbf{Reflexivity.}
    For all $a, b \in \mathbb{N}$: $(a {-\!\!-} b) = (a {-\!\!-} b)$.
  \item[\textbf{(ii)}] \textbf{Symmetry.}
    For all $a, b, c, d \in \mathbb{N}$: if $(a {-\!\!-} b) = (c {-\!\!-} d)$
    then $(c {-\!\!-} d) = (a {-\!\!-} b)$.
\end{enumerate}

\vspace{0.75em}

\noindent\textbf{Logical form.}
\begin{enumerate}
  \item[(i)] $\forall a, b \in \mathbb{N}:\; a + b = a + b$.
  \item[(ii)] $\forall a,b,c,d \in \mathbb{N}:\; a+d = c+b \Rightarrow c+b = a+d$.
\end{enumerate}

\vspace{0.75em}

\noindent\textbf{Available toolkit.}
Definition~4.1.1 (equality of integers),
properties of equality in $\mathbb{N}$ (reflexivity, symmetry of $=$).
