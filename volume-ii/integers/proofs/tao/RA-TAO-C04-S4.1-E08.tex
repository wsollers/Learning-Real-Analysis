\subsubsection[Induction Fails for Z (Ex 4.1.8)]{Proof --- RA-TAO-C04-S4.1-E08}

\bigskip

\noindent\textbf{Source.}
\srccite{TaoAnalysis1}{Chapter~4, \S4.1, Exercise~4.1.8}.

\vspace{0.75em}

\noindent\textbf{Goal.}
Show that the principle of induction (Axiom~P5) does not apply
directly to the integers.

\medskip
More precisely: exhibit a property $P(n)$ pertaining to integers
such that:
\begin{enumerate}
  \item[(i)] $P(0)$ is true,
  \item[(ii)] $P(n) \Rightarrow P(n\pp)$ for all integers $n$,
  \item[(iii)] yet $P(n)$ is \emph{not} true for all integers $n$.
\end{enumerate}

\vspace{0.75em}

\noindent\textbf{Logical form.}
$\exists P:\; P(0) \wedge [\forall n \in \mathbb{Z},\; P(n) \Rightarrow P(n\pp)]
\wedge [\exists n \in \mathbb{Z},\; \neg P(n)]$.

\vspace{0.75em}

\noindent\textbf{Available toolkit.}
Recall $n\pp := n + 1$ for any integer $n$.
The negative integers $-1, -2, -3, \ldots$ are in $\mathbb{Z}$.
Knowledge that $\mathbb{Z}$ has no smallest element.

\vspace{0.75em}

\noindent\textbf{Task.}
This is a \emph{counterexample construction}, not a deductive proof.
Name the property $P$, verify conditions (i)--(iii) explicitly.

\vspace{0.75em}

\noindent\textbf{Strategy.}
(Fill in before beginning --- what kind of property might be true
for all non-negative integers but fail for some negative integer?)
